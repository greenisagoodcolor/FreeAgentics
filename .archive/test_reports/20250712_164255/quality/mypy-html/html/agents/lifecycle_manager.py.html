"""Agent Lifecycle Manager for FreeAgentics.

This module provides comprehensive lifecycle management for agents including:
- Creation, initialization, and cleanup hooks
- Resource tracking and limits
- Automatic cleanup of inactive agents
- Memory pressure monitoring
- Graceful shutdown procedures
"""

import asyncio
import gc
import logging
import os
import time
import weakref
from contextlib import contextmanager
from dataclasses import dataclass, field
from typing import Any, Callable, Dict, List, Optional

import psutil

logger = logging.getLogger(__name__)


@dataclass
class AgentLifecycleEvent:
    """Represents a lifecycle event for an agent."""

    agent_id: str
    event_type: str  # 'created', 'started', 'stopped', 'cleaned', 'error'
    timestamp: float
    details: Dict[str, Any] = field(default_factory=dict)


@dataclass
class AgentResourceUsage:
    """Tracks resource usage for an individual agent."""

    agent_id: str
    memory_mb: float
    cpu_percent: float
    last_active: float
    operation_count: int = 0
    error_count: int = 0


class AgentLifecycleManager:
    """Manages agent lifecycle including creation, cleanup, and resource monitoring."""

    def __init__(
        self, max_agents=100, max_memory_mb=1024, inactive_timeout=3600, cleanup_interval=300
    ):
        """Initialize the lifecycle manager.

        Args:
            max_agents: Maximum number of concurrent agents
            max_memory_mb: Maximum total memory usage in MB
            inactive_timeout: Seconds before inactive agent cleanup
            cleanup_interval: Seconds between cleanup cycles
        """
        self.max_agents = max_agents
        self.max_memory_mb = max_memory_mb
        self.inactive_timeout = inactive_timeout
        self.cleanup_interval = cleanup_interval

        # Agent tracking
        self._agents: Dict[str, weakref.ref] = {}
        self._agent_metadata: Dict[str, Dict[str, Any]] = {}
        self._resource_usage: Dict[str, AgentResourceUsage] = {}
        self._lifecycle_events: List[AgentLifecycleEvent] = []

        # Lifecycle hooks
        self._on_create_hooks: List[Callable] = []
        self._on_start_hooks: List[Callable] = []
        self._on_stop_hooks: List[Callable] = []
        self._on_cleanup_hooks: List[Callable] = []

        # Cleanup task
        self._cleanup_task: Optional[asyncio.Task] = None
        self._running = False

        # Memory pressure monitoring
        self._memory_pressure_threshold = 0.9  # 90% of max memory
        self._under_memory_pressure = False

    def register_agent(self, agent: Any, metadata: Optional[Dict[str, Any]] = None) -> bool:
        """Register an agent with the lifecycle manager.

        Args:
            agent: The agent instance
            metadata: Optional metadata about the agent

        Returns:
            True if registered successfully, False otherwise
        """
        agent_id = getattr(agent, "agent_id", str(id(agent)))

        # Check limits:
        if len(self._agents) >= self.max_agents:
            logger.warning(f"Cannot register agent {agent_id}: max agents limit reached")
            return False

        # Check memory pressure
        if self._under_memory_pressure:
            logger.warning(f"Cannot register agent {agent_id}: system under memory pressure")
            return False

        # Register agent
        self._agents[agent_id] = weakref.ref(agent, self._create_cleanup_callback(agent_id))
        self._agent_metadata[agent_id] = metadata or {}
        self._resource_usage[agent_id] = AgentResourceUsage(
            agent_id=agent_id, memory_mb=0, cpu_percent=0, last_active=time.time()
        )

        # Record event
        self._record_event(agent_id, "created", {"metadata": metadata})

        # Run creation hooks
        for hook in self._on_create_hooks:
            try:
                hook(agent, metadata)
            except Exception as e:
                logger.error(f"Error in creation hook: {e}")

        return True

    def unregister_agent(self, agent_id: str) -> bool:
        """Unregister an agent from lifecycle management.

        Args:
            agent_id: The agent ID
        """
        if agent_id not in self._agents:
            return False

        # Get agent reference
        agent_ref = self._agents.get(agent_id)
        if agent_ref:
            agent = agent_ref()
            if agent:
                # Run cleanup hooks
                for hook in self._on_cleanup_hooks:
                    try:
                        hook(agent)
                    except Exception as e:
                        logger.error(f"Error in cleanup hook: {e}")

        # Remove from tracking
        del self._agents[agent_id]
        if agent_id in self._agent_metadata:
            del self._agent_metadata[agent_id]
        if agent_id in self._resource_usage:
            del self._resource_usage[agent_id]

        # Record event
        self._record_event(agent_id, "cleaned", {})

        # Force garbage collection if many agents removed:
        if len(self._agents) % 100 == 0:
            gc.collect()

        return True

    def _create_cleanup_callback(self, agent_id: str) -> Callable:
        """Create a cleanup callback for when agent is garbage collected."""

        def cleanup():
            self._record_event(agent_id, "garbage_collected", {})
            # Clean up metadata
            if agent_id in self._agent_metadata:
                del self._agent_metadata[agent_id]
            if agent_id in self._resource_usage:
                del self._resource_usage[agent_id]

        return cleanup

    def _record_event(self, agent_id, event_type, details=None):
        """Record a lifecycle event."""
        if details is None:
            details = {}
        event = AgentLifecycleEvent(
            agent_id=agent_id,
            event_type=event_type,
            timestamp=time.time(),
            details=details,
        )

        self._lifecycle_events.append(event)

        # Limit event history to prevent memory growth
        self._limit_event_history()

    def _limit_event_history(self):
        """Limit event history to prevent excessive memory usage."""
        if len(self._lifecycle_events) > 10000:
            self._lifecycle_events = self._lifecycle_events[-5000:]

    def update_agent_activity(self, agent_id):
        """Update last activity timestamp for an agent."""
        if agent_id in self._resource_usage:
            self._resource_usage[agent_id].last_active = time.time()
            self._resource_usage[agent_id].operation_count += 1

    def get_inactive_agents(self) -> List[str]:
        """Get list of inactive agent IDs."""
        current_time = time.time()
        inactive_threshold = current_time - self.inactive_timeout

        inactive_agents = []
        for agent_id, usage in self._resource_usage.items():
            if usage.last_active < inactive_threshold:
                inactive_agents.append(agent_id)

        return inactive_agents

    async def start_monitoring(self):
        """Start background monitoring and cleanup tasks."""
        if self._running:
            return

        self._running = True
        self._cleanup_task = asyncio.create_task(self._cleanup_loop())
        logger.info("Agent lifecycle monitoring started")

    async def stop_monitoring(self):
        """Stop background monitoring."""
        self._running = False
        if self._cleanup_task:
            self._cleanup_task.cancel()
            from contextlib import suppress

            with suppress(Exception):
                await self._cleanup_task
        logger.info("Agent lifecycle monitoring stopped")

    async def _cleanup_loop(self):
        """Background cleanup loop."""
        while self._running:
            try:
                # Update resource usage
                self._update_resource_usage()

                # Check memory pressure
                self._check_memory_pressure()

                # Clean up inactive agents
                inactive_agents = self.get_inactive_agents()
                for agent_id in inactive_agents:
                    logger.info(f"Cleaning up inactive agent: {agent_id}")
                    self.unregister_agent(agent_id)

                # Clean up dead references
                dead_refs = []
                for agent_id, ref in self._agents.items():
                    if ref() is None:
                        dead_refs.append(agent_id)

                for agent_id in dead_refs:
                    self.unregister_agent(agent_id)

            except Exception as e:
                logger.error(f"Error in cleanup loop: {e}")

            # Wait for next cycle
            await asyncio.sleep(self.cleanup_interval)

    def _update_resource_usage(self):
        """Update resource usage for all agents."""
        process = psutil.Process(os.getpid())
        total_memory = process.memory_info().rss / 1024 / 1024

        # Estimate per-agent memory
        active_agents = len(self._agents)
        if active_agents > 0:
            estimated_per_agent = total_memory / active_agents
            for agent_id in self._resource_usage:
                self._resource_usage[agent_id].memory_mb = estimated_per_agent

    def _check_memory_pressure(self):
        """Check if system is under memory pressure."""
        process = psutil.Process(os.getpid())
        current_memory_mb = process.memory_info().rss / 1024 / 1024

        self._under_memory_pressure = (
            current_memory_mb > self.max_memory_mb * self._memory_pressure_threshold
        )

        if self._under_memory_pressure:
            logger.warning(
                f"Memory pressure detected: {current_memory_mb:.1f}MB / {self.max_memory_mb}MB"
            )

    def add_lifecycle_hook(self, hook_type, callback):
        """Add a lifecycle hook.

        Args:
            hook_type: One of 'create', 'start', 'stop', 'cleanup'
            callback: Function to call on lifecycle event
        """
        if hook_type == "create":
            self._on_create_hooks.append(callback)
        elif hook_type == "start":
            self._on_start_hooks.append(callback)
        elif hook_type == "stop":
            self._on_stop_hooks.append(callback)
        elif hook_type == "cleanup":
            self._on_cleanup_hooks.append(callback)
        else:
            raise ValueError(f"Unknown hook type: {hook_type}")

    @contextmanager
    def managed_agent(self, agent_class, *args, **kwargs):
        """Context manager for automatic agent lifecycle management.

        Example:
            with lifecycle_manager.managed_agent(
                BasicExplorerAgent, "agent_1", "Explorer"
            ) as agent:
                agent.perceive(observation)
                agent.select_action()
        """
        agent = None
        try:
            # Create agent
            agent = agent_class(*args, **kwargs)
            agent_id = getattr(agent, "agent_id", str(id(agent)))

            # Register with lifecycle manager
            self.register_agent(agent)

            # Start agent if it has a start method
            if hasattr(agent, "start"):
                agent.start()
            for hook in self._on_start_hooks:
                hook(agent)

            yield agent

        finally:
            if agent:
                # Stop agent if it has a stop method:
                if hasattr(agent, "stop"):
                    agent.stop()
                    for hook in self._on_stop_hooks:
                        hook(agent)

                # Unregister from lifecycle manager
                agent_id = getattr(agent, "agent_id", str(id(agent)))
                self.unregister_agent(agent_id)

    def get_statistics(self) -> Dict[str, Any]:
        """Get lifecycle statistics."""
        total_agents = len(self._agents)
        active_agents = sum(
            1
            for usage in self._resource_usage.values()
            if time.time() - usage.last_active < 60  # Active in last minute
        )

        total_memory = sum(usage.memory_mb for usage in self._resource_usage.values())

        event_counts = {}
        for event in self._lifecycle_events:
            event_counts[event.event_type] = event_counts.get(event.event_type, 0) + 1

        return {
            "total_agents": total_agents,
            "active_agents": active_agents,
            "inactive_agents": total_agents - active_agents,
            "total_memory_mb": total_memory,
            "memory_pressure": self._under_memory_pressure,
            "event_counts": event_counts,
            "agents_per_gb": ((1024 / total_memory * total_agents) if total_memory > 0 else 0),
        }

    def cleanup_all(self):
        """Clean up all registered agents."""
        agent_ids = list(self._agents.keys())
        for agent_id in agent_ids:
            self.unregister_agent(agent_id)

        # Force garbage collection
        gc.collect()

        logger.info(f"Cleaned up {len(agent_ids)} agents")


# Global instance for convenience
lifecycle_manager = AgentLifecycleManager()


# Example lifecycle hooks
def log_agent_creation(agent):
    """Example hook: Log agent creation."""
    logger.info(f"Agent created: {getattr(agent, 'agent_id', 'unknown')}")


def cleanup_agent_resources(agent):
    """Example hook: Clean up agent-specific resources."""
    # Clear any caches
    if hasattr(agent, "_cache"):
        agent._cache.clear()

    # Remove from any global registries
    if hasattr(agent, "unregister_from_world"):
        agent.unregister_from_world()


# Register default hooks
lifecycle_manager.add_lifecycle_hook("create", log_agent_creation)
lifecycle_manager.add_lifecycle_hook("cleanup", cleanup_agent_resources)
