"""Coordination optimization for multi-agent systems.

This module implements optimizations to improve coordination efficiency
from the current 28.4% at 50 agents, targeting reduced overhead and
better resource utilization.
"""

import asyncio
import logging
import time
from collections import deque
from concurrent.futures import ThreadPoolExecutor
from contextlib import asynccontextmanager
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

import numpy as np

logger = logging.getLogger(__name__)


@dataclass
class CoordinationMetrics:
    """Metrics for coordination efficiency tracking."""

    coordination_id: str
    agent_count: int
    start_time: float = field(default_factory=time.time)
    end_time: Optional[float] = None
    successful_coordinations: int = 0
    failed_coordinations: int = 0
    total_messages: int = 0
    coordination_overhead_ms: float = 0.0

    @property
    def duration(self) -> float:
        """Get coordination duration."""
        if self.end_time:
            return self.end_time - self.start_time
        return time.time() - self.start_time

    @property
    def success_rate(self) -> float:
        """Get coordination success rate."""
        total = self.successful_coordinations + self.failed_coordinations
        return self.successful_coordinations / total if total > 0 else 0.0

    @property
    def efficiency(self) -> float:
        """Calculate coordination efficiency."""
        if self.duration == 0:
            return 0.0
        # Efficiency = successful coordinations per second per agent
        return self.successful_coordinations / (self.duration * self.agent_count)


class CoordinationPool:
    """Pool for managing coordination tasks efficiently."""

    def __init__(self, max_workers: int = 4):
        """Initialize coordination pool."""
        self.max_workers = max_workers
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.active_coordinations: Dict[str, CoordinationMetrics] = {}
        self.coordination_queue = asyncio.Queue(maxsize=100)
        self._shutdown = False

        # Performance optimizations
        self.batch_size = 5  # Process coordinations in batches
        self.coordination_cache: Dict[str, Any] = {}
        self.last_coordination_cleanup = time.time()

    async def start(self):
        """Start the coordination pool."""
        self._shutdown = False
        asyncio.create_task(self._process_coordination_queue())
        logger.info("Coordination pool started")

    async def shutdown(self):
        """Shutdown the coordination pool."""
        self._shutdown = True
        self.executor.shutdown(wait=True)
        logger.info("Coordination pool shutdown")

    async def coordinate_agents(
        self, agents: List[Any], coordination_type: str = "default"
    ) -> CoordinationMetrics:
        """Coordinate a group of agents efficiently."""
        coordination_id = f"{coordination_type}_{len(agents)}_{int(time.time() * 1000)}"

        metrics = CoordinationMetrics(coordination_id=coordination_id, agent_count=len(agents))
        self.active_coordinations[coordination_id] = metrics

        try:
            # Use different strategies based on agent count
            if len(agents) <= 10:
                await self._coordinate_small_group(agents, metrics)
            elif len(agents) <= 30:
                await self._coordinate_medium_group(agents, metrics)
            else:
                await self._coordinate_large_group(agents, metrics)

        except Exception as e:
            logger.error(f"Coordination {coordination_id} failed: {e}")
            metrics.failed_coordinations += 1

        finally:
            metrics.end_time = time.time()
            self._cleanup_coordination(coordination_id)

        return metrics

    async def _coordinate_small_group(self, agents, metrics):
        """Coordinate small groups (<=10 agents) with direct coordination."""
        coordination_tasks = []

        for i, agent in enumerate(agents):
            for j, other_agent in enumerate(agents[i + 1 :], i + 1):
                task = asyncio.create_task(self._coordinate_pair(agent, other_agent, metrics))
                coordination_tasks.append(task)

        # Execute all pairs concurrently
        results = await asyncio.gather(*coordination_tasks, return_exceptions=True)

        for result in results:
            if result is True:
                metrics.successful_coordinations += 1
            else:
                metrics.failed_coordinations += 1

    async def _coordinate_medium_group(self, agents: List[Any], metrics: CoordinationMetrics):
        """Coordinate medium groups (11-30 agents) with hierarchical coordination."""
        subgroup_size = 5
        subgroups = [agents[i : i + subgroup_size] for i in range(0, len(agents), subgroup_size)]

        # Coordinate within subgroups first
        subgroup_tasks = []
        for subgroup in subgroups:
            task = asyncio.create_task(self._coordinate_small_group(subgroup, metrics))
            subgroup_tasks.append(task)

        await asyncio.gather(*subgroup_tasks)

        # Then coordinate between subgroups (representatives)
        representatives = [subgroup[0] for subgroup in subgroups if subgroup]
        await self._coordinate_small_group(representatives, metrics)

    async def _coordinate_large_group(self, agents: List[Any], metrics: CoordinationMetrics):
        """Coordinate large groups (>30 agents) with optimized protocols."""
        hub_interval = max(10, len(agents) // 5)  # At most 5 hubs
        hubs = agents[::hub_interval]

        # Assign non-hub agents to nearest hubs
        hub_groups = {hub.agent_id: [hub] for hub in hubs}

        for agent in agents:
            if agent not in hubs:
                # Assign to first available hub (simple round-robin)
                hub_id = hubs[len(hub_groups[hubs[0].agent_id]) % len(hubs)].agent_id
                hub_groups[hub_id].append(agent)

        # Coordinate within each hub group
        hub_tasks = []
        for hub_agents in hub_groups.values():
            if len(hub_agents) > 1:
                task = asyncio.create_task(self._coordinate_small_group(hub_agents, metrics))
                hub_tasks.append(task)

        await asyncio.gather(*hub_tasks)

        # Coordinate between hubs
        if len(hubs) > 1:
            await self._coordinate_small_group(hubs, metrics)

    async def _coordinate_pair(
        self, agent1: Any, agent2: Any, metrics: CoordinationMetrics
    ) -> bool:
        """Coordinate a pair of agents."""
        try:
            # Simulate coordination overhead
            start_time = time.time()

            # Check if agents can coordinate (simplified)
            if not (hasattr(agent1, "is_active") and hasattr(agent2, "is_active")):
                return False

            if not (agent1.is_active and agent2.is_active):
                return False

            # Simulate message exchange
            await asyncio.sleep(0.001)  # 1ms coordination time

            coordination_time = (time.time() - start_time) * 1000  # Convert to ms
            metrics.coordination_overhead_ms += coordination_time
            metrics.total_messages += 2  # Bidirectional message

            return True

        except Exception as e:
            logger.warning(f"Pair coordination failed: {e}")
            return False

    async def _process_coordination_queue(self):
        """Process queued coordination requests."""
        while not self._shutdown:
            try:
                # Wait for coordination requests
                await asyncio.sleep(0.1)

                # Cleanup old coordinations periodically
                if time.time() - self.last_coordination_cleanup > 60:
                    self._cleanup_old_coordinations()
                    self.last_coordination_cleanup = time.time()

            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Error in coordination queue processing: {e}")

    def _cleanup_coordination(self, coordination_id: str):
        """Clean up completed coordination."""
        if coordination_id in self.active_coordinations:
            # Keep metrics for a short time for reporting
            metrics = self.active_coordinations[coordination_id]
            logger.debug(
                f"Coordination {coordination_id} completed: "
                f"efficiency={metrics.efficiency:.3f}, "
                f"success_rate={metrics.success_rate:.1%}"
            )

    def _cleanup_old_coordinations(self):
        """Clean up old coordination metrics."""
        cutoff_time = time.time() - 300  # Keep for 5 minutes
        old_coordinations = [
            coord_id
            for coord_id, metrics in self.active_coordinations.items()
            if metrics.end_time and metrics.end_time < cutoff_time
        ]

        for coord_id in old_coordinations:
            del self.active_coordinations[coord_id]

        if old_coordinations:
            logger.debug(f"Cleaned up {len(old_coordinations)} old coordinations")

    def get_coordination_stats(self) -> Dict[str, Any]:
        """Get current coordination statistics."""
        active_count = len(self.active_coordinations)
        if not self.active_coordinations:
            return {
                "active_coordinations": 0,
                "avg_efficiency": 0.0,
                "avg_success_rate": 0.0,
                "total_agents_coordinating": 0,
            }

        metrics_list = list(self.active_coordinations.values())

        return {
            "active_coordinations": active_count,
            "avg_efficiency": np.mean([m.efficiency for m in metrics_list]),
            "avg_success_rate": np.mean([m.success_rate for m in metrics_list]),
            "total_agents_coordinating": sum(m.agent_count for m in metrics_list),
            "avg_overhead_ms": np.mean([m.coordination_overhead_ms for m in metrics_list]),
        }


class AdaptiveCoordinationScheduler:
    """Adaptive scheduler that optimizes coordination based on load."""

    def __init__(self):
        self.coordination_pool = CoordinationPool()
        self.load_monitor = LoadMonitor()
        self.scheduling_strategy = "adaptive"

        # Adaptive parameters
        self.max_concurrent_coordinations = 5
        self.coordination_batch_size = 10
        self.efficiency_threshold = 0.2  # Below this, adjust strategy

    async def start(self):
        """Start the adaptive scheduler."""
        await self.coordination_pool.start()
        await self.load_monitor.start()
        logger.info("Adaptive coordination scheduler started")

    async def shutdown(self):
        """Shutdown the scheduler."""
        await self.coordination_pool.shutdown()
        await self.load_monitor.shutdown()
        logger.info("Adaptive coordination scheduler shutdown")

    async def schedule_coordination(
        self, agents: List[Any], priority: int = 0
    ) -> CoordinationMetrics:
        """Schedule coordination with adaptive optimization."""
        # Monitor current load
        current_load = await self.load_monitor.get_current_load()

        # Adjust coordination strategy based on load
        if current_load > 0.8:  # High load
            return await self._coordinate_with_backpressure(agents)
        elif current_load > 0.5:  # Medium load
            return await self._coordinate_with_batching(agents)
        else:  # Low load
            return await self.coordination_pool.coordinate_agents(agents)

    async def _coordinate_with_backpressure(self, agents: List[Any]) -> CoordinationMetrics:
        """Coordinate with backpressure to prevent overload."""
        if len(agents) > 20:
            # Split into smaller groups
            subgroups = [agents[i : i + 10] for i in range(0, len(agents), 10)]

            # Coordinate subgroups sequentially to reduce load
            all_metrics = []
            for subgroup in subgroups:
                metrics = await self.coordination_pool.coordinate_agents(subgroup)
                all_metrics.append(metrics)

                # Small delay between subgroups
                await asyncio.sleep(0.1)

            # Combine metrics
            combined_metrics = self._combine_metrics(all_metrics)
            return combined_metrics
        else:
            return await self.coordination_pool.coordinate_agents(agents)

    async def _coordinate_with_batching(self, agents: List[Any]) -> CoordinationMetrics:
        """Coordinate with intelligent batching."""
        if len(agents) > self.coordination_batch_size:
            batches = [
                agents[i : i + self.coordination_batch_size]
                for i in range(0, len(agents), self.coordination_batch_size)
            ]

            # Process batches with limited concurrency
            semaphore = asyncio.Semaphore(self.max_concurrent_coordinations)

            async def coordinate_batch(batch):
                async with semaphore:
                    return await self.coordination_pool.coordinate_agents(batch)

            tasks = [coordinate_batch(batch) for batch in batches]
            all_metrics = await asyncio.gather(*tasks)

            return self._combine_metrics(all_metrics)
        else:
            return await self.coordination_pool.coordinate_agents(agents)

    def _combine_metrics(self, metrics_list: List[CoordinationMetrics]) -> CoordinationMetrics:
        """Combine multiple coordination metrics."""
        if not metrics_list:
            return CoordinationMetrics("combined", 0)

        combined = CoordinationMetrics(
            coordination_id="combined",
            agent_count=sum(m.agent_count for m in metrics_list),
            start_time=min(m.start_time for m in metrics_list),
        )

        combined.end_time = max(m.end_time for m in metrics_list if m.end_time)
        combined.successful_coordinations = sum(m.successful_coordinations for m in metrics_list)
        combined.failed_coordinations = sum(m.failed_coordinations for m in metrics_list)
        combined.total_messages = sum(m.total_messages for m in metrics_list)
        combined.coordination_overhead_ms = sum(m.coordination_overhead_ms for m in metrics_list)

        return combined


class LoadMonitor:
    """Monitor system load for adaptive coordination."""

    def __init__(self):
        self.cpu_usage_history = deque(maxlen=10)
        self.memory_usage_history = deque(maxlen=10)
        self.coordination_load_history = deque(maxlen=10)
        self._monitoring = False
        self._monitor_task: Optional[asyncio.Task] = None

    async def start(self):
        """Start load monitoring."""
        self._monitoring = True
        self._monitor_task = asyncio.create_task(self._monitor_loop())
        logger.info("Load monitor started")

    async def shutdown(self):
        """Shutdown load monitoring."""
        self._monitoring = False
        if self._monitor_task:
            self._monitor_task.cancel()
        logger.info("Load monitor shutdown")

    async def _monitor_loop(self):
        """Monitor system resources continuously."""
        import psutil

        while self._monitoring:
            try:
                # Monitor CPU and memory
                cpu_percent = psutil.cpu_percent(interval=0.1)
                memory_percent = psutil.virtual_memory().percent

                self.cpu_usage_history.append(cpu_percent)
                self.memory_usage_history.append(memory_percent)

                # Monitor coordination load (placeholder)
                coordination_load = len(
                    coordination_scheduler.coordination_pool.active_coordinations
                )
                self.coordination_load_history.append(coordination_load)

                await asyncio.sleep(1)  # Monitor every second

            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.warning(f"Error in load monitoring: {e}")

    async def get_current_load(self) -> float:
        """Get current system load (0.0 to 1.0)."""
        if not (self.cpu_usage_history and self.memory_usage_history):
            return 0.5  # Default moderate load

        # Combine CPU and memory load
        avg_cpu = np.mean(list(self.cpu_usage_history)) / 100.0
        avg_memory = np.mean(list(self.memory_usage_history)) / 100.0

        # Weight CPU more heavily for coordination tasks
        combined_load = (avg_cpu * 0.7) + (avg_memory * 0.3)

        return min(combined_load, 1.0)

@asynccontextmanager
async def coordination_context():
    """Context manager for coordination operations."""
    await coordination_scheduler.start()
    try:
        yield coordination_scheduler
    finally:
        await coordination_scheduler.shutdown()


async def optimize_coordination_for_agents(agents: List[Any]) -> CoordinationMetrics:
    """High-level function to coordinate agents with optimization."""
    return await coordination_scheduler.schedule_coordination(agents)


def get_coordination_efficiency_stats() -> Dict[str, Any]:
    """Get current coordination efficiency statistics."""
    pool_stats = coordination_scheduler.coordination_pool.get_coordination_stats()
    # Add scheduler-specific stats
    stats = {
        **pool_stats,
        "scheduler_strategy": coordination_scheduler.scheduling_strategy,
        "max_concurrent_coordinations": coordination_scheduler.max_concurrent_coordinations,
        "coordination_batch_size": coordination_scheduler.coordination_batch_size,
    }

    return stats


async def benchmark_coordination_scaling(max_agents: int = 50) -> Dict[int, float]:
    """Benchmark coordination efficiency at different scales."""
    from tests.performance.performance_regression_tests import MockAgent

    results = {}

    for agent_count in range(10, max_agents + 1, 10):
        # Create mock agents
        agents = [MockAgent(f"bench_agent_{i}") for i in range(agent_count)]

        # Measure coordination efficiency
        start_time = time.time()
        metrics = await optimize_coordination_for_agents(agents)
        total_time = time.time() - start_time

        # Calculate efficiency
        efficiency = (
            metrics.efficiency
            if metrics.efficiency > 0
            else (metrics.successful_coordinations / (total_time * agent_count))
        )

        results[agent_count] = efficiency

        logger.info(
            f"Coordination with {agent_count} agents: "
            f"efficiency={efficiency:.3f}, "
            f"success_rate={metrics.success_rate:.1%}"
        )

    return results
