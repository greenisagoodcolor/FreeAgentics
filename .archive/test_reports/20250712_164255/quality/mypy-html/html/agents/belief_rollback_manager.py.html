"""Enhanced Belief Rollback Manager for Active Inference Agents.

This module extends the belief versioning system with advanced rollback capabilities,
including cascade rollback for dependent beliefs, rollback validation, and integration
with the free energy trigger system.
"""

import logging
from collections import defaultdict, deque
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any, Dict, List, Optional, Set, Tuple

import numpy as np

from agents.belief_versioning import BeliefVersion, BeliefVersioningSystem
from agents.free_energy_triggers import TriggerCondition, TriggerEvent

logger = logging.getLogger(__name__)


class RollbackStrategy(Enum):
    """Strategies for handling rollbacks."""

    EXACT = "exact"  # Restore exact state
    SMOOTH = "smooth"  # Smooth transition to target state
    INCREMENTAL = "incremental"  # Gradual rollback over multiple steps
    SAFE = "safe"  # Rollback with safety checks


class DependencyType(Enum):
    """Types of belief dependencies."""

    CAUSAL = "causal"  # A causes B
    TEMPORAL = "temporal"  # A precedes B
    HIERARCHICAL = "hierarchical"  # A is parent of B
    CONSTRAINT = "constraint"  # A constrains B


@dataclass
class BeliefDependency:
    """Represents a dependency between beliefs."""

    source_agent: str
    target_agent: str
    dependency_type: DependencyType
    strength: float = 1.0  # 0 to 1
    metadata: Dict[str, Any] = field(default_factory=dict)


@dataclass
class RollbackPlan:
    """Plan for executing a rollback operation."""

    rollback_id: str
    primary_agent: str
    target_version: str
    affected_agents: List[str]
    strategy: RollbackStrategy
    steps: List["RollbackStep"]
    estimated_impact: float
    created_at: datetime = field(default_factory=datetime.now)

    def get_total_steps(self) -> int:
        """Get total number of steps in plan."""
        return len(self.steps)


@dataclass
class RollbackStep:
    """Single step in a rollback plan."""

    step_id: str
    agent_id: str
    target_version: str
    dependencies: List[str]  # Steps that must complete first
    priority: int  # Higher priority executed first
    estimated_impact: float
    metadata: Dict[str, Any] = field(default_factory=dict)


@dataclass
class RollbackResult:
    """Result of a rollback operation."""

    success: bool
    rollback_plan: RollbackPlan
    executed_steps: List[RollbackStep]
    failed_steps: List[Tuple[RollbackStep, str]]  # Step and error message
    new_versions: Dict[str, BeliefVersion]  # agent_id -> new version
    metrics: Dict[str, Any]
    duration: float  # seconds


class BeliefRollbackManager:
    """Manages complex rollback operations for belief states."""

    def __init__(self, versioning_system, trigger_manager=None, safety_threshold=0.5):
        """Initialize rollback manager.

        Args:
            versioning_system: Belief versioning system
            trigger_manager: Optional free energy trigger manager
            safety_threshold: Threshold for safe rollbacks
        """
        self.versioning = versioning_system
        self.trigger_manager = trigger_manager
        self.safety_threshold = safety_threshold

        # Dependency tracking:
        self.dependencies: Dict[str, List[BeliefDependency]] = defaultdict(list)
        self.reverse_dependencies: Dict[str, List[BeliefDependency]] = defaultdict(list)

        # Rollback history
        self.rollback_history: deque = deque(maxlen=1000)
        self.active_rollbacks: Dict[str, RollbackPlan] = {}

        # Safety mechanisms
        self.blocked_agents: Set[str] = set()  # Agents blocked from rollback
        self.rollback_limits: Dict[str, int] = defaultdict(int)  # Rollback count limits

        # Statistics
        self.stats = {
            "total_rollbacks": 0,
            "successful_rollbacks": 0,
            "failed_rollbacks": 0,
            "cascade_rollbacks": 0,
            "safety_blocks": 0,
        }

        logger.info("Initialized belief rollback manager")

    def register_dependency(self, dependency):
        """Register a dependency between agents.

        Args:
            dependency: Belief dependency to register
        """
        self.dependencies[dependency.source_agent].append(dependency)
        self.reverse_dependencies[dependency.target_agent].append(dependency)

        logger.debug(
            f"Registered dependency: {dependency.source_agent} -> {dependency.target_agent}"
        )

    async def plan_rollback(
        self,
        agent_id: str,
        target_version_id: str,
        strategy: RollbackStrategy = RollbackStrategy.SAFE,
        cascade: bool = True,
    ) -> RollbackPlan:
        """Create a rollback plan.

        Args:
            agent_id: Primary agent to rollback
            target_version_id: Target version to rollback to
            strategy: Rollback strategy
            cascade: Whether to include dependent agents

        Returns:
            Rollback plan

        Raises:
            ValueError: If rollback cannot be planned
        """
        # Validate target version exists
        target_version = await self.versioning.get_version(agent_id, target_version_id)
        if not target_version:
            raise ValueError(f"Target version {target_version_id} not found")

        # Check safety
        if agent_id in self.blocked_agents:
            raise ValueError(f"Agent {agent_id} is blocked from rollback")

        # Generate rollback ID
        rollback_id = f"rollback_{agent_id}_{int(datetime.now().timestamp())}"

        # Find affected agents
        affected_agents = {agent_id}
        if cascade:
            affected_agents.update(await self._find_dependent_agents(agent_id))

        # Create rollback steps
        steps = await self._create_rollback_steps(
            agent_id, target_version, affected_agents, strategy
        )

        # Calculate impact
        impact = await self._estimate_rollback_impact(steps)

        # Create plan
        plan = RollbackPlan(
            rollback_id=rollback_id,
            primary_agent=agent_id,
            target_version=target_version_id,
            affected_agents=list(affected_agents),
            strategy=strategy,
            steps=steps,
            estimated_impact=impact,
        )

        logger.info(f"Created rollback plan {rollback_id} affecting {len(affected_agents)} agents")

        return plan

    async def execute_rollback(self, plan: RollbackPlan, dry_run: bool = False) -> RollbackResult:
        """Execute a rollback plan.

        Args:
            plan: Rollback plan to execute
            dry_run: If True, simulate without making changes

        Returns:
            Rollback result
        """
        start_time = datetime.now()

        # Track execution
        self.active_rollbacks[plan.rollback_id] = plan
        executed_steps = []
        failed_steps = []
        new_versions = {}

        try:
            # Validate plan is still valid
            await self._validate_rollback_plan(plan)

            # Sort steps by priority and dependencies
            sorted_steps = await self._sort_rollback_steps(plan.steps)

            # Execute steps
            for step in sorted_steps:
                try:
                    if dry_run:
                        logger.info(f"[DRY RUN] Would execute step {step.step_id}")
                        executed_steps.append(step)
                    else:
                        new_version = await self._execute_rollback_step(step, plan.strategy)
                        new_versions[step.agent_id] = new_version
                        executed_steps.append(step)

                        # Notify trigger system
                        if self.trigger_manager:
                            await self._notify_trigger_system(step.agent_id, new_version)

                except Exception as e:
                    logger.error(f"Failed to execute rollback step {step.step_id}: {e}")
                    failed_steps.append((step, str(e)))

                    # Decide whether to continue
                    if plan.strategy == RollbackStrategy.SAFE:
                        logger.warning("Stopping rollback due to failure in SAFE mode")
                        break

            # Calculate metrics
            duration = (datetime.now() - start_time).total_seconds()
            success = len(failed_steps) == 0

            metrics = {
                "total_steps": len(plan.steps),
                "executed_steps": len(executed_steps),
                "failed_steps": len(failed_steps),
                "duration": duration,
                "agents_affected": len(plan.affected_agents),
            }

            # Create result
            result = RollbackResult(
                success=success,
                rollback_plan=plan,
                executed_steps=executed_steps,
                failed_steps=failed_steps,
                new_versions=new_versions,
                metrics=metrics,
                duration=duration,
            )

            # Update statistics
            self.stats["total_rollbacks"] += 1
            if success:
                self.stats["successful_rollbacks"] += 1
            else:
                self.stats["failed_rollbacks"] += 1

            if len(plan.affected_agents) > 1:
                self.stats["cascade_rollbacks"] += 1

            # Store in history
            self.rollback_history.append(result)

            logger.info(f"Rollback {plan.rollback_id} completed: success={success}")

            return result

        finally:
            # Clean up
            del self.active_rollbacks[plan.rollback_id]

    async def validate_rollback(
        self, agent_id: str, target_version_id: str
    ) -> Tuple[bool, List[str]]:
        """Validate if a rollback is safe to perform.

        Args:
            agent_id: Agent to rollback
            target_version_id: Target version

        Returns:
            Tuple of (is_valid, list of issues)
        """
        issues = []

        # Check if agent is blocked
        if agent_id in self.blocked_agents:
            issues.append(f"Agent {agent_id} is blocked from rollback")

        # Check rollback limits
        if self.rollback_limits[agent_id] >= 5:  # Configurable limit
            issues.append(f"Agent {agent_id} has exceeded rollback limit")

        # Check version exists
        target_version = await self.versioning.get_version(agent_id, target_version_id)
        if not target_version:
            issues.append(f"Target version {target_version_id} not found")
        else:
            # Check version age
            age = (datetime.now() - target_version.timestamp).total_seconds()
            if age > 86400:  # 24 hours
                issues.append(f"Target version is {age / 3600:.1f} hours old")

            # Check belief state validity
            if target_version.entropy and target_version.entropy > 2.0:
                issues.append(f"Target version has high entropy: {target_version.entropy:.2f}")

        # Check for active rollbacks
        if any(agent_id in plan.affected_agents for plan in self.active_rollbacks.values()):
            issues.append(f"Agent {agent_id} is involved in an active rollback")

        is_valid = len(issues) == 0

        return is_valid, issues

    async def _find_dependent_agents(self, agent_id: str) -> Set[str]:
        """Find all agents that depend on the given agent."""
        dependent_agents = set()
        to_visit = [agent_id]
        visited = set()

        while to_visit:
            current = to_visit.pop()
            if current in visited:
                continue

            visited.add(current)

            # Find dependencies
            for dep in self.dependencies.get(current, []):
                if dep.target_agent not in visited:
                    dependent_agents.add(dep.target_agent)
                    to_visit.append(dep.target_agent)

        return dependent_agents

    async def _create_rollback_steps(
        self,
        primary_agent: str,
        target_version: BeliefVersion,
        affected_agents: Set[str],
        strategy: RollbackStrategy,
    ) -> List[RollbackStep]:
        """Create rollback steps for all affected agents."""
        steps = []
        step_counter = 0

        # Create primary step
        primary_step = RollbackStep(
            step_id=f"step_{step_counter}",
            agent_id=primary_agent,
            target_version=target_version.version_id,
            dependencies=[],
            priority=100,  # Highest priority
            estimated_impact=1.0,
        )
        steps.append(primary_step)
        step_counter += 1

        # Create dependent steps
        for agent_id in affected_agents:
            if agent_id == primary_agent:
                continue

            # Find appropriate target version for dependent
            dep_target_version = await self._find_dependent_target_version(
                agent_id, target_version.timestamp
            )
            if dep_target_version:
                step = RollbackStep(
                    step_id=f"step_{step_counter}",
                    agent_id=agent_id,
                    target_version=dep_target_version.version_id,
                    dependencies=[primary_step.step_id],
                    priority=50,  # Lower priority than primary
                    estimated_impact=0.5,
                )
                steps.append(step)
                step_counter += 1

        return steps

    async def _find_dependent_target_version(
        self, agent_id: str, target_timestamp: datetime
    ) -> Optional[BeliefVersion]:
        """Find appropriate rollback version for dependent agent."""
        # Get version history
        history = await self.versioning.get_version_history(agent_id)

        # Find version closest to target timestamp
        best_version = None
        min_time_diff = float("inf")
        for version in history:
            time_diff = abs((version.timestamp - target_timestamp).total_seconds())
            if time_diff < min_time_diff:
                min_time_diff = time_diff
                best_version = version

        return best_version

    async def _estimate_rollback_impact(self, steps: List[RollbackStep]) -> float:
        """Estimate the impact of a rollback."""
        if not steps:
            return 0.0

        # Simple impact calculation based on number of steps and priorities
        total_impact = sum(step.estimated_impact for step in steps)
        normalized_impact = total_impact / len(steps)

        return normalized_impact

    async def _validate_rollback_plan(self, plan):
        """Validate that a rollback plan is still valid."""
        # Check all target versions still exist
        for step in plan.steps:
            version = await self.versioning.get_version(step.agent_id, step.target_version)
            if not version:
                raise ValueError(f"Target version {step.target_version} no longer exists")

    async def _sort_rollback_steps(self, steps: List[RollbackStep]) -> List[RollbackStep]:
        """Sort rollback steps by priority and dependencies."""
        # Build dependency graph
        in_degree = {step.step_id: len(step.dependencies) for step in steps}

        # Topological sort with priority
        sorted_steps = []
        available = [step for step in steps if in_degree[step.step_id] == 0]
        available.sort(key=lambda s: s.priority, reverse=True)

        while available:
            step = available.pop(0)
            sorted_steps.append(step)

            # Update dependencies
            for other_step in steps:
                if step.step_id in other_step.dependencies:
                    in_degree[other_step.step_id] -= 1
                    if in_degree[other_step.step_id] == 0:
                        available.append(other_step)
                        available.sort(key=lambda s: s.priority, reverse=True)

        if len(sorted_steps) != len(steps):
            raise ValueError("Circular dependencies detected in rollback plan")

        return sorted_steps

    async def _execute_rollback_step(
        self, step: RollbackStep, strategy: RollbackStrategy
    ) -> BeliefVersion:
        """Execute a single rollback step."""
        # Get target version
        target_version = await self.versioning.get_version(step.agent_id, step.target_version)
        if not target_version:
            raise ValueError(f"Target version {step.target_version} not found")

        # Apply strategy-specific modifications
        belief_state = target_version.belief_state.copy()

        if strategy == RollbackStrategy.SMOOTH:
            # Smooth transition: blend with current state
            current_version = await self.versioning.get_current_version(step.agent_id)
            if current_version:
                alpha = 0.7  # Blend factor
                belief_state = alpha * belief_state + (1 - alpha) * current_version.belief_state
                belief_state = belief_state / np.sum(belief_state)  # Normalize

        elif strategy == RollbackStrategy.INCREMENTAL:
            # Incremental: partial step towards target
            current_version = await self.versioning.get_current_version(step.agent_id)
            if current_version:
                step_size = 0.5  # How far to move
                diff = belief_state - current_version.belief_state
                belief_state = current_version.belief_state + step_size * diff
                belief_state = np.maximum(belief_state, 0)  # Ensure non-negative
                belief_state = belief_state / np.sum(belief_state)  # Normalize

        # Create new version
        new_version = await self.versioning.create_version(
            agent_id=step.agent_id,
            belief_state=belief_state,
            operation=f"rollback_{strategy.value}",
            metadata={
                "rollback_step_id": step.step_id,
                "target_version": step.target_version,
                "strategy": strategy.value,
            },
        )

        # Update rollback limits
        self.rollback_limits[step.agent_id] += 1

        return new_version

    async def _notify_trigger_system(self, agent_id, new_version):
        """Notify trigger system of rollback."""
        if not self.trigger_manager:
            return

        try:
            # Create trigger event for rollback
            event = TriggerEvent(
                event_id=f"rollback_{agent_id}_{int(datetime.now().timestamp())}",
                trigger_type=TriggerCondition.BELIEF_OPERATION_APPLIED,
                timestamp=datetime.now(),
                agent_id=agent_id,
                data={
                    "rollback": True,
                    "version_id": new_version.version_id,
                    "belief_state": new_version.belief_state.tolist(),
                    "entropy": new_version.entropy,
                },
                priority=3,  # High priority
            )

            self.trigger_manager.queue_event(event)

        except Exception as e:
            logger.error(f"Failed to notify trigger system: {e}")

    def block_agent(self, agent_id, reason):
        """Block an agent from rollback.

        Args:
            agent_id: Agent to block
            reason: Reason for blocking
        """
        self.blocked_agents.add(agent_id)
        logger.warning(f"Blocked agent {agent_id} from rollback: {reason}")
        self.stats["safety_blocks"] += 1

    def unblock_agent(self, agent_id):
        """Unblock an agent for rollback.

        Args:
            agent_id: Agent to unblock
        """
        self.blocked_agents.discard(agent_id)
        logger.info(f"Unblocked agent {agent_id} for rollback")

    def get_rollback_history(
        self, agent_id: Optional[str] = None, limit: int = 100
    ) -> List[RollbackResult]:
        """Get rollback history.

        Args:
            agent_id: Filter by agent (None for all)
            limit: Maximum results to return

        Returns:
            List of rollback results
        """
        history = list(self.rollback_history)
        if agent_id:
            history = [r for r in history if agent_id in r.rollback_plan.affected_agents]

        # Sort by timestamp (newest first)
        history.sort(key=lambda r: r.rollback_plan.created_at, reverse=True)

        return history[:limit]

    def get_statistics(self) -> Dict[str, Any]:
        """Get rollback manager statistics."""
        return {
            **self.stats,
            "active_rollbacks": len(self.active_rollbacks),
            "blocked_agents": len(self.blocked_agents),
            "total_dependencies": sum(len(deps) for deps in self.dependencies.values()),
        }

    def reset_rollback_limits(self, agent_id: Optional[str] = None):
        """Reset rollback limits for agent(s).

        Args:
            agent_id: Agent to reset (None for all)
        """
        if agent_id:
            self.rollback_limits[agent_id] = 0
        else:
            self.rollback_limits.clear()

        logger.info(f"Reset rollback limits for {agent_id or 'all agents'}")


# Convenience functions
def create_rollback_manager(
    versioning_system: BeliefVersioningSystem,
    trigger_manager: Optional[Any] = None,
    config: Optional[Dict[str, Any]] = None,
) -> BeliefRollbackManager:
    """Create a rollback manager with optional configuration."""
    config = config or {}
    return BeliefRollbackManager(
        versioning_system=versioning_system,
        trigger_manager=trigger_manager,
        safety_threshold=config.get("safety_threshold", 0.5),
    )
