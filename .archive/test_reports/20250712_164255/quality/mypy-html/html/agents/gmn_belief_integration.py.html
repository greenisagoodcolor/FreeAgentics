"""Integration layer between GMN parser and thread-safe belief management.

This module bridges the GMN parser with the thread-safety mechanisms to ensure
concurrent belief updates from GMN specifications are handled safely.

Note: Per NEMESIS audit, this is part of the architectural blueprint. The actual
Active Inference functionality is 0% complete.
"""

import logging
from dataclasses import dataclass
from datetime import datetime
from typing import Any, Dict, List, Optional, Tuple

from agents.thread_safety import (
    BeliefUpdateRequest,
    get_global_belief_manager,
)

logger = logging.getLogger(__name__)

# Try to import GMN parser
try:
    from inference.active.gmn_parser import parse_gmn_spec

    GMN_AVAILABLE = True
except ImportError:
    GMN_AVAILABLE = False
    logger.warning("GMN parser not available - using mock implementation")

    def parse_gmn_spec(spec: str, observation=None) -> Dict[str, Any]:
        """Mock GMN parser."""
        return {
            "nodes": {"agent_belief": {"type": "belie", "properties": {}}},
            "edges": [],
            "metadata": {"mock": True},
        }


@dataclass
class GMNBeliefUpdate:
    """Represents a belief update derived from GMN specification."""

    agent_id: str
    node_id: str
    node_type: str
    belief_data: Dict[str, Any]
    dependencies: List[str] = None
    timestamp: datetime = None

    def __post_init__(self):
        if self.dependencies is None:
            self.dependencies = []
        if self.timestamp is None:
            self.timestamp = datetime.now()


class GMNBeliefIntegrator:
    """Integrates GMN specifications with thread-safe belief management.

    Provides:
    - GMN spec validation for belief updates
    - Atomic belief updates from GMN nodes
    - Dependency resolution for belief propagation
    - Free energy recalculation triggers
    """

    def __init__(self, belief_manager=None):
        """Initialize GMN belief integrator."""
        self.belief_manager = belief_manager or get_global_belief_manager()
        self._gmn_specs: Dict[str, Dict] = {}  # agent_id -> GMN spec
        self._update_history: List[GMNBeliefUpdate] = []

    def validate_gmn_spec(
        self, agent_id: str, gmn_spec: str, observation=None
    ) -> Tuple[bool, Optional[str]]:
        """Validate a GMN specification for belief updates.

        Args:
            agent_id: Agent identifier
            gmn_spec: GMN specification string

        Returns:
            Tuple of (is_valid, error_message)
        """
        try:
            if not GMN_AVAILABLE:
                return True, None  # Mock validation always passes

            parsed_spec = parse_gmn_spec(gmn_spec)

            # Basic validation
            if "nodes" not in parsed_spec:
                return False, "GMN spec missing 'nodes' section"

            # Check for belief-related nodes
            belief_nodes = [
                node
                for node_id, node in parsed_spec["nodes"].items()
                if node.get("type") in ["belie", "state", "observation"]
            ]

            if not belief_nodes:
                return (
                    False,
                    "GMN spec must contain at least one belief/state/observation node",
                )

            # Store validated spec
            self._gmn_specs[agent_id] = parsed_spec

            logger.info(f"Successfully validated GMN spec for agent {agent_id}")
            return True, None

        except Exception as e:
            error_msg = f"GMN validation failed: {str(e)}"
            logger.error(error_msg)
            return False, error_msg

    async def apply_gmn_belief_updates(self, agent_id: str, observation=None) -> bool:
        """Apply belief updates from GMN specification.

        Args:
            agent_id: Agent identifier

        Returns:
            True if updates were successful
        """
        if agent_id not in self._gmn_specs:
            logger.warning(f"No GMN spec found for agent {agent_id}")
            return False

        spec = self._gmn_specs[agent_id]

        try:
            # Extract belief updates from GMN nodes
            updates = self._extract_belief_updates(agent_id, spec)

            # Apply updates through thread-safe manager
            for update in updates:
                belief_request = BeliefUpdateRequest(
                    agent_id=update.agent_id,
                    belief_type=update.node_type,
                    belief_data=update.belief_data,
                    metadata={
                        "source": "gmn",
                        "node_id": update.node_id,
                        "dependencies": update.dependencies,
                    },
                )

                success = await self.belief_manager.update_beliefs(belief_request)
                if not success:
                    logger.error(f"Failed to apply belief update for node {update.node_id}")
                    return False

                # Track the update
                self._update_history.append(update)

            logger.info(f"Applied {len(updates)} GMN belief updates for agent {agent_id}")
            return True

        except Exception as e:
            logger.error(f"Error applying GMN belief updates: {e}")
            return False

    def _extract_belief_updates(
        self, agent_id: str, spec: Dict[str, Any], observation=None
    ) -> List[GMNBeliefUpdate]:
        """Extract belief updates from parsed GMN specification."""
        updates = []

        nodes = spec.get("nodes", {})
        edges = spec.get("edges", [])

        for node_id, node_data in nodes.items():
            node_type = node_data.get("type", "unknown")

            # Process belief-related nodes
            if node_type in ["belie", "state", "observation"]:
                belief_data = self._extract_belief_data(node_data)
                dependencies = self._extract_dependencies(node_id, edges)

                update = GMNBeliefUpdate(
                    agent_id=agent_id,
                    node_id=node_id,
                    node_type=node_type,
                    belief_data=belief_data,
                    dependencies=dependencies,
                )
                updates.append(update)

        return updates

    def _extract_belief_data(self, node_data: Dict[str, Any], observation=None) -> Dict[str, Any]:
        """Extract belief data from GMN node."""
        properties = node_data.get("properties", {})

        # Default belief structure
        belief_data = {
            "confidence": properties.get("confidence", 0.5),
            "certainty": properties.get("certainty", 0.5),
            "evidence": properties.get("evidence", {}),
            "source": "gmn_node",
            "last_updated": datetime.now().isoformat(),
        }

        # Include any additional properties
        for key, value in properties.items():
            if key not in belief_data:
                belief_data[key] = value

        return belief_data

    def _extract_dependencies(
        self, node_id: str, edges: List[Dict[str, Any]], observation=None
    ) -> List[str]:
        """Extract dependencies for a node from GMN edges."""
        dependencies = []

        for edge in edges:
            if edge.get("target") == node_id:
                source = edge.get("source")
                if source:
                    dependencies.append(source)

        return dependencies

    async def process_gmn_trigger(
        self, agent_id: str, trigger_data: Dict[str, Any], observation=None
    ) -> bool:
        """Process a GMN-based trigger for belief updates.

        Args:
            agent_id: Agent identifier
            trigger_data: Trigger information

        Returns:
            True if trigger was processed successfully
        """
        try:
            trigger_type = trigger_data.get("type", "unknown")

            if trigger_type == "belief_update":
                return await self.apply_gmn_belief_updates(agent_id)
            elif trigger_type == "revalidation":
                spec = self._gmn_specs.get(agent_id)
                if spec:
                    # Re-extract and apply updates
                    return await self.apply_gmn_belief_updates(agent_id)

            logger.warning(f"Unknown GMN trigger type: {trigger_type}")
            return False

        except Exception as e:
            logger.error(f"Error processing GMN trigger: {e}")
            return False

    def get_update_history(
        self, agent_id: str = None, limit: int = 100, observation=None
    ) -> List[GMNBeliefUpdate]:
        """Get history of belief updates.

        Args:
            agent_id: Optional agent filter
            limit: Maximum number of updates to return

        Returns:
            List of belief updates
        """
        history = self._update_history

        if agent_id:
            history = [u for u in history if u.agent_id == agent_id]

        # Sort by timestamp (newest first)
        history.sort(key=lambda u: u.timestamp, reverse=True)

        return history[:limit]

    def get_statistics(self, observation=None) -> Dict[str, Any]:
        """Get integration statistics."""
        total_updates = len(self._update_history)

        # Group by agent
        agent_stats = {}
        for update in self._update_history:
            agent_id = update.agent_id
            if agent_id not in agent_stats:
                agent_stats[agent_id] = {"count": 0, "node_types": set()}
            agent_stats[agent_id]["count"] += 1
            agent_stats[agent_id]["node_types"].add(update.node_type)

        # Convert sets to lists for JSON serialization
        for stats in agent_stats.values():
            stats["node_types"] = list(stats["node_types"])

        return {
            "total_updates": total_updates,
            "agents_with_gmn": len(self._gmn_specs),
            "agent_statistics": agent_stats,
            "gmn_available": GMN_AVAILABLE,
        }


# Global integrator instance
_gmn_integrator: Optional[GMNBeliefIntegrator] = None


def get_gmn_integrator(observation=None) -> GMNBeliefIntegrator:
    """Get the global GMN belief integrator."""
    global _gmn_integrator
    if _gmn_integrator is None:
        _gmn_integrator = GMNBeliefIntegrator()
    return _gmn_integrator
