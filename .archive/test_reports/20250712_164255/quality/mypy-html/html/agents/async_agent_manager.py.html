"""Asynchronous Agent Manager for Multi-Agent Scaling.

This module provides async/await based multi-agent processing with thread pools
for improved concurrent agent operations and I/O coordination.
Key features:
1. Thread pool for concurrent PyMDP operations
2. Async coordination layer for I/O and coordination
3. Batched processing for efficiency
4. Better resource utilization than single-threaded processing
"""

import asyncio
import logging
import multiprocessing as mp
import time
import uuid
from concurrent.futures import ThreadPoolExecutor
from dataclasses import dataclass
from typing import Any, Dict, List, Optional, Tuple

logger = logging.getLogger(__name__)


@dataclass
class AgentOperation:
    """Represents a single agent operation to be processed."""

    agent_id: str
    operation_type: str  # 'perceive', 'update_beliefs', 'select_action', 'step'
    data: Dict[str, Any]
    timestamp: float
    operation_id: str = None

    def __post_init__(self):
        """__post_init__ method."""
        if self.operation_id is None:
            self.operation_id = str(uuid.uuid4())


@dataclass
class AgentOperationResult:
    """Result of an agent operation."""

    operation_id: str
    agent_id: str
    success: bool
    result: Any
    error: Optional[str]
    execution_time_ms: float


def _process_agent_operation(
    agent_config: Dict[str, Any], operation: AgentOperation
) -> AgentOperationResult:
    """Process a single agent operation in a thread pool.

    This function runs in a thread pool to improve concurrency for I/O-bound operations
    while sharing the same environment and dependencies.
    """
    start_time = time.time()
    try:
        agent = _create_agent_from_config(agent_config)
        result = _execute_operation(agent, operation)
        execution_time = (time.time() - start_time) * 1000

        return AgentOperationResult(
            operation_id=operation.operation_id,
            agent_id=operation.agent_id,
            success=True,
            result=result,
            error=None,
            execution_time_ms=execution_time,
        )

    except Exception as e:
        return _create_error_result(operation, start_time, e)


def _create_agent_from_config(agent_config: Dict[str, Any]):
    """Create and configure agent from config."""
    # Import here to avoid pickling issues
    import sys

    sys.path.append("/home/green/FreeAgentics")
    from agents.base_agent import BasicExplorerAgent

    agent = BasicExplorerAgent(
        agent_config["agent_id"], agent_config["name"], agent_config.get("grid_size", 5)
    )

    # Set agent state if provided
    if "position" in agent_config:
        agent.position = agent_config["position"]
    if "total_steps" in agent_config:
        agent.total_steps = agent_config["total_steps"]
    if agent_config.get("is_active", False):
        agent.start()

    return agent


def _execute_operation(agent, operation: AgentOperation) -> Dict[str, Any]:
    """Execute specific operation on agent."""
    operation_handlers = {
        "perceive": _handle_perceive,
        "update_beliefs": _handle_update_beliefs,
        "select_action": _handle_select_action,
        "step": _handle_step,
    }

    handler = operation_handlers.get(operation.operation_type)
    if not handler:
        raise ValueError(f"Unknown operation type: {operation.operation_type}")

    return handler(agent, operation.data)


def _handle_perceive(agent, data: Dict[str, Any]) -> Dict[str, Any]:
    """Handle perceive operation."""
    agent.perceive(data["observation"])
    return {"status": "perceived", "metrics": agent.metrics}


def _handle_update_beliefs(agent, data: Dict[str, Any]) -> Dict[str, Any]:
    """Handle belief update operation."""
    agent.update_beliefs()
    return {"status": "beliefs_updated", "beliefs": agent.beliefs, "metrics": agent.metrics}


def _handle_select_action(agent, data: Dict[str, Any]) -> Dict[str, Any]:
    """Handle action selection operation."""
    action = agent.select_action()
    return {"action": action, "metrics": agent.metrics}


def _handle_step(agent, data: Dict[str, Any]) -> Dict[str, Any]:
    """Handle full step operation."""
    observation = data["observation"]
    agent.perceive(observation)
    agent.update_beliefs()
    action = agent.select_action()
    return {"action": action, "beliefs": agent.beliefs, "metrics": agent.metrics}


def _create_error_result(
    operation: AgentOperation, start_time: float, error: Exception
) -> AgentOperationResult:
    """Create error result."""
    execution_time = (time.time() - start_time) * 1000
    return AgentOperationResult(
        operation_id=operation.operation_id,
        agent_id=operation.agent_id,
        success=False,
        result=None,
        error=str(error),
        execution_time_ms=execution_time,
    )


class AsyncAgentManager:
    """Manages asynchronous multi-agent execution with thread pool backend."""

    def __init__(
        self,
        max_workers: Optional[int] = None,
        batch_size: int = 10,
        process_timeout: float = 5.0,
    ):
        """Initialize async agent manager.

        Args:
            max_workers: Maximum thread pool workers (default: CPU count * 2)
            batch_size: Number of operations to batch together
            process_timeout: Timeout for processing operations
        """
        self.max_workers = max_workers or (mp.cpu_count() * 2)
        self.batch_size = batch_size
        self.process_timeout = process_timeout

        # Thread pool for agent operations
        self.executor = ThreadPoolExecutor(max_workers=self.max_workers)

        # Agent configurations
        self.agent_configs: Dict[str, Dict[str, Any]] = {}

        # Pending operations
        self.pending_operations: List[AgentOperation] = []

        # Statistics
        self.stats = {
            "total_operations": 0,
            "successful_operations": 0,
            "failed_operations": 0,
            "average_execution_time": 0.0,
            "batch_count": 0,
        }

    def register_agent(self, agent_id: str, agent_config: Dict[str, Any]) -> None:
        """Register an agent for async processing."""
        self.agent_configs[agent_id] = {
            "agent_id": agent_id,
            **agent_config,
        }
        logger.info(f"Registered agent {agent_id} for async processing")

    def unregister_agent(self, agent_id: str) -> None:
        """Unregister an agent."""
        if agent_id in self.agent_configs:
            del self.agent_configs[agent_id]
            logger.info(f"Unregistered agent {agent_id}")

    async def process_operation(
        self, agent_id: str, operation_type: str, data: Dict[str, Any]
    ) -> AgentOperationResult:
        """Process a single operation for an agent asynchronously."""
        if agent_id not in self.agent_configs:
            raise ValueError(f"Agent {agent_id} not registered")

        operation = AgentOperation(
            agent_id=agent_id,
            operation_type=operation_type,
            data=data,
            timestamp=time.time(),
        )

        # Submit to thread pool
        loop = asyncio.get_event_loop()
        result = await loop.run_in_executor(
            self.executor,
            _process_agent_operation,
            self.agent_configs[agent_id],
            operation,
        )

        # Update statistics
        self.stats["total_operations"] += 1
        if result.success:
            self.stats["successful_operations"] += 1
        else:
            self.stats["failed_operations"] += 1

        # Update average execution time
        total_ops = self.stats["total_operations"]
        avg_time = self.stats["average_execution_time"]
        self.stats["average_execution_time"] = (
            avg_time * (total_ops - 1) + result.execution_time_ms
        ) / total_ops

        return result

    async def batch_process_operations(
        self, operations: List[Tuple[str, str, Dict[str, Any]]]
    ) -> List[AgentOperationResult]:
        """Process multiple operations in batch.

        Args:
            operations: List of (agent_id, operation_type, data) tuples

        Returns:
            List of operation results
        """
        # Create operation objects
        agent_operations = []
        for agent_id, op_type, data in operations:
            if agent_id in self.agent_configs:
                agent_operations.append(
                    AgentOperation(
                        agent_id=agent_id,
                        operation_type=op_type,
                        data=data,
                        timestamp=time.time(),
                    )
                )

        # Process in parallel using thread pool
        loop = asyncio.get_event_loop()
        futures = []

        for operation in agent_operations:
            future = loop.run_in_executor(
                self.executor,
                _process_agent_operation,
                self.agent_configs[operation.agent_id],
                operation,
            )
            futures.append(future)

        # Wait for all operations with timeout
        results = await asyncio.gather(*futures, return_exceptions=True)

        # Process results
        processed_results = []
        for i, result in enumerate(results):
            if isinstance(result, Exception):
                # Create error result for exceptions
                operation = agent_operations[i]
                processed_results.append(_create_error_result(operation, time.time(), result))
            else:
                processed_results.append(result)

        # Update statistics
        self.stats["batch_count"] += 1
        for result in processed_results:
            self.stats["total_operations"] += 1
            if result.success:
                self.stats["successful_operations"] += 1
            else:
                self.stats["failed_operations"] += 1

        return processed_results

    async def execute_parallel_step(
        self, observations: Dict[str, Any]
    ) -> Dict[str, Dict[str, Any]]:
        """Execute a parallel step for all registered agents.

        Args:
            observations: Dict mapping agent_id to observation

        Returns:
            Dict mapping agent_id to step result
        """
        operations = []
        for agent_id in self.agent_configs:
            if agent_id in observations:
                operations.append((agent_id, "step", {"observation": observations[agent_id]}))

        # Batch process all operations
        results = await self.batch_process_operations(operations)

        # Convert to result dict
        result_dict = {}
        for result in results:
            if result.success:
                result_dict[result.agent_id] = result.result
            else:
                result_dict[result.agent_id] = {"error": result.error}

        return result_dict

    async def shutdown(self) -> None:
        """Shutdown the async agent manager."""
        # Cancel pending operations
        self.pending_operations.clear()

        # Shutdown thread pool
        self.executor.shutdown(wait=True)

        logger.info("Async agent manager shut down")

    def get_statistics(self) -> Dict[str, Any]:
        """Get execution statistics."""
        return {
            **self.stats,
            "registered_agents": len(self.agent_configs),
            "max_workers": self.max_workers,
            "batch_size": self.batch_size,
        }


async def demonstrate_async_manager():
    """Demonstrate async agent manager functionality."""
    # Create manager
    manager = AsyncAgentManager(max_workers=4, batch_size=5)

    # Register agents
    for i in range(3):
        manager.register_agent(
            f"agent_{i}",
            {
                "name": f"AsyncAgent{i}",
                "grid_size": 5,
                "position": [2, 2],
            },
        )

    # Process individual operations
    print("Processing individual operations...")
    result = await manager.process_operation(
        "agent_0", "perceive", {"observation": {"cell_type": "empty"}}
    )
    print(f"Individual result: {result.success}, time: {result.execution_time_ms:.2f}ms")

    # Batch process operations
    print("\nBatch processing operations...")
    batch_ops = [
        ("agent_0", "update_beliefs", {}),
        ("agent_1", "update_beliefs", {}),
        ("agent_2", "update_beliefs", {}),
    ]
    batch_results = await manager.batch_process_operations(batch_ops)
    print(f"Batch processed {len(batch_results)} operations")

    # Parallel step execution
    print("\nExecuting parallel step...")
    observations = {
        "agent_0": {"cell_type": "empty"},
        "agent_1": {"cell_type": "goal"},
        "agent_2": {"cell_type": "empty"},
    }
    step_results = await manager.execute_parallel_step(observations)
    print(f"Parallel step completed for {len(step_results)} agents")

    # Get statistics
    stats = manager.get_statistics()
    print(f"\nStatistics: {stats}")

    # Shutdown
    await manager.shutdown()


if __name__ == "__main__":
    # Run demonstration
    asyncio.run(demonstrate_async_manager())
