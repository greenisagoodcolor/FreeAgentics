"""Thread-Safety Mechanisms for Belief Updates in Active Inference Framework.

This module implements comprehensive thread-safety measures to ensure concurrent
belief updates don't cause race conditions or inconsistent states.
"""

import asyncio
import logging
import threading
import time
from collections import defaultdict, deque
from contextlib import asynccontextmanager, contextmanager
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any, Callable, Dict, List, Optional, Set
from uuid import uuid4

logger = logging.getLogger(__name__)


class LockType(Enum):
    """Types of locks for different belief operations."""

    READ_ONLY = "read_only"
    READ_WRITE = "read_write"
    EXCLUSIVE = "exclusive"
    QUEUE_LOCK = "queue_lock"


@dataclass
class BeliefTransaction:
    """Represents a belief update transaction."""

    transaction_id: str = field(default_factory=lambda: str(uuid4()))
    agent_id: str = ""
    operation_type: str = ""
    timestamp: datetime = field(default_factory=datetime.now)
    lock_type: LockType = LockType.READ_WRITE
    timeout_seconds: float = 30.0
    metadata: Dict[str, Any] = field(default_factory=dict)

    # Transaction state
    started: bool = False
    committed: bool = False
    rolled_back: bool = False

    # Performance tracking
    start_time: Optional[float] = None
    end_time: Optional[float] = None


@dataclass
class ThreadSafetyMetrics:
    total_transactions: int = 0
    successful_transactions: int = 0
    failed_transactions: int = 0
    deadlock_detections: int = 0
    timeout_errors: int = 0
    lock_contention_count: int = 0
    avg_transaction_time_ms: float = 0.0
    max_concurrent_transactions: int = 0
    current_active_transactions: int = 0


class BeliefDeadlockDetector:
    """Detects and resolves potential deadlocks in belief updates."""

    def __init__(self, detection_interval: float = 5.0):
        """Initialize deadlock detector.

        Args:
            detection_interval: How often to check for deadlocks (seconds)
        """
        self.detection_interval = detection_interval
        self.dependency_graph: Dict[str, Set[str]] = defaultdict(set)
        self.transaction_agents: Dict[str, str] = {}
        self.last_check_time: float = time.time()
        self._lock = threading.RLock()

    def add_dependency(self, transaction_id: str, agent_id: str, depends_on: Set[str]) -> None:
        """Add a dependency for deadlock detection.

        Args:
            transaction_id: Transaction identifier
            agent_id: Agent being accessed
            depends_on: Set of agents this transaction depends on
        """
        with self._lock:
            self.transaction_agents[transaction_id] = agent_id
            self.dependency_graph[agent_id] = depends_on.copy()

    def remove_dependency(self, transaction_id: str) -> None:
        """Remove dependency when transaction completes.

        Args:
            transaction_id: Transaction identifier
        """
        with self._lock:
            if transaction_id in self.transaction_agents:
                agent_id = self.transaction_agents.pop(transaction_id)
                if agent_id in self.dependency_graph:
                    del self.dependency_graph[agent_id]

    def detect_deadlock(self) -> Optional[List[str]]:
        """Detect potential deadlocks.

        Returns:
            List of agent IDs involved in deadlock, or None if no deadlock
        """
        with self._lock:
            # Check if enough time has passed since last check
            current_time = time.time()
            if current_time - self.last_check_time < self.detection_interval:
                return None

            self.last_check_time = current_time

            # Use DFS to detect cycles in dependency graph
            visited = set()
            rec_stack = set()

            def has_cycle(node: str) -> Optional[List[str]]:
                if node in rec_stack:
                    # Found cycle - extract cycle path
                    cycle_start = node
                    cycle = [cycle_start]
                    return cycle

                if node in visited:
                    return None

                visited.add(node)
                rec_stack.add(node)

                for neighbor in self.dependency_graph.get(node, set()):
                    cycle = has_cycle(neighbor)
                    if cycle:
                        if cycle[0] != node:
                            cycle.insert(0, node)
                        return cycle

                rec_stack.remove(node)
                return None

            # Check each agent in the dependency graph
            for agent_id in self.dependency_graph:
                if agent_id not in visited:
                    cycle = has_cycle(agent_id)
                    if cycle:
                        return cycle

            return None


class BeliefLockManager:
    """Manages locks for belief state access."""

    def __init__(self):
        """Initialize lock manager."""
        self._agent_locks: Dict[str, threading.RLock] = {}
        self._lock_registry_lock = threading.RLock()
        self._read_write_locks: Dict[str, threading.RLock] = {}
        self._exclusive_locks: Dict[str, threading.Lock] = {}
        self._queue_locks: Dict[str, asyncio.Queue] = {}

        # Lock usage tracking
        self._lock_usage: Dict[str, int] = defaultdict(int)
        self._lock_wait_times: Dict[str, List[float]] = defaultdict(list)

    def get_agent_lock(self, agent_id: str) -> threading.RLock:
        """Get or create a lock for an agent.

        Args:
            agent_id: Agent identifier

        Returns:
            RLock for the agent
        """
        with self._lock_registry_lock:
            if agent_id not in self._agent_locks:
                self._agent_locks[agent_id] = threading.RLock()
            return self._agent_locks[agent_id]

    def get_read_write_lock(self, agent_id: str) -> threading.RLock:
        """Get or create a read-write lock for an agent.

        Args:
            agent_id: Agent identifier

        Returns:
            RLock for read-write operations
        """
        with self._lock_registry_lock:
            if agent_id not in self._read_write_locks:
                self._read_write_locks[agent_id] = threading.RLock()
            return self._read_write_locks[agent_id]

    def get_exclusive_lock(self, agent_id: str) -> threading.Lock:
        """Get or create an exclusive lock for an agent.

        Args:
            agent_id: Agent identifier

        Returns:
            Lock for exclusive operations
        """
        with self._lock_registry_lock:
            if agent_id not in self._exclusive_locks:
                self._exclusive_locks[agent_id] = threading.Lock()
            return self._exclusive_locks[agent_id]

    async def get_queue_lock(self, agent_id: str, maxsize: int = 100) -> asyncio.Queue:
        """Get or create a queue-based lock for an agent.

        Args:
            agent_id: Agent identifier
            maxsize: Maximum queue size

        Returns:
            Queue for serializing operations
        """
        if agent_id not in self._queue_locks:
            self._queue_locks[agent_id] = asyncio.Queue(maxsize=maxsize)
        return self._queue_locks[agent_id]

    @contextmanager
    def acquire_lock(self, agent_id: str, lock_type: LockType, timeout: float):
        """Acquire a lock with timeout and tracking.

        Args:
            agent_id: Agent identifier
            lock_type: Type of lock to acquire
            timeout: Timeout in seconds

        Yields:
            Lock context
        """
        start_time = time.time()
        lock_acquired = False

        try:
            if lock_type == LockType.READ_ONLY or lock_type == LockType.READ_WRITE:
                lock = self.get_read_write_lock(agent_id)
            elif lock_type == LockType.EXCLUSIVE:
                lock = self.get_exclusive_lock(agent_id)
            else:
                lock = self.get_agent_lock(agent_id)

            # Try to acquire lock with timeout
            lock_acquired = lock.acquire(timeout=timeout)

            if not lock_acquired:
                raise TimeoutError(
                    f"Failed to acquire {lock_type.value} lock for agent {agent_id} "
                    f"within {timeout}s"
                )

            # Track lock usage
            self._lock_usage[agent_id] += 1
            acquire_time = time.time() - start_time
            self._lock_wait_times[agent_id].append(acquire_time)

            # Limit history size
            if len(self._lock_wait_times[agent_id]) > 100:
                self._lock_wait_times[agent_id] = self._lock_wait_times[agent_id][-50:]

            yield lock

        finally:
            if lock_acquired:
                lock.release()

    def get_lock_statistics(self) -> Dict[str, Any]:
        """Get lock usage statistics.

        Returns:
            Dictionary with lock statistics
        """
        stats = {
            "total_agents_with_locks": len(self._agent_locks),
            "lock_usage_by_agent": dict(self._lock_usage),
            "avg_wait_times_by_agent": {},
        }

        for agent_id, wait_times in self._lock_wait_times.items():
            if wait_times:
                stats["avg_wait_times_by_agent"][agent_id] = sum(wait_times) / len(wait_times)

        return stats


class BeliefTransactionManager:
    """Manages belief update transactions with thread-safety."""

    def __init__(self, enable_deadlock_detection: bool = True):
        """Initialize transaction manager.

        Args:
            enable_deadlock_detection: Whether to enable deadlock detection
        """
        self.lock_manager = BeliefLockManager()
        self.deadlock_detector = BeliefDeadlockDetector() if enable_deadlock_detection else None

        # Transaction tracking
        self.active_transactions: Dict[str, BeliefTransaction] = {}
        self.transaction_history: deque = deque(maxlen=1000)
        self.metrics = ThreadSafetyMetrics()

        # Synchronization
        self._transaction_lock = threading.RLock()
        self._shutdown_event = threading.Event()

        # Background deadlock detection
        if self.deadlock_detector:
            self._deadlock_thread = threading.Thread(
                target=self._deadlock_detection_loop, daemon=True
            )
            self._deadlock_thread.start()

    def create_transaction(
        self,
        agent_id: str,
        operation_type: str,
        lock_type: LockType = LockType.READ_WRITE,
        timeout_seconds: float = 30.0,
        metadata: Optional[Dict[str, Any]] = None,
    ) -> BeliefTransaction:
        """Create a new belief update transaction.

        Args:
            agent_id: Target agent identifier
            operation_type: Type of operation (e.g., "update", "merge", "rollback")
            lock_type: Type of lock required
            timeout_seconds: Transaction timeout
            metadata: Additional metadata

        Returns:
            BeliefTransaction object
        """
        transaction = BeliefTransaction(
            agent_id=agent_id,
            operation_type=operation_type,
            lock_type=lock_type,
            timeout_seconds=timeout_seconds,
            metadata=metadata or {},
        )

        with self._transaction_lock:
            self.active_transactions[transaction.transaction_id] = transaction
            self.metrics.total_transactions += 1
            self.metrics.current_active_transactions = len(self.active_transactions)
            self.metrics.max_concurrent_transactions = max(
                self.metrics.max_concurrent_transactions, self.metrics.current_active_transactions
            )

        return transaction

    @contextmanager
    def execute_transaction(self, transaction: BeliefTransaction):
        """Execute a transaction with proper locking and error handling.

        Args:
            transaction: Transaction to execute

        Yields:
            Transaction context
        """
        transaction.start_time = time.time()
        transaction.started = True

        try:
            # Add to deadlock detection if enabled:
            if self.deadlock_detector:
                depends_on = set()  # Could be populated based on transaction dependencies
                self.deadlock_detector.add_dependency(
                    transaction.transaction_id, transaction.agent_id, depends_on
                )

            # Acquire lock
            with self.lock_manager.acquire_lock(
                transaction.agent_id, transaction.lock_type, transaction.timeout_seconds
            ):
                yield transaction

                # Mark as committed
                transaction.committed = True
            transaction.end_time = time.time()

            # Update metrics
            with self._transaction_lock:
                self.metrics.successful_transactions += 1

                # Update average transaction time
                transaction_time_ms = (transaction.end_time - transaction.start_time) * 1000
                total_successful = self.metrics.successful_transactions
                current_avg = self.metrics.avg_transaction_time_ms
                self.metrics.avg_transaction_time_ms = (
                    current_avg * (total_successful - 1) + transaction_time_ms
                ) / total_successful

        except TimeoutError as e:
            transaction.rolled_back = True
            transaction.end_time = time.time()

            with self._transaction_lock:
                self.metrics.failed_transactions += 1
                self.metrics.timeout_errors += 1

            logger.error(f"Transaction {transaction.transaction_id} timed out: {e}")
            raise

        except Exception as e:
            transaction.rolled_back = True
            transaction.end_time = time.time()

            with self._transaction_lock:
                self.metrics.failed_transactions += 1

            logger.error(f"Transaction {transaction.transaction_id} failed: {e}")
            raise

        finally:
            # Clean up
            if self.deadlock_detector:
                self.deadlock_detector.remove_dependency(transaction.transaction_id)

            with self._transaction_lock:
                if transaction.transaction_id in self.active_transactions:
                    completed_transaction = self.active_transactions.pop(transaction.transaction_id)
                    self.transaction_history.append(completed_transaction)

                self.metrics.current_active_transactions = len(self.active_transactions)

    @asynccontextmanager
    async def execute_async_transaction(self, transaction: BeliefTransaction):
        """Execute an async transaction with queue-based locking.

        Args:
            transaction: Transaction to execute

        Yields:
            Transaction context
        """
        transaction.start_time = time.time()
        transaction.started = True

        try:
            # Get queue-based lock for async operations
            queue = await self.lock_manager.get_queue_lock(transaction.agent_id)

            # Add transaction to queue
            await asyncio.wait_for(
                queue.put(transaction.transaction_id), timeout=transaction.timeout_seconds
            )

            yield transaction

            # Mark as committed
            transaction.committed = True
            transaction.end_time = time.time()

            # Update metrics
            with self._transaction_lock:
                self.metrics.successful_transactions += 1

        except Exception:
            # Mark as failed
            transaction.failed = True
            transaction.end_time = time.time()
            with self._transaction_lock:
                self.metrics.failed_transactions += 1
            raise

        finally:
            # Remove from queue
            try:
                await asyncio.wait_for(queue.get(), timeout=1.0)
                queue.task_done()
            except asyncio.TimeoutError:
                pass  # Queue cleanup might fail, that's ok

            # Clean up completed transaction
            with self._transaction_lock:
                if transaction.transaction_id in self.active_transactions:
                    completed_transaction = self.active_transactions.pop(transaction.transaction_id)
                    self.transaction_history.append(completed_transaction)

                self.metrics.current_active_transactions = len(self.active_transactions)

    def _deadlock_detection_loop(self) -> None:
        """Background thread for deadlock detection."""
        while not self._shutdown_event.is_set():
            try:
                if self.deadlock_detector:
                    deadlock_cycle = self.deadlock_detector.detect_deadlock()
                    if deadlock_cycle:
                        self.metrics.deadlock_detections += 1
                        logger.warning(f"Deadlock detected involving agents: {deadlock_cycle}")

                        # Simple deadlock resolution: abort youngest transaction
                        self._resolve_deadlock(deadlock_cycle)

                time.sleep(1.0)  # Check every second
            except Exception as e:
                logger.error(f"Error in deadlock detection: {e}")
                time.sleep(5.0)  # Wait longer on error

    def _resolve_deadlock(self, deadlock_cycle: List[str]) -> None:
        """Resolve deadlock by aborting transactions.

        Args:
            deadlock_cycle: List of agent IDs involved in deadlock
        """
        try:
            # Find transactions involved in deadlock
            involved_transactions = []

            with self._transaction_lock:
                for transaction in self.active_transactions.values():
                    if transaction.agent_id in deadlock_cycle:
                        involved_transactions.append(transaction)

                if involved_transactions:
                    # Sort by creation time and abort the newest
                    involved_transactions.sort(key=lambda t: t.timestamp)
                    transaction_to_abort = involved_transactions[-1]

                    logger.warning(
                        f"Resolving deadlock by aborting transaction "
                        f"{transaction_to_abort.transaction_id}"
                    )

                    # Mark transaction for rollback
                    transaction_to_abort.rolled_back = True
        except Exception as e:
            logger.error(f"Error resolving deadlock: {e}")

    def get_metrics(self) -> ThreadSafetyMetrics:
        """Get current thread-safety metrics.

        Returns:
            ThreadSafetyMetrics object
        """
        with self._transaction_lock:
            return ThreadSafetyMetrics(
                total_transactions=self.metrics.total_transactions,
                successful_transactions=self.metrics.successful_transactions,
                failed_transactions=self.metrics.failed_transactions,
                deadlock_detections=self.metrics.deadlock_detections,
                timeout_errors=self.metrics.timeout_errors,
                lock_contention_count=self.metrics.lock_contention_count,
                avg_transaction_time_ms=self.metrics.avg_transaction_time_ms,
                max_concurrent_transactions=self.metrics.max_concurrent_transactions,
                current_active_transactions=self.metrics.current_active_transactions,
            )

    def shutdown(self) -> None:
        """Shutdown the transaction manager."""
        self._shutdown_event.set()

        if hasattr(self, "_deadlock_thread"):
            self._deadlock_thread.join(timeout=5.0)


# Global transaction manager instance
_global_transaction_manager: Optional[BeliefTransactionManager] = None


def get_belief_transaction_manager(
    enable_deadlock_detection: bool = True,
) -> BeliefTransactionManager:
    """Get or create the global belief transaction manager.

    Args:
        enable_deadlock_detection: Whether to enable deadlock detection

    Returns:
        BeliefTransactionManager instance
    """
    global _global_transaction_manager
    if _global_transaction_manager is None:
        _global_transaction_manager = BeliefTransactionManager(enable_deadlock_detection)

    return _global_transaction_manager


def reset_belief_transaction_manager() -> None:
    """Reset the global belief transaction manager."""
    global _global_transaction_manager
    if _global_transaction_manager:
        _global_transaction_manager.shutdown()
    _global_transaction_manager = None


# Decorators for thread-safe belief operations
def thread_safe_belief_operation(lock_type: LockType = LockType.READ_WRITE, timeout: float = 30.0):
    """Decorator for thread-safe belief operations.

    Args:
        lock_type: Type of lock to acquire
        timeout: Operation timeout

    Returns:
        Decorated function
    """

    def decorator(func: Callable) -> Callable:
        def wrapper(*args, **kwargs):
            # Extract agent_id from arguments
            agent_id = None
            if args and hasattr(args[0], "agent_id"):
                agent_id = args[0].agent_id
            elif "agent_id" in kwargs:
                agent_id = kwargs["agent_id"]
            else:
                # Try to find agent_id in first few arguments
                for arg in args[:3]:
                    if isinstance(arg, str) and len(arg) > 0:
                        agent_id = arg
                        break

            if not agent_id:
                raise ValueError("Could not determine agent_id for thread-safe operation")

            # Create and execute transaction
            manager = get_belief_transaction_manager()
            transaction = manager.create_transaction(

def async_thread_safe_belief_operation(
    lock_type: LockType = LockType.READ_WRITE, timeout: float = 30.0
):
    """Decorator for async thread-safe belief operations.

    Args:
        lock_type: Type of lock to acquire
        timeout: Operation timeout

        Decorated async function
    """

    def decorator(func: Callable) -> Callable:
        async def wrapper(*args, **kwargs):
            # Extract agent_id from arguments
            agent_id = None
            if args and hasattr(args[0], "agent_id"):
                agent_id = args[0].agent_id
            elif "agent_id" in kwargs:
                agent_id = kwargs["agent_id"]
            else:
                # Try to find agent_id in first few arguments
                for arg in args[:3]:
                    if isinstance(arg, str) and len(arg) > 0:
                        agent_id = arg
                        break

            if not agent_id:
                raise ValueError("Could not determine agent_id for async thread-safe operation")

            # Create and execute async transaction
            manager = get_belief_transaction_manager()
            transaction = manager.create_transaction(
                agent_id=agent_id,
                operation_type=func.__name__,
                lock_type=lock_type,
                timeout_seconds=timeout,
            )

            async with manager.execute_async_transaction(transaction):
                return await func(*args, **kwargs)

        return wrapper

    return decorator
