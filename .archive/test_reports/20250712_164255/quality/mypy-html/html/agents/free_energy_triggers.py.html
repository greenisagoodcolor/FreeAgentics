"""Free Energy Recalculation Triggers for Active Inference Agents.

This module implements event-driven triggers that automatically initiate
free energy recalculation when belief states change, ensuring optimal
inference performance while avoiding excessive computation.
"""

import asyncio
import logging
import time
from abc import ABC, abstractmethod
from collections import deque
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any, Callable, Dict, List, Optional

import numpy as np

logger = logging.getLogger(__name__)


class TriggerCondition(Enum):
    """Types of conditions that can trigger free energy recalculation."""

    BELIEF_CHANGE_THRESHOLD = "belief_change_threshold"
    OBSERVATION_NOVELTY = "observation_novelty"
    TIME_INTERVAL = "time_interval"
    BATCH_SIZE = "batch_size"
    ENTROPY_CHANGE = "entropy_change"
    PREDICTION_ERROR = "prediction_error"
    GMN_UPDATE = "gmn_update"
    CONVERSATION_UPDATE = "conversation_update"
    BELIEF_OPERATION_APPLIED = "belief_operation_applied"
    CONDITIONAL_RULE_TRIGGERED = "conditional_rule_triggered"


@dataclass
class TriggerEvent:
    """Represents a trigger event for free energy recalculation."""

    event_id: str
    trigger_type: TriggerCondition
    timestamp: datetime
    agent_id: str
    data: Dict[str, Any] = field(default_factory=dict)
    priority: int = 1  # 1=low, 2=medium, 3=high
    processed: bool = False


@dataclass
class TriggerConfig:
    """Configuration for trigger conditions."""

    belief_change_threshold: float = 0.01
    observation_novelty_threshold: float = 0.1
    time_interval_ms: int = 1000
    batch_size: int = 10
    entropy_change_threshold: float = 0.05
    prediction_error_threshold: float = 0.1
    min_time_between_triggers_ms: int = 100
    enable_adaptive_thresholds: bool = True
    max_trigger_rate_per_second: float = 10.0

    # Priority mapping
    critical_trigger_priority: int = 10
    important_trigger_priority: int = 5
    normal_trigger_priority: int = 1
    manual_trigger_priority: int = 20


class AgentTriggerState:
    """Tracks trigger state for a single agent."""

    def __init__(self, agent_id: str):
        self.agent_id = agent_id
        self.last_belief: Optional[np.ndarray] = None
        self.last_trigger_time: float = time.time()
        self.last_recalc_time: float = time.time()
        self.pending_events: deque = deque(maxlen=100)
        self.trigger_count: int = 0
        self.recalc_count: int = 0
        self.current_belief: Optional[np.ndarray] = None
        self.metadata: Dict[str, Any] = {}


class TriggerStrategy(ABC):
    """Abstract base class for trigger strategies."""

    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.enabled = config.get("enabled", True)

    @abstractmethod
    def should_trigger(self, context: Dict[str, Any]) -> bool:
        """Check if trigger condition is met."""

    @abstractmethod
    def get_priority(self, context: Dict[str, Any]) -> int:
        """Get trigger priority based on context."""


class BeliefChangeStrategy(TriggerStrategy):
    """Triggers when belief state changes exceed threshold."""

    def should_trigger(self, context: Dict[str, Any]) -> bool:
        """Check if belief change exceeds threshold."""
        if not self.enabled:
            return False

        current_belief = context.get("current_belief")
        previous_belief = context.get("previous_belief")

        if current_belief is None or previous_belief is None:
            return False

        threshold = self.config.get("threshold", 0.01)
        change = np.linalg.norm(current_belief - previous_belief)

        return change > threshold

    def get_priority(self, context: Dict[str, Any]) -> int:
        """Get priority based on magnitude of belief change."""
        current_belief = context.get("current_belief")
        previous_belief = context.get("previous_belief")

        if current_belief is None or previous_belief is None:
            return 1

        change = np.linalg.norm(current_belief - previous_belief)

        if change > 0.1:
            return 3  # High priority
        elif change > 0.05:
            return 2  # Medium priority
        else:
            return 1  # Low priority


class TimeIntervalStrategy(TriggerStrategy):
    """Triggers at regular time intervals."""

    def should_trigger(self, context: Dict[str, Any]) -> bool:
        """Check if time interval has elapsed."""
        if not self.enabled:
            return False

        last_trigger_time = context.get("last_trigger_time", 0)
        interval_ms = self.config.get("interval_ms", 1000)

        current_time = time.time() * 1000
        return (current_time - last_trigger_time) > interval_ms

    def get_priority(self, context: Dict[str, Any]) -> int:
        """Time-based triggers have normal priority."""
        return 1


class FreeEnergyTriggerManager:
    """Manages free energy recalculation triggers for multiple agents."""

    def __init__(self, config: Optional[TriggerConfig] = None):
        """Initialize trigger manager."""
        self.config = config or TriggerConfig()
        self.agent_states: Dict[str, AgentTriggerState] = {}
        self.strategies: List[TriggerStrategy] = []
        self.event_queue: deque = deque()
        self.callbacks: List[Callable] = []
        self.is_running = False
        self._task: Optional[asyncio.Task] = None

        # Setup default strategies
        self._setup_default_strategies()

    def _setup_default_strategies(self):
        """Setup default trigger strategies."""
        # Belief change strategy
        belief_config = {
            "enabled": True,
            "threshold": self.config.belief_change_threshold,
        }
        self.strategies.append(BeliefChangeStrategy(belief_config))

        # Time interval strategy
        time_config = {"enabled": True, "interval_ms": self.config.time_interval_ms}
        self.strategies.append(TimeIntervalStrategy(time_config))

    def register_agent(self, agent_id: str):
        """Register an agent for trigger monitoring."""
        if agent_id not in self.agent_states:
            self.agent_states[agent_id] = AgentTriggerState(agent_id)
            logger.info(f"Registered agent {agent_id} for trigger monitoring")

    def unregister_agent(self, agent_id: str):
        """Unregister an agent from trigger monitoring."""
        if agent_id in self.agent_states:
            del self.agent_states[agent_id]
            logger.info(f"Unregistered agent {agent_id} from trigger monitoring")

    def register_callback(self, callback: Callable):
        """Register a callback for trigger events."""
        self.callbacks.append(callback)

    def on_belief_update(
        self, agent_id: str, new_belief: np.ndarray, metadata: Dict[str, Any] = None
    ) -> List[TriggerEvent]:
        """Handle belief update and check for triggers."""
        if agent_id not in self.agent_states:
            self.register_agent(agent_id)

        agent_state = self.agent_states[agent_id]
        previous_belief = agent_state.current_belief
        agent_state.current_belief = new_belief

        # Create context for trigger evaluation
        context = {
            "agent_id": agent_id,
            "current_belief": new_belief,
            "previous_belief": previous_belief,
            "last_trigger_time": agent_state.last_trigger_time * 1000,
            "metadata": metadata or {},
        }

        # Check all strategies
        triggered_events = []
        for strategy in self.strategies:
            if strategy.should_trigger(context):
                event = TriggerEvent(
                    event_id=f"trigger_{agent_id}_{int(time.time() * 1000)}",
                    trigger_type=TriggerCondition.BELIEF_CHANGE_THRESHOLD,
                    timestamp=datetime.now(),
                    agent_id=agent_id,
                    data=context,
                    priority=strategy.get_priority(context),
                )
                triggered_events.append(event)
                self.queue_event(event)

        if triggered_events:
            agent_state.last_trigger_time = time.time()
            agent_state.trigger_count += len(triggered_events)

        return triggered_events

    def check_triggers(self, context: Dict[str, Any]) -> Optional[TriggerEvent]:
        """Check if any triggers should fire based on context."""
        for strategy in self.strategies:
            if strategy.should_trigger(context):
                agent_id = context.get("agent_id", "unknown")
                event = TriggerEvent(
                    event_id=f"trigger_{agent_id}_{int(time.time() * 1000)}",
                    trigger_type=TriggerCondition.GMN_UPDATE,
                    timestamp=datetime.now(),
                    agent_id=agent_id,
                    data=context,
                    priority=strategy.get_priority(context),
                )
                return event
        return None

    def queue_event(self, event: TriggerEvent):
        """Queue a trigger event for processing."""
        self.event_queue.append(event)

        # Update agent state
        if event.agent_id in self.agent_states:
            self.agent_states[event.agent_id].pending_events.append(event)

    async def start(self):
        """Start the trigger manager."""
        if not self.is_running:
            self.is_running = True
            self._task = asyncio.create_task(self._process_events())
            logger.info("Free energy trigger manager started")

    async def stop(self):
        """Stop the trigger manager."""
        self.is_running = False
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass
            logger.info("Free energy trigger manager stopped")

    async def _process_events(self):
        """Process trigger events from the queue."""
        while self.is_running:
            if self.event_queue:
                event = self.event_queue.popleft()
                await self._handle_event(event)
            else:
                await asyncio.sleep(0.01)  # Short sleep when no events

    async def _handle_event(self, event: TriggerEvent):
        """Handle a single trigger event."""
        try:
            # Execute all registered callbacks
            for callback in self.callbacks:
                try:
                    if asyncio.iscoroutinefunction(callback):
                        await callback(event)
                    else:
                        callback(event)
                except Exception as e:
                    logger.error(f"Error in trigger callback: {e}")

            # Mark event as processed
            event.processed = True

            # Update agent statistics
            if event.agent_id in self.agent_states:
                self.agent_states[event.agent_id].recalc_count += 1
                self.agent_states[event.agent_id].last_recalc_time = time.time()

        except Exception as e:
            logger.error(f"Error handling trigger event {event.event_id}: {e}")

    def get_agent_state(self, agent_id: str) -> Optional[Dict[str, Any]]:
        """Get current state for an agent."""
        if agent_id not in self.agent_states:
            return None

        state = self.agent_states[agent_id]
        return {
            "agent_id": agent_id,
            "current_belief": state.current_belief,
            "last_belief": state.last_belief,
            "trigger_count": state.trigger_count,
            "recalc_count": state.recalc_count,
            "last_trigger_time": state.last_trigger_time,
            "last_recalc_time": state.last_recalc_time,
            "pending_events": len(state.pending_events),
            "metadata": state.metadata,
        }

    def get_statistics(self) -> Dict[str, Any]:
        """Get trigger manager statistics."""
        total_triggers = sum(state.trigger_count for state in self.agent_states.values())
        total_recalcs = sum(state.recalc_count for state in self.agent_states.values())

        return {
            "registered_agents": len(self.agent_states),
            "total_triggers": total_triggers,
            "total_recalculations": total_recalcs,
            "pending_events": len(self.event_queue),
            "active_strategies": len(self.strategies),
            "is_running": self.is_running,
        }


def create_trigger_manager(
    config: Optional[Dict[str, Any]] = None,
) -> FreeEnergyTriggerManager:
    """Create a trigger manager with optional configuration."""
    trigger_config = TriggerConfig()

    if config:
        for key, value in config.items():
            if hasattr(trigger_config, key):
                setattr(trigger_config, key, value)

    return FreeEnergyTriggerManager(trigger_config)


# Example usage
async def demonstrate_triggers():
    """Demonstrate trigger functionality."""
    # Create trigger manager
    manager = create_trigger_manager()

    # Register an agent
    agent_id = "demo_agent"
    manager.register_agent(agent_id)

    # Register a callback
    def trigger_callback(event: TriggerEvent):
        print(f"Trigger fired: {event.trigger_type.value} for agent {event.agent_id}")

    manager.register_callback(trigger_callback)

    # Start manager
    await manager.start()

    # Simulate belief updates
    for i in range(5):
        belief = np.random.rand(4)
        belief = belief / np.sum(belief)  # Normalize

        events = manager.on_belief_update(agent_id, belief)
        print(f"Belief update {i} triggered {len(events)} events")

        await asyncio.sleep(0.1)

    # Get statistics
    stats = manager.get_statistics()
    print(f"Manager statistics: {stats}")

    # Stop manager
    await manager.stop()


if __name__ == "__main__":
    asyncio.run(demonstrate_triggers())
