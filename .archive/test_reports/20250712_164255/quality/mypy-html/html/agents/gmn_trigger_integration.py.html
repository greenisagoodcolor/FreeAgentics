"""Integration layer between GMN Manager and Free Energy Trigger system.

This module provides seamless integration between GMN specification changes
and free energy recalculation triggers, ensuring optimal inference performance.
"""

import asyncio
import logging
from typing import Any, Callable, Dict, List, Optional

import numpy as np

from agents.free_energy_triggers import (
    TriggerCondition,
    TriggerEvent,
    create_trigger_manager,
)
from agents.gmn_manager import GMNManager

logger = logging.getLogger(__name__)


class GMNTriggerIntegration:
    """Integrates GMN specification changes with free energy trigger system."""

    def __init__(self, agent_id, trigger_manager=None, gmn_manager=None):
        """Initialize GMN-trigger integration.

        Args:
            agent_id: Agent identifier
            trigger_manager: Optional trigger manager instance
            gmn_manager: Optional GMN manager instance
        """
        self.agent_id = agent_id
        self.trigger_manager = trigger_manager or create_trigger_manager()
        self.gmn_manager = gmn_manager or GMNManager(agent_id)

        # Register agent with trigger manager
        self.trigger_manager.register_agent(agent_id)

        # Track current belief state
        self.current_belief: Optional[np.ndarray] = None
        self.free_energy_callbacks: List[Callable] = []

        # Statistics
        self.stats = {
            "gmn_triggered_recalculations": 0,
            "belief_operation_triggers": 0,
            "conditional_rule_triggers": 0,
            "total_recalculations": 0,
        }

        # Setup integration
        self._setup_integration()

    def _setup_integration(self):
        """Setup integration between GMN and trigger systems."""
        # Register callback for free energy recalculation
        self.trigger_manager.register_callback(self._handle_trigger_event)
        logger.info(f"GMN-Trigger integration setup for agent {self.agent_id}")

    async def _handle_trigger_event(self, event):
        """Handle trigger events and perform free energy recalculation.

        Args:
            event: Trigger event to handle
        """
        try:
            if event.agent_id != self.agent_id:
                return

            logger.debug(
                f"Handling trigger event {event.trigger_type.value} for agent {self.agent_id}"
            )

            # Get current GMN model if available
            gmn_model = await self.gmn_manager.get_parsed_gmn_model()

            # Calculate free energy
            free_energy = await self._calculate_free_energy(gmn_model, event)

            # Update statistics
            self.stats["total_recalculations"] += 1

            if event.trigger_type == TriggerCondition.GMN_UPDATE:
                self.stats["gmn_triggered_recalculations"] += 1
            elif event.trigger_type == TriggerCondition.BELIEF_OPERATION_APPLIED:
                self.stats["belief_operation_triggers"] += 1
            elif event.trigger_type == TriggerCondition.CONDITIONAL_RULE_TRIGGERED:
                self.stats["conditional_rule_triggers"] += 1

            # Execute registered callbacks
            for callback in self.free_energy_callbacks:
                try:
                    if asyncio.iscoroutinefunction(callback):
                        await callback(self.agent_id, free_energy, event)
                    else:
                        callback(self.agent_id, free_energy, event)
                except Exception as e:
                    logger.error(f"Error in free energy callback: {e}")

            logger.info(f"Free energy recalculated for agent {self.agent_id}: {free_energy:.4f}")

        except Exception as e:
            logger.error(f"Error handling trigger event for agent {self.agent_id}: {e}")

    async def _calculate_free_energy(
        self, gmn_model: Optional[Dict[str, Any]], event: TriggerEvent
    ) -> float:
        """Calculate variational free energy using real PyMDP operations.

        Args:
            gmn_model: Parsed GMN model
            event: Trigger event that initiated calculation

        Returns:
            Calculated free energy value

        Raises:
            RuntimeError: When PyMDP is not available or calculation fails
        """
        # CRITICAL: No graceful fallbacks - fail hard if PyMDP unavailable
        try:
            pass
        except ImportError:
            raise RuntimeError("PyMDP is required for free energy calculations but not available")

        # Get current belief state
        agent_state = self.trigger_manager.get_agent_state(self.agent_id)
        if not agent_state:
            raise RuntimeError(
                f"No agent state found for {self.agent_id} - cannot calculate free energy"
            )

        current_belief = agent_state["current_belief"]
        if current_belief is None:
            raise RuntimeError(f"No belief state available for agent {self.agent_id}")

        # Require GMN model for proper calculation
        if not gmn_model:
            raise RuntimeError("GMN model required for free energy calculation")

        # Use real PyMDP operations for calculation
        free_energy = self._calculate_gmn_free_energy_pymdp(gmn_model, current_belief)

        return free_energy

    def _calculate_gmn_free_energy_pymdp(
        self, gmn_model: Dict[str, Any], belief: np.ndarray
    ) -> float:
        """Calculate free energy using real PyMDP operations.

        Args:
            gmn_model: Parsed GMN model with PyMDP-compatible matrices
            belief: Current belief state

        Returns:
            Free energy value calculated using PyMDP

        Raises:
            RuntimeError: If PyMDP operations fail
        """
        try:
            from pymdp import utils as pymdp_utils
        except ImportError:
            raise RuntimeError("PyMDP required but not available")

        # Extract model components - MUST be present
        A_matrices = gmn_model.get("A")
        if not A_matrices:
            raise RuntimeError(
                "Observation model (A matrices) required for free energy calculation"
            )

        gmn_model.get("B")
        C_vectors = gmn_model.get("C")
        D_vectors = gmn_model.get("D")  # Prior beliefs

        # Convert to PyMDP format
        try:
            # Ensure proper format for PyMDP
            if isinstance(A_matrices, list) and len(A_matrices) > 0:
                A = pymdp_utils.to_obj_array(A_matrices)
            else:
                raise RuntimeError("Invalid A matrices format for PyMDP")

            # Get current observation from event data if available
            observation = None
            if hasattr(self, "current_observation"):
                observation = self.current_observation

            # Use PyMDP's actual free energy calculation
            if observation is not None:
                # Calculate free energy with actual observation
                obs_idx = observation if isinstance(observation, int) else np.argmax(observation)

                # Get likelihood for current observation
                likelihood = A[0][obs_idx, :]

                # Calculate negative log likelihood term
                log_likelihood = np.sum(belief * np.log(likelihood + 1e-16))

                # Calculate entropy of beliefs
                -np.sum(belief * np.log(belief + 1e-16))

                # Calculate KL divergence from prior if available
                kl_div = 0.0
                if D_vectors and len(D_vectors) > 0:
                    prior = np.array(D_vectors[0])
                    if len(prior) == len(belief):
                        kl_div = np.sum(belief * (np.log(belief + 1e-16) - np.log(prior + 1e-16)))

                # Calculate preference term if available
                expected_utility = 0.0
                if C_vectors and len(C_vectors) > 0:
                    C = np.array(C_vectors[0])
                    if len(C) == len(belief):
                        expected_utility = np.sum(belief * C)

                # Variational free energy: F = KL[Q(s)||P(s)] - E_Q[log P(o|s)] - E_Q[log C]
                free_energy = kl_div - log_likelihood - expected_utility

            else:
                # Without observation, calculate expected free energy
                # This is the complexity term only
                if D_vectors and len(D_vectors) > 0:
                    prior = np.array(D_vectors[0])
                    free_energy = np.sum(belief * (np.log(belief + 1e-16) - np.log(prior + 1e-16)))
                else:
                    # Use uniform prior
                    uniform_prior = np.ones_like(belief) / len(belief)
                    free_energy = np.sum(
                        belief * (np.log(belief + 1e-16) - np.log(uniform_prior + 1e-16))
                    )

            return float(free_energy)

        except Exception as e:
            raise RuntimeError(f"PyMDP free energy calculation failed: {e}")

    async def on_gmn_specification_changed(self, specification_id, changes=None):
        """Handle GMN specification changes.

        Args:
            specification_id: ID of changed specification
            changes: Dictionary describing the changes
        """
        try:
            logger.info(f"GMN specification changed for agent {self.agent_id}: {specification_id}")

            # Trigger free energy recalculation
            context = {
                "agent_id": self.agent_id,
                "gmn_specification_changed": True,
                "specification_id": specification_id,
                "changes": changes,
            }

            # Check triggers
            event = self.trigger_manager.check_triggers(context)
            if event:
                self.trigger_manager.queue_event(event)

        except Exception as e:
            logger.error(f"Error handling GMN specification change: {e}")


def create_gmn_trigger_integration(
    agent_id: str, trigger_config: Optional[Dict[str, Any]] = None
) -> GMNTriggerIntegration:
    """Create a GMN-trigger integration instance.

    Args:
        agent_id: Agent identifier
        trigger_config: Optional trigger manager configuration

    Returns:
        GMN-trigger integration instance
    """
    trigger_manager = create_trigger_manager(trigger_config)
    gmn_manager = GMNManager(agent_id)

    return GMNTriggerIntegration(agent_id, trigger_manager, gmn_manager)
