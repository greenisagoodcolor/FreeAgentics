    # Fallback error handling implementations
    class ErrorHandler:
        def __init__(self, agent_id: str) -> None:
            self.agent_id = agent_id

        def handle_error(self, error: Exception, context: str) -> None:
            logging.error(f"Agent {self.agent_id} error in {context}: {error}")

        def get_error_summary(self) -> Dict[str, int]:
            return {"total_errors": 0}

    class PyMDPErrorHandler:
        def __init__(self, agent_id: str) -> None:
            self.agent_id = agent_id

        def safe_execute(self, operation_name: str, operation: Callable[[], Any], fallback: Callable[[], Any]) -> Tuple[bool, Any, Optional[str]]:
            try:
                return True, operation(), None
            except Exception as e:
                return False, fallback(), str(e)

        def get_error_report(self) -> Dict[str, int]:
            return {"total_errors": 0}

    class InferenceError(Exception):

        return obs

    def validate_action(action: Any, valid_actions: List[Any] = None) -> Any:
        if valid_actions is None:
            return action
        return action if action in valid_actions else (valid_actions[0] if valid_actions else action)

    def validate_pymdp_matrices(A: Any, B: Any, C: Any, D: Any) -> Tuple[bool, str]:
        return True, "Mock validation"

    def safe_numpy_conversion(value: Any, target_type: type = int, default: Any = None) -> Any:
        if default is None:
            default = target_type()
        try:
            return target_type(value)
        except (ValueError, TypeError):
            return default

    def safe_array_index(array: Any, index: Any, default: Any) -> Any:
        try:
            return array[index]
        except (IndexError, KeyError, TypeError):
            return default

    def safe_pymdp_operation(operation_name: str, default_value: Any = None):
        """Decorator for safe PyMDP operations."""
        def decorator(func):
            def wrapper(*args, **kwargs):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    logging.error(f"PyMDP operation {operation_name} failed: {e}")
                    return default_value
            return wrapper
        return decorator

    def with_error_handling(operation_name: str, fallback_result: Any = None):
        """Decorator for error handling with fallback result."""
        def decorator(func):
            def wrapper(*args, **kwargs):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    logging.error(f"Operation {operation_name} failed: {e}")
                    return fallback_result
            return wrapper
        return decorator

def safe_array_to_int(value: Any, default: int = 0) -> int:
    """Safely convert array or scalar value to integer."""
    try:
        if hasattr(value, "ndim"):
            # This is a numpy array
            if value.ndim == 0:
                # 0-dimensional array (scalar)
                return int(value.item())
            elif value.size == 0:
                raise ValueError("Empty array cannot be converted to integer")
            elif value.size == 1:
                # Single element array
                return int(value.item())
            else:
                # Multi-element array - take first element
                return int(value.flat[0])  # Use flat to handle any shape
        elif hasattr(value, "__len__") and hasattr(value, "__getitem__"):
            # Other array-like objects (lists, tuples, etc.)
            if len(value) == 0:
                raise ValueError("Empty array cannot be converted to integer")
            else:
                return int(value[0])
        elif hasattr(value, "item"):
            # numpy scalar
            return int(value.item())
        else:
            # regular scalar
            return int(value)
    except (TypeError, ValueError, IndexError) as e:
        logger.warning(
            f"Cannot convert {type(value)} value {value} to integer: {e}, "
            f"using default {default}"
        )
        return default


@dataclass
class AgentConfig:
    """Configuration for an Active Inference agent."""

    agent_id: str
    name: str
    use_pymdp: bool = True
    use_llm: bool = False
    performance_mode: str = "balanced"  # "fast", "balanced", "accurate"
    selective_update_interval: int = 1
    planning_horizon: int = 3
    enable_observability: bool = True


class ActiveInferenceAgent(ABC):
    """Base class for Active Inference agents using PyMDP."""

    def __init__(self, agent_id: str, name: str, config: Dict[str, Any]):
        """Initialize the Active Inference agent."""
        self.agent_id = agent_id
        self.name = name
        self.config = config
        self.is_active = False
        self.created_at = datetime.now()
        self.last_action_at: Optional[datetime] = None
        self.total_steps = 0

        # Performance configuration
        self.performance_mode = config.get("performance_mode", "balanced")
        self.selective_update_interval = config.get("selective_update_interval", 1)
        self.matrix_cache: Dict[str, Any] = {}  # Cache for normalized matrices

        # Active Inference components
        self.pymdp_agent: Optional[Any] = None  # PyMDP agent instance
        self.beliefs: Dict[str, Any] = {}  # Beliefs about hidden states
        self.preferences: Dict[str, Any] = {}  # Preferred observations
        self.policies: List[Any] = []  # Available action policies
        self.current_observation: Optional[Any] = None

        # GMN and LLM integration
        self.gmn_spec: Optional[Dict[str, Any]] = None
        if LLM_AVAILABLE and config.get("use_llm", False):
            llm_config_dict = config.get("llm_config", {})
            llm_config = LocalLLMConfig()
            for key, value in llm_config_dict.items():
                if hasattr(llm_config, key):
                    setattr(llm_config, key, value)
            self.llm_manager = LocalLLMManager(llm_config)
        else:
            self.llm_manager: Optional[Any] = None

        # Initialize PyMDP if available:
        if PYMDP_AVAILABLE and config.get("use_pymdp", True):
            self._initialize_pymdp()

        # Metrics
        self.metrics = {
            "total_observations": 0,
            "total_actions": 0,
            "avg_free_energy": 0.0,
            "belief_entropy": 0.0,
        }

        # Observability integration
        self.observability_enabled = OBSERVABILITY_AVAILABLE and config.get(
            "enable_observability", True
        )

        # Error handling
        self.error_handler = ErrorHandler(self.agent_id)
        self.pymdp_error_handler = PyMDPErrorHandler(self.agent_id)

        # Performance optimization
        self.performance_optimizer = PerformanceOptimizer()
        self.performance_metrics: Dict[str, Any] = {}

        logger.info(
            f"Created agent {self.agent_id} ({self.name}) - PyMDP: {PYMDP_AVAILABLE}, "
            f"Performance Mode: {self.performance_mode}"
        )

    def _initialize_pymdp(self) -> None:
        """Initialize PyMDP agent with default configuration."""
        if not PYMDP_AVAILABLE:
            logger.warning("PyMDP not available, skipping initialization")
            return

        try:
            # Default grid world setup
            num_states = 16  # 4x4 grid
            num_obs = 5  # empty, obstacle, goal, agent, out_of_bounds
            num_actions = 5  # up, down, left, right, stay

            # Create default observation model (A matrix)
            A = np.eye(num_obs, num_states)
            A = utils.norm_dist(A)

            # Create default transition model (B matrix)
            B = np.zeros((num_states, num_states, num_actions))
            for action in range(num_actions):
                B[:, :, action] = np.eye(num_states)

            # Create default preferences (C vector)
            C = np.zeros(num_obs)
            C[0] = 1.0  # Prefer empty spaces

            # Create default prior (D vector)
            D = utils.norm_dist(np.ones(num_states))

            # Create PyMDP agent
            self.pymdp_agent = PyMDPAgent(
                A=A,
                B=B,
                C=C,
                D=D,
                policy_len=self._get_policy_length(),
                inference_algo="VANILLA",
                use_states_info_gain=True,
                use_param_info_gain=self._get_param_info_gain(),
                use_utility=True,
                gamma=self._get_gamma(),
                alpha=self._get_alpha(),
                inference_horizon=1,
                save_belief_hist=False,
            )

            logger.info(f"Initialized default PyMDP agent for {self.agent_id}")
        except Exception as e:
            logger.error(f"Failed to initialize PyMDP: {e}")
            self.pymdp_agent = None

    def _get_policy_length(self) -> int:
        """Get adaptive policy length based on performance mode."""
        if self.performance_mode == "fast":
            return 1
        elif self.performance_mode == "balanced":
            return 2
        else:  # "accurate"
            return 3

    def _get_param_info_gain(self) -> bool:
        """Get parameter info gain setting based on performance mode."""
        return self.performance_mode != "fast"

    def _get_gamma(self) -> float:
        """Get adaptive gamma (policy precision) based on performance mode."""
        if self.performance_mode == "fast":
            return 8.0
        elif self.performance_mode == "balanced":
            return 12.0
        else:  # "accurate"
            return 16.0

    def _get_alpha(self) -> float:
        """Get adaptive alpha (action precision) based on performance mode."""
        if self.performance_mode == "fast":
            return 8.0
        elif self.performance_mode == "balanced":
            return 12.0
        else:  # "accurate"
            return 16.0

    def _should_update_beliefs(self) -> bool:
        """Determine if beliefs should be updated this step."""
        if self.selective_update_interval <= 1:
            return True
        return (self.total_steps % self.selective_update_interval) == 0

    def _get_cached_matrix(
        self, matrix_name: str, matrix_data: Any, normalization_func: Callable[[Any], Any]
    ) -> Any:
        """Get cached normalized matrix or compute and cache it."""
        if not hasattr(self, "matrix_cache"):
            self.matrix_cache: Dict[str, Any] = {}

        cache_key = f"{matrix_name}_{hash(str(matrix_data))}"

        if cache_key not in self.matrix_cache:
            self.matrix_cache[cache_key] = normalization_func(matrix_data)
            logger.debug(f"Cached matrix {matrix_name} for agent {self.agent_id}")

        return self.matrix_cache[cache_key]

    def load_gmn_spec(self, gmn_spec: str) -> None:
        """Load GMN specification for the agent."""
        if not GMN_AVAILABLE:
            logger.error("GMN parser not available")
            return

        try:
            self.gmn_spec = parse_gmn_spec(gmn_spec)
            logger.info(f"Loaded GMN spec for agent {self.agent_id}")

            # Reinitialize PyMDP with new spec
            if PYMDP_AVAILABLE:
                self._initialize_pymdp_from_gmn()
        except Exception as e:
            logger.error(f"Failed to parse GMN spec: {e}")

    def _initialize_pymdp_from_gmn(self) -> None:
        """Initialize PyMDP agent from GMN specification."""
        if not self.gmn_spec:
            return

        try:
            num_states = self.gmn_spec.get("num_states", [4])
            num_obs = self.gmn_spec.get("num_obs", [4])
            num_actions = self.gmn_spec.get("num_actions", [4])

            # Get matrices from GMN spec or use defaults
            A_matrices = self.gmn_spec.get("A", [])
            B_matrices = self.gmn_spec.get("B", [])
            C_vectors = self.gmn_spec.get("C", [])
            D_vectors = self.gmn_spec.get("D", [])

            # Create default matrices if not provided:
            if not A_matrices:
                A = np.eye(num_obs[0], num_states[0])
                A_matrices = [A]

            if not B_matrices:
                B = np.zeros((num_states[0], num_states[0], num_actions[0]))
                for a in range(num_actions[0]):
                    B[:, :, a] = np.eye(num_states[0])
                B_matrices = [B]

            if not C_vectors:
                C = np.zeros(num_obs[0])
                C[0] = 1.0
                C_vectors = [C]

            if not D_vectors:
                D = utils.norm_dist(np.ones(num_states[0]))
                D_vectors = [D]

            # Create PyMDP agent with GMN-specified model
            self.pymdp_agent = PyMDPAgent(
                A=A_matrices,
                B=B_matrices,
                C=C_vectors,
                D=D_vectors,
                use_utility=True,
                use_states_info_gain=True,
                use_param_info_gain=False,
                inference_horizon=self.config.get("planning_horizon", 3),
            )

            logger.info(f"Successfully initialized PyMDP agent from GMN spec for {self.agent_id}")
        except Exception as e:
            logger.error(f"Failed to initialize PyMDP from GMN: {e}")
            self.pymdp_agent = None

    def _process_observation_with_llm(self, observation: Any) -> Any:
        """Process observation with LLM if available."""
        if not self.llm_manager:
            return observation

        try:
            prompt = (
                f"Analyze this observation for an active inference agent: "
                f"{observation}\n"
                "Provide structured analysis for belief updating."
            )
            response = self.llm_manager.generate_response(prompt)
            logger.debug(f"LLM interpretation: {response}")

            # Add LLM interpretation to observation
            if isinstance(observation, dict):
                observation["llm_interpretation"] = response

        except Exception as e:
            logger.error(f"LLM processing failed: {e}")

        return observation

    @abstractmethod
    def perceive(self, observation: Any) -> None:
        """Process incoming observation."""

    @abstractmethod
    def update_beliefs(self) -> None:
        """Update beliefs about hidden states using Active Inference."""

    @abstractmethod
    def select_action(self) -> Any:
        """Select action based on current beliefs and preferences."""

    @with_error_handling("agent_step", fallback_result="stay")
    def step(self, observation: Any) -> Any:
        """Execute one step of the agent's Active Inference loop."""
        if not self.is_active:
            raise RuntimeError(f"Agent {self.agent_id} is not active")

        try:
            # Validate and sanitize observation
            observation = validate_observation(observation)

            # Use LLM to process observation if available:
            if self.llm_manager and self.config.get("use_llm", False):
                observation = self._process_observation_with_llm(observation)

            # Active Inference loop
            self.perceive(observation)

            if self._should_update_beliefs():
                self.update_beliefs()

            action = self.select_action()

            # Update metrics
            self.total_steps += 1
            self.last_action_at = datetime.now()
            self.metrics["total_observations"] += 1
            self.metrics["total_actions"] += 1

            # Compute and store free energy
            if self.pymdp_agent and PYMDP_AVAILABLE:
                fe_components = self.compute_free_energy()
                if fe_components and "total_free_energy" in fe_components:
                    self.metrics.update(fe_components)

            logger.debug(f"Agent {self.agent_id} step {self.total_steps}: {action}")

            return action

        except Exception as e:
            logger.error(f"Agent {self.agent_id} step failed: {e}")
            self.error_handler.handle_error(e, "agent_step")
            return "stay"

    def start(self) -> None:
        """Start the agent."""
        if hasattr(self, "_initialize_pymdp") and self.pymdp_agent is None:
            self._initialize_pymdp()
        self.is_active = True
        logger.info(f"Agent {self.agent_id} started")

    def stop(self) -> None:
        """Stop the agent."""
        self.is_active = False
        logger.info(f"Agent {self.agent_id} stopped")

    def get_status(self) -> Dict[str, Any]:
        """Get current agent status."""
        status = {
            "agent_id": self.agent_id,
            "name": self.name,
            "is_active": self.is_active,
            "created_at": self.created_at.isoformat(),
            "last_action_at": (self.last_action_at.isoformat() if self.last_action_at else None),
            "total_steps": self.total_steps,
            "metrics": self.metrics,
        }

        # Add error handling status
        if hasattr(self, "error_handler"):
            status["error_summary"] = self.error_handler.get_error_summary()

        # Add PyMDP-specific error handling status
        if hasattr(self, "pymdp_error_handler"):
            status["pymdp_error_report"] = self.pymdp_error_handler.get_error_report()

        return status

    def compute_free_energy(self) -> Dict[str, float]:
        """Compute free energy components."""
        if not self.pymdp_agent or not PYMDP_AVAILABLE:
            return {"error": "PyMDP not available"}

        if not hasattr(self.pymdp_agent, "qs") or self.pymdp_agent.qs is None:
            return {"error": "No beliefs available"}

        try:
            # Get current beliefs
            qs = self.pymdp_agent.qs[0]  # Single factor for simplicity

            # Get observation model and current observation
            A = self.pymdp_agent.A[0]
            obs_idx = self.current_observation[0] if self.current_observation else 0

            # Compute accuracy term
            likelihood = A[obs_idx, :]
            epsilon = 1e-10
            accuracy = np.sum(qs * np.log(likelihood + epsilon))

            # Get prior beliefs
            prior = (
                self.pymdp_agent.D[0]
                if hasattr(self.pymdp_agent, "D")
                else np.ones_like(qs) / len(qs)
            )

            # Compute complexity term
            complexity = np.sum(qs * (np.log(qs + epsilon) - np.log(prior + epsilon)))

            # Total variational free energy
            free_energy = complexity - accuracy

            return {
                "total_free_energy": float(free_energy),
                "accuracy": float(accuracy),
                "complexity": float(complexity),
                "surprise": float(-np.log(likelihood[np.argmax(qs)] + epsilon)),
            }

        except Exception as e:
            logger.error(f"Failed to compute free energy: {e}")
            return {"error": str(e)}


class BasicExplorerAgent(ActiveInferenceAgent):
    """Basic explorer agent for grid world environments."""

    def __init__(
        self,
        agent_id: str,
        name: str = "BasicExplorer",
        grid_size: int = 4,
        position: Tuple[int, int] = (0, 0),
    ):
        """Initialize the BasicExplorerAgent."""
        self.grid_size = grid_size
        self.position = position
        self.num_states = grid_size * grid_size
        self.num_obs = 5  # empty, obstacle, goal, agent, out_of_bounds
        self.num_actions = 5  # up, down, left, right, stay

        self.action_map = {0: "up", 1: "down", 2: "left", 3: "right", 4: "stay"}
        self.actions = list(self.action_map.values())

        # Configuration
        config = {
            "use_pymdp": PYMDP_AVAILABLE,
            "use_llm": False,
            "grid_size": grid_size,
            "performance_mode": "fast",
            "selective_update_interval": 2,
        }

        # Initialize parent
        super().__init__(agent_id, name, config)

        # Simplified belief state (fallback)
        self.uncertainty_map = np.ones((grid_size, grid_size))
        self.uncertainty_map[self.position[0], self.position[1]] = 0

        # Preferences
        self.exploration_rate = 0.3

    def _initialize_pymdp(self) -> None:
        """Initialize PyMDP agent for grid world exploration."""
        if not PYMDP_AVAILABLE:
            logger.warning("PyMDP not available, using fallback behavior")
            return

        try:
            # A matrix: P(observation|state) - observation model
            A = np.zeros((self.num_obs, self.num_states))

            # Most states observe "empty"
            for state in range(self.num_states):
                A[0, state] = 0.8  # Empty
                A[1, state] = 0.05  # Obstacle
                A[2, state] = 0.05  # Goal
                A[3, state] = 0.05  # Agent
                A[4, state] = 0.05  # Out of bounds

            # Add noise for robustness
            A = A + 0.01
            A = self._get_cached_matrix("A_observation", A, utils.norm_dist)
            # B matrix: P(next_state|state,action) - transition dynamics
            B = np.zeros((self.num_states, self.num_states, self.num_actions), dtype=np.float32)

            # Action deltas
            action_deltas = {
                0: (-1, 0),  # up
                1: (1, 0),  # down
                2: (0, -1),  # left
                3: (0, 1),  # right
                4: (0, 0),  # stay
            }

            # Build transition matrix
            for state in range(self.num_states):
                x, y = state // self.grid_size, state % self.grid_size

                for action_idx, (dx, dy) in action_deltas.items():
                    next_x, next_y = x + dx, y + dy

                    # Check bounds
                    if 0 <= next_x < self.grid_size and 0 <= next_y < self.grid_size:
                        next_state = next_x * self.grid_size + next_y
                        B[next_state, state, action_idx] = 0.9
                        B[state, state, action_idx] = 0.1
                    else:
                        B[state, state, action_idx] = 1.0

            # Normalize B matrix
            col_sums = np.sum(B, axis=0, keepdims=True)
            col_sums[col_sums == 0] = 1.0
            B = B / col_sums

            # C matrix: Observation preferences
            C = np.zeros(self.num_obs)
            C[0] = 2.0  # Prefer empty spaces
            C[1] = -4.0  # Avoid obstacles
            C[2] = 8.0  # Prefer goals
            C[3] = 0.0  # Neutral about agents
            C[4] = -4.0  # Avoid out-of-bounds

            # D matrix: Initial state distribution
            D = np.ones(self.num_states) / self.num_states
            current_state = self.position[0] * self.grid_size + self.position[1]
            D[current_state] *= 2.0
            D = D / D.sum()

            # Validate matrices
            is_valid, validation_msg = validate_pymdp_matrices(A, B, C, D)
            if not is_valid:
                raise ValueError(f"PyMDP matrix validation failed: {validation_msg}")

            # Create PyMDP agent
            self.pymdp_agent = PyMDPAgent(
                A=A,
                B=B,
                C=C,
                D=D,
                policy_len=self._get_policy_length(),
                inference_algo="VANILLA",
                use_states_info_gain=True,
                use_param_info_gain=self._get_param_info_gain(),
                use_utility=True,
                gamma=self._get_gamma(),
                alpha=self._get_alpha(),
                inference_horizon=1,
                save_belief_hist=False,
            )

            logger.info(f"Initialized PyMDP agent for {self.agent_id}")

        except Exception as e:
            logger.error(f"Failed to initialize PyMDP: {e}")
            self.pymdp_agent = None

    def perceive(self, observation: Dict[str, Any]) -> None:
        """Process incoming observation."""
        if "position" in observation:
            self.position = observation["position"]

        # Convert observation to PyMDP format
        if self.pymdp_agent and PYMDP_AVAILABLE:
            surroundings = np.array(observation.get("surroundings", []))

            if surroundings.size > 0:
                center_val = surroundings[1, 1] if surroundings.shape == (3, 3) else 0

                # Map to observation categories:
                if center_val == -2:  # Out of bounds
                    obs_idx = 4
                elif center_val == -1:  # Obstacle
                    obs_idx = 1
                elif center_val == 1:  # Goal
                    obs_idx = 2
                elif center_val == 2:  # Other agent
                    obs_idx = 3
                else:  # Empty
                    obs_idx = 0

                self.current_observation = [obs_idx]
                self.metrics["last_observation"] = obs_idx
        else:
            # Fallback: update uncertainty map
            x, y = self.position
            self.uncertainty_map[x, y] = 0

            # Reduce uncertainty in neighboring cells
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nx, ny = x + dx, y + dy
                if 0 <= nx < self.grid_size and 0 <= ny < self.grid_size:
                    self.uncertainty_map[nx, ny] *= 0.8

    @performance_monitor("belief_update")
    @safe_pymdp_operation("belief_update", default_value=None)
    def update_beliefs(self) -> None:
        """Update beliefs about hidden states."""
        if self._should_use_pymdp_beliefs():
            self._update_beliefs_with_pymdp()
        else:
            self._fallback_update_beliefs()

    def _should_use_pymdp_beliefs(self) -> bool:
        """Check if PyMDP belief update should be used."""
        return (
            self.pymdp_agent is not None
            and PYMDP_AVAILABLE
            and hasattr(self, "current_observation")
        )

    def _update_beliefs_with_pymdp(self) -> None:
        """Update beliefs using PyMDP inference."""
        try:
            # Perform state inference
            success, _, error = self.pymdp_error_handler.safe_execute(
                "state_inference",
                lambda: self.pymdp_agent.infer_states(self.current_observation),
                lambda: None,
            )

            if not success and error:
                logger.warning(f"State inference failed: {error}")

            # Update metrics based on beliefs
            self._update_belief_metrics()

        except Exception as e:
            logger.warning(f"PyMDP belief update failed: {e}")
            self._fallback_update_beliefs()

    def _update_belief_metrics(self) -> None:
        """Update metrics based on current belief state."""
        if not (hasattr(self.pymdp_agent, "qs") and self.pymdp_agent.qs is not None):
            return

        qs = self.pymdp_agent.qs

        # Calculate and store entropy
        entropy = self._calculate_belief_entropy(qs)
        self.metrics["belief_entropy"] = float(entropy)

        # Store beliefs for debugging if needed
        self._store_debug_beliefs(qs)

        # Update free energy periodically
        self._update_free_energy_metrics()

    def _calculate_belief_entropy(self, qs) -> float:
        """Calculate entropy of belief distribution."""
        epsilon = 1e-10
        if len(qs) == 1:
            factor = qs[0]
            return -np.sum(factor * np.log(factor + epsilon))
        else:
            return sum(-np.sum(factor * np.log(factor + epsilon)) for factor in qs)

    def _store_debug_beliefs(self, qs) -> None:
        """Store beliefs for debugging if debug mode is enabled."""
        if self.config.get("debug_mode", False):
            self.beliefs["state_posterior"] = [q.tolist() for q in qs]

    def _update_free_energy_metrics(self) -> None:
        """Update free energy metrics periodically."""
        if self.total_steps % 10 == 0:
            if hasattr(self.pymdp_agent, "F") and self.pymdp_agent.F is not None:
                self.metrics["avg_free_energy"] = float(np.mean(self.pymdp_agent.F))

    def _fallback_update_beliefs(self) -> None:
        """Fallback belief update when PyMDP fails."""
        self.metrics["belief_entropy"] = np.mean(self.uncertainty_map)

        # Increase uncertainty over time
        self.uncertainty_map *= 1.01
        self.uncertainty_map = np.clip(self.uncertainty_map, 0, 1)

    @performance_monitor("action_selection")
    @safe_pymdp_operation("action_selection", default_value="stay")
    def select_action(self) -> str:
        """Select action based on current beliefs."""
        if self.pymdp_agent and PYMDP_AVAILABLE:
            try:
                # Perform action inference
                success, _, error = self.pymdp_error_handler.safe_execute(
                    "action_inference", lambda: self.pymdp_agent.infer_policies(), lambda: None
                )

                if not success and error:
                    logger.warning(f"Action inference failed: {error}")
                    return self._fallback_select_action()

                # Get action and policy information
                action_idx = self.pymdp_agent.sample_action()

                # Convert to safe integer
                action_idx_converted, error = self.pymdp_error_handler.safe_execute(
                    "action_index_conversion", lambda: safe_array_to_int(action_idx, 4), lambda: 4
                )

                if error:
                    logger.warning(f"Action index conversion failed: {error}")

                # Convert to action string
                selected_action = safe_array_index(self.action_map, action_idx_converted, "stay")

                # Store metrics
                if hasattr(self.pymdp_agent, "G") and self.pymdp_agent.G is not None:
                    self.metrics["expected_free_energy"] = float(np.min(self.pymdp_agent.G))

                return selected_action

            except Exception as e:
                logger.warning(f"PyMDP action selection failed: {e}")
                return self._fallback_select_action()

        return self._fallback_select_action()

    def _fallback_select_action(self) -> str:
        """Fallback action selection when PyMDP fails."""
        # Simple exploration strategy
        x, y = self.position
        action_values = {}

        for action in self.actions:
            new_x, new_y = self._simulate_action(x, y, action)

            if 0 <= new_x < self.grid_size and 0 <= new_y < self.grid_size:
                # Value based on uncertainty reduction
                uncertainty_reduction = self.uncertainty_map[new_x, new_y]
                action_values[action] = uncertainty_reduction
            else:
                action_values[action] = -1  # Invalid action

        # Add exploration noise
        if np.random.random() < self.exploration_rate:
            valid_actions = [a for a, v in action_values.items() if v >= 0]
            return np.random.choice(valid_actions) if valid_actions else "stay"
        else:
            return max(action_values, key=action_values.get)

    def _simulate_action(self, x: int, y: int, action: str) -> Tuple[int, int]:
        """Simulate the result of taking an action."""
        if action == "up":
            return x - 1, y
        elif action == "down":
            return x + 1, y
        elif action == "left":
            return x, y - 1
        elif action == "right":
            return x, y + 1
        else:
            return x, y
