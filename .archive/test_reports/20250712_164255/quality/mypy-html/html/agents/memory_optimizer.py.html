import gc
import logging
import os
import sys
import tracemalloc
import weakref
from contextlib import contextmanager
from functools import lru_cache
from typing import Any, Dict, List, Optional, Set

import numpy as np
import psutil

"""Memory optimization strategies for FreeAgentics agents.

This module implements various memory optimization techniques to reduce
per-agent memory usage from 59.4MB to the target 34.5MB.
"""


logger = logging.getLogger(__name__)


class MemoryPool:
    """Object pool for reusing frequently allocated objects."""

    def __init__(self, factory, max_size=100):
        """Initialize memory pool."""
        self.factory = factory
        self.max_size = max_size
        self._pool: List[Any] = []
        self._in_use: Set[int] = set()

    def acquire(self) -> Any:
        """Acquire an object from the pool."""
        if self._pool:
            obj = self._pool.pop()
            self._in_use.add(id(obj))
            return obj
        else:
            obj = self.factory()
            self._in_use.add(id(obj))
            return obj

    def release(self, obj: Any) -> None:
        """Return an object to the pool."""
        obj_id = id(obj)
        if obj_id in self._in_use:
            self._in_use.remove(obj_id)
            if len(self._pool) < self.max_size:
                # Reset object state if it has a reset method:
                if hasattr(obj, "reset"):
                    obj.reset()
                    self._pool.append(obj)
                else:
                    # Let garbage collector handle it
                    del obj


class MatrixCache:
    """LRU cache for frequently used matrix operations."""

    def __init__(self, max_memory_mb: float = 10.0):
        self.max_memory_bytes = max_memory_mb * 1024 * 1024
        self._cache: Dict[str, np.ndarray] = {}
        self._memory_usage = 0
        self._access_count: Dict[str, int] = {}

    def get(self, key: str) -> Optional[np.ndarray]:
        """Retrieve matrix from cache."""
        if key in self._cache:
            self._access_count[key] += 1
            return self._cache[key].copy()
        return None

    def put(self, key: str, matrix: np.ndarray) -> None:
        """Store matrix in cache with memory management."""
        matrix_size = matrix.nbytes

        # Evict least recently used items if needed
        while self._memory_usage + matrix_size > self.max_memory_bytes and self._cache:
            # Find least accessed item
            lru_key = min(self._access_count.items(), key=lambda x: x[1])[0]
            self._evict(lru_key)

        if self._memory_usage + matrix_size <= self.max_memory_bytes:
            self._cache[key] = matrix.copy()
            self._memory_usage += matrix_size
            self._access_count[key] = 1

    def _evict(self, key: str) -> None:
        """Remove item from cache."""
        if key in self._cache:
            self._memory_usage -= self._cache[key].nbytes
            del self._cache[key]
            del self._access_count[key]

    def clear(self) -> None:
        """Clear entire cache."""
        self._cache.clear()
        self._access_count.clear()
        self._memory_usage = 0


class BeliefStateCompressor:
    """Compress belief states to reduce memory usage."""

    @staticmethod
    def compress_belief(belief: np.ndarray, threshold: float = 1e-6) -> Dict[str, Any]:
        """Compress belief state by storing only significant values."""
        # Find indices of significant values
        significant_indices = np.where(np.abs(belief) > threshold)[0]

        return {
            "shape": belief.shape,
            "indices": significant_indices,
            "values": belief[significant_indices],
            "dtype": str(belief.dtype),
        }

    @staticmethod
    def decompress_belief(compressed: Dict[str, Any]) -> np.ndarray:
        """Reconstruct belief state from compressed format."""
        belief = np.zeros(compressed["shape"], dtype=np.dtype(compressed["dtype"]))
        belief[compressed["indices"]] = compressed["values"]
        return belief

    @staticmethod
    def compress_sparse_matrix(matrix: np.ndarray, threshold: float = 1e-6) -> Dict[str, Any]:
        """Compress sparse matrix using coordinate format."""
        rows, cols = np.where(np.abs(matrix) > threshold)
        values = matrix[rows, cols]

        return {
            "shape": matrix.shape,
            "rows": rows,
            "cols": cols,
            "values": values,
            "dtype": str(matrix.dtype),
        }

    @staticmethod
    def decompress_sparse_matrix(compressed: Dict[str, Any]) -> np.ndarray:
        """Reconstruct sparse matrix from compressed format."""
        matrix = np.zeros(compressed["shape"], dtype=np.dtype(compressed["dtype"]))
        matrix[compressed["rows"], compressed["cols"]] = compressed["values"]
        return matrix


class AgentMemoryOptimizer:
    """Main memory optimizer for agents."""

    def __init__(self):
        self.matrix_cache = MatrixCache(max_memory_mb=5.0)
        self.belief_compressor = BeliefStateCompressor()
        self._weak_refs: Dict[str, weakref.ref] = {}

        # Configure numpy for memory efficiency
        self._configure_numpy()

        # Set up periodic garbage collection
        self._gc_counter = 0
        self._gc_interval = 100

    def _configure_numpy(self):
        """Configure numpy for memory efficiency."""
        # Use float32 instead of float64 where possible
        np.seterr(all="raise")  # Catch numerical issues early

    def optimize_agent_memory(self, agent) -> None:
        """Apply memory optimizations to an agent."""
        # Compress belief states
        self._optimize_belief_states(agent)

        # Optimize matrix storage
        self._optimize_matrix_storage(agent)

        # Clear unnecessary attributes
        self._clear_unnecessary_attributes(agent)

        # Periodic garbage collection
        self._perform_periodic_gc()

    def _optimize_belief_states(self, agent) -> None:
        """Optimize agent belief states for memory efficiency."""
        if hasattr(agent, "beliefs") and isinstance(agent.beliefs, np.ndarray):
            compressed = self.belief_compressor.compress_belief(agent.beliefs)
            agent._compressed_beliefs = compressed
            # Keep weak reference to original for lazy reconstruction
            agent._beliefs_ref = weakref.ref(agent.beliefs)
            del agent.beliefs

    def _optimize_matrix_storage(self, agent) -> None:
        """Optimize matrix storage using caching."""
        if hasattr(agent, "transition_matrix"):
            key = f"transition_{agent.agent_id}"
            self.matrix_cache.put(key, agent.transition_matrix)
            del agent.transition_matrix

    def _perform_periodic_gc(self) -> None:
        """Perform periodic garbage collection to free memory."""
        self._gc_counter += 1
        if self._gc_counter >= self._gc_interval:
            gc.collect(1)  # Collect only young generation
            self._gc_counter = 0

    def _clear_unnecessary_attributes(self, agent) -> None:
        """Remove unnecessary attributes from agent."""
        # List of attributes that can be safely removed or compressed
        removable_attrs = [
            "_debug_info",
            "_temp_cache",
            "_history_full",
            "_raw_observations",
            "_intermediate_results",
        ]

        for attr in removable_attrs:
            if hasattr(agent, attr):
                delattr(agent, attr)

    def get_agent_beliefs(self, agent) -> np.ndarray:
        """Retrieve agent beliefs, decompressing if needed."""
        if hasattr(agent, "beliefs"):
            return agent.beliefs
        elif hasattr(agent, "_compressed_beliefs"):
            # Try weak reference first
            if hasattr(agent, "_beliefs_ref"):
                beliefs = agent._beliefs_ref()
                if beliefs is not None:
                    return beliefs

            # Decompress if weak reference is dead
            beliefs = self.belief_compressor.decompress_belief(agent._compressed_beliefs)
            agent._beliefs_ref = weakref.ref(beliefs)
            return beliefs
        else:
            raise AttributeError(f"Agent {agent.agent_id} has no beliefs")

    def get_transition_matrix(self, agent_id: str) -> Optional[np.ndarray]:
        """Retrieve transition matrix from cache."""
        key = f"transition_{agent_id}"
        return self.matrix_cache.get(key)

    @contextmanager
    def memory_efficient_computation(self):
        """Context manager for memory-efficient computation."""
        # Store current memory usage
        process = psutil.Process(os.getpid())
        initial_memory = process.memory_info().rss / 1024 / 1024  # MB

        # Disable garbage collection during computation
        gc_enabled = gc.isenabled()
        gc.disable()

        try:
            yield
        finally:
            # Re-enable garbage collection
            if gc_enabled:
                gc.enable()

            # Force collection
            gc.collect()

            # Log memory usage
            final_memory = process.memory_info().rss / 1024 / 1024  # MB
            memory_delta = final_memory - initial_memory

            if memory_delta > 10:  # More than 10MB increase
                logger.warning(f"Memory increased by {memory_delta:.1f}MB during computation")

    def profile_agent_memory(self, agent) -> Dict[str, float]:
        """Profile memory usage of an agent."""
        memory_breakdown = {}

        # Get size of major attributes
        for attr_name in dir(agent):
            if not attr_name.startswith("_"):
                try:
                    attr = getattr(agent, attr_name)
                    if isinstance(attr, np.ndarray):
                        memory_breakdown[attr_name] = attr.nbytes / 1024 / 1024  # MB
                    elif hasattr(attr, "__sizeof__"):
                        memory_breakdown[attr_name] = sys.getsizeof(attr) / 1024 / 1024
                except Exception:
                    pass

        return memory_breakdown

    def optimize_matrix_operations(self, matrix: np.ndarray) -> np.ndarray:
        """Optimize matrix for memory efficiency."""
        # Convert to float32 if appropriate
        if matrix.dtype == np.float64:
            # Check if float32 precision is sufficient
            max_val = np.abs(matrix).max()
            if max_val < 1e6:  # Within float32 range
                matrix = matrix.astype(np.float32)

        # Use sparse representation if beneficial
        sparsity = np.count_nonzero(matrix) / matrix.size
        if sparsity < 0.1:  # Less than 10% non-zero
            # Return compressed format indicator
            # (actual sparse matrix library would be used in production)
            logger.info(f"Matrix is {sparsity * 100:.1f}% sparse, consider sparse format")

        return matrix


# Global instance for easy access
memory_optimizer = AgentMemoryOptimizer()


def optimize_agent_pool(agents: List[Any]) -> Dict[str, Any]:
    """Optimize memory for a pool of agents."""
    initial_memory = psutil.Process(os.getpid()).memory_info().rss / 1024 / 1024
    stats = {"agents_optimized": 0, "memory_saved_mb": 0, "errors": []}

    for agent in agents:
        try:
            memory_optimizer.optimize_agent_memory(agent)
            stats["agents_optimized"] += 1
        except Exception as e:
            stats["errors"].append(f"Agent {getattr(agent, 'agent_id', 'unknown')}: {str(e)}")

    # Force garbage collection
    gc.collect()

    final_memory = psutil.Process(os.getpid()).memory_info().rss / 1024 / 1024
    stats["memory_saved_mb"] = initial_memory - final_memory

    return stats

def enable_memory_profiling():
    """Enable detailed memory profiling for debugging."""
    tracemalloc.start()


def print_memory_usage():
    """Print current memory usage statistics."""
    current, peak = tracemalloc.get_traced_memory()
    logger.info(f"Current memory usage: {current / 1024 / 1024:.1f}MB")
    logger.info(f"Peak memory usage: {peak / 1024 / 1024:.1f}MB")

    # Get top memory allocations
    snapshot = tracemalloc.take_snapshot()
    top_stats = snapshot.statistics("lineno")

    logger.info("Top 10 memory allocations:")
    for stat in top_stats[:10]:
        logger.info(f"{stat}")
