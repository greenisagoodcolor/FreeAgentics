"""Pattern predictor agent using Active Inference for pattern recognition and prediction."""

import logging
from collections import deque
from datetime import datetime
from typing import Any, Dict, List

from agents.base_agent import PYMDP_AVAILABLE, ActiveInferenceAgent
from agents.error_handling import safe_pymdp_operation
from agents.pymdp_error_handling import PyMDPErrorHandler

logger = logging.getLogger(__name__)


class PatternPredictorAgent(ActiveInferenceAgent):
    """Agent that predicts patterns using Active Inference.

        This agent:
    - Learns patterns from environmental observations
    - Makes predictions about future states
    - Uses Active Inference for pattern recognition
    - Adapts predictions based on prediction errors
    """

    def __init__(
        self,
        agent_id,
        name="PatternPredictor",
        grid_size=10,
        prediction_window=10,
        model_complexity="high",
    ):
        """Initialize pattern predictor agent.

            Args:
        agent_id: Unique agent identifier
        name: Agent name
        grid_size: Size of the grid world
        prediction_window: Number of steps to predict ahead
        model_complexity: Complexity level of prediction model
        """
        # Set attributes needed for PyMDP initialization
        self.grid_size = grid_size
        self.position = [grid_size // 2, grid_size // 2]
        self.prediction_window = prediction_window
        self.model_complexity = model_complexity

        # PyMDP state space
        self.num_states = grid_size * grid_size
        self.num_obs = 6  # Different observation types
        self.num_actions = 5  # Movement actions + predict action

        config = {
            "use_pymdp": PYMDP_AVAILABLE,
            "use_llm": True,
            "grid_size": grid_size,
            "agent_type": "pattern_predictor",
            "prediction_window": prediction_window,
            "model_complexity": model_complexity,
        }

        super().__init__(agent_id, name, config)

        # Pattern-specific state
        self.observation_history = deque(maxlen=100)
        self.pattern_library: Dict[str, Dict[str, Any]] = {}
        self.predictions: List[Dict[str, Any]] = []
        self.prediction_accuracy = 0.0

        # Pattern detection parameters
        self.min_pattern_length = 3
        self.max_pattern_length = 20
        self.pattern_confidence_threshold = 0.7

        # Action mapping
        self.action_map = {
            0: "up",
            1: "down",
            2: "left",
            3: "right",
            4: "predict",
        }

        # Initialize attributes from base class
        self.total_observations = 0
        self.total_actions = 0
        self.use_llm = config.get("use_llm", True)
        self.llm_manager = None

        # Initialize error handling
        self.pymdp_error_handler = PyMDPErrorHandler(self.agent_id)

        # Prediction tracking
        self.correct_predictions = 0
        self.total_predictions = 0

    def perceive(self, observation: Any) -> None:
        """Process an observation from the environment.

        Args:
            observation: Observation from the environment
        """
        self.total_observations += 1

        # Store observation in history
        self.observation_history.append(
            {
                "observation": observation,
                "timestamp": datetime.now(),
                "step": self.total_observations,
            }
        )

        # Process pattern-specific observations
        if isinstance(observation, dict):
            # Extract features for pattern detection
            _ = self._extract_features(observation)

            # Detect patterns in recent observations
            if len(self.observation_history) >= self.min_pattern_length:
                self._detect_patterns()

            # Verify previous predictions
            self._verify_predictions(observation)

            # Update position if available
            if "position" in observation:
                self.position = [observation["position"]["x"], observation["position"]["y"]]

        # Call parent class method if available
        if hasattr(super(), "perceive"):
            super().perceive(observation)

    def _extract_features(self, observation: Dict[str, Any]) -> Dict[str, Any]:
        """Extract relevant features from observation for pattern detection.

        Args:
            observation: Raw observation

        Returns:
            Dictionary of extracted features
        """
        features = {}

        # Extract positional features
        if "position" in observation:
            features["position"] = (observation["position"]["x"], observation["position"]["y"])

        # Extract environmental features
        if "nearby_cells" in observation:
            features["environment"] = observation["nearby_cells"]

        # Extract agent-related features
        if "agents" in observation:
            features["agent_count"] = len(observation["agents"])
            features["agent_positions"] = [(a["x"], a["y"]) for a in observation["agents"]]

        # Extract resource-related features
        if "resources" in observation:
            features["resource_count"] = len(observation["resources"])
            features["resource_types"] = [r["type"] for r in observation["resources"]]

        return features

    def _detect_patterns(self) -> None:
        """Detect patterns in observation history."""
        # Simple pattern detection: look for repeating sequences
        history_features = [
            self._extract_features(h["observation"])
            for h in self.observation_history
            if isinstance(h["observation"], dict)
        ]

        if len(history_features) < self.min_pattern_length:
            return

        # Look for repeating sequences
        for length in range(
            self.min_pattern_length, min(self.max_pattern_length, len(history_features) // 2)
        ):
            for start in range(len(history_features) - length * 2):
                pattern = history_features[start : start + length]
                next_seq = history_features[start + length : start + length * 2]

                if self._sequences_match(pattern, next_seq):
                    pattern_id = f"pattern_{len(self.pattern_library)}"
                    self.pattern_library[pattern_id] = {
                        "pattern": pattern,
                        "length": length,
                        "confidence": self._calculate_confidence(pattern, history_features),
                        "discovered_at": datetime.now(),
                        "occurrences": 1,
                    }
                    logger.info(f"Discovered new pattern: {pattern_id} (length={length})")

    def _sequences_match(self, seq1: List[Dict], seq2: List[Dict]) -> bool:
        """Check if two sequences match (with some tolerance)."""
        if len(seq1) != len(seq2):
            return False

        matches = 0
        for f1, f2 in zip(seq1, seq2):
            if self._features_similar(f1, f2):
                matches += 1

        # Allow some variation
        return matches >= len(seq1) * 0.8

    def _features_similar(self, f1: Dict, f2: Dict) -> bool:
        """Check if two feature sets are similar."""
        # Simple similarity check - can be made more sophisticated
        if "position" in f1 and "position" in f2:
            # Check if positions are close
            dist = abs(f1["position"][0] - f2["position"][0]) + abs(
                f1["position"][1] - f2["position"][1]
            )
            if dist > 2:
                return False

        # Check other features
        for key in ["agent_count", "resource_count"]:
            if key in f1 and key in f2:
                if abs(f1.get(key, 0) - f2.get(key, 0)) > 1:
                    return False

        return True

    def _calculate_confidence(self, pattern: List[Dict], history: List[Dict]) -> float:
        """Calculate confidence score for a pattern."""
        # Count occurrences of pattern in history
        occurrences = 0
        for i in range(len(history) - len(pattern)):
            if self._sequences_match(pattern, history[i : i + len(pattern)]):
                occurrences += 1

        # Confidence based on frequency
        return min(occurrences / (len(history) / len(pattern)), 1.0)

    def make_prediction(self, steps_ahead: int = 1) -> Dict[str, Any]:
        """Make a prediction about future states.

        Args:
            steps_ahead: Number of steps to predict ahead

        Returns:
            Prediction dictionary
        """
        prediction = {
            "steps_ahead": steps_ahead,
            "predicted_at": datetime.now(),
            "predictions": [],
            "confidence": 0.0,
            "metadata": {},
        }

        # Use detected patterns to make predictions
        if self.pattern_library:
            # Find the most confident matching pattern
            best_pattern = None
            best_confidence = 0.0

            for pattern_id, pattern_info in self.pattern_library.items():
                if pattern_info["confidence"] > best_confidence:
                    # Check if recent history matches this pattern
                    recent = [
                        self._extract_features(h["observation"])
                        for h in list(self.observation_history)[-pattern_info["length"] :]
                    ]

                    if self._sequences_match(pattern_info["pattern"], recent):
                        best_pattern = pattern_info
                        best_confidence = pattern_info["confidence"]

            if best_pattern and best_confidence > self.pattern_confidence_threshold:
                # Make prediction based on pattern continuation
                prediction["predictions"] = best_pattern["pattern"][:steps_ahead]
                prediction["confidence"] = best_confidence
                prediction["pattern_id"] = best_pattern.get("id", "unknown")

        # Store prediction for later verification
        self.predictions.append(prediction)
        self.total_predictions += 1

        return prediction

    def _verify_predictions(self, observation: Dict[str, Any]) -> None:
        """Verify previous predictions against actual observations."""
        current_features = self._extract_features(observation)

        # Check recent predictions
        for pred in self.predictions[-10:]:  # Check last 10 predictions
            if pred["predicted_at"] and len(pred["predictions"]) > 0:
                # Check if this observation matches the prediction
                if self._features_similar(pred["predictions"][0], current_features):
                    self.correct_predictions += 1
                    pred["verified"] = True
                    pred["accurate"] = True

        # Update accuracy
        if self.total_predictions > 0:
            self.prediction_accuracy = self.correct_predictions / self.total_predictions

    def get_pattern_summary(self) -> Dict[str, Any]:
        """Get summary of detected patterns and prediction performance.

        Returns:
            Summary dictionary
        """
        return {
            "patterns_detected": len(self.pattern_library),
            "prediction_accuracy": self.prediction_accuracy,
            "total_predictions": self.total_predictions,
            "correct_predictions": self.correct_predictions,
            "high_confidence_patterns": sum(
                1
                for p in self.pattern_library.values()
                if p["confidence"] > self.pattern_confidence_threshold
            ),
        }

    @safe_pymdp_operation("pymdp_init", default_value=None)
    def _initialize_pymdp(self) -> None:
        """Initialize PyMDP agent for pattern prediction."""
        if not PYMDP_AVAILABLE:
            logger.warning("PyMDP not available for pattern predictor")
            return None
