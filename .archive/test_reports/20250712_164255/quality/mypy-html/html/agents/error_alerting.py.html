"""Alert and notification system for agent error management.

This module provides comprehensive alerting capabilities for monitoring
agent health, detecting error patterns, and dispatching notifications
through various channels.
"""

import logging
from datetime import datetime, timedelta
from enum import Enum
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger(__name__)


class AlertSeverity(Enum):
    """Alert severity levels."""

    INFO = "info"
    WARNING = "warning"
    CRITICAL = "critical"
    EMERGENCY = "emergency"


class AlertChannel(Enum):
    """Alert delivery channels."""

    LOG = "log"
    WEBHOOK = "webhook"
    EMAIL = "email"
    METRICS = "metrics"


class Alert:
    """Represents an alert event."""

    def __init__(
        self,
        alert_id: str,
        title: str,
        description: str,
        severity: AlertSeverity,
        agent_id: str,
        error_type: str,
        metadata: Optional[Dict[str, Any]] = None,
        channels: Optional[List[AlertChannel]] = None,
    ):
        self.alert_id = alert_id
        self.title = title
        self.description = description
        self.severity = severity
        self.agent_id = agent_id
        self.error_type = error_type
        self.metadata = metadata or {}
        self.channels = channels or [AlertChannel.LOG]
        self.timestamp = datetime.utcnow()
        self.resolved = False
        self.resolution_time: Optional[datetime] = None

    def resolve(self, resolution_notes: str = "") -> None:
        """Mark alert as resolved."""
        self.resolved = True
        self.resolution_time = datetime.utcnow()
        self.metadata["resolution_notes"] = resolution_notes

    def to_dict(self) -> Dict[str, Any]:
        """Convert alert to dictionary."""
        return {
            "alert_id": self.alert_id,
            "title": self.title,
            "description": self.description,
            "severity": self.severity.value,
            "agent_id": self.agent_id,
            "error_type": self.error_type,
            "metadata": self.metadata,
            "channels": [ch.value for ch in self.channels],
            "timestamp": self.timestamp.isoformat(),
            "resolved": self.resolved,
            "resolution_time": self.resolution_time.isoformat() if self.resolution_time else None,
        }


class AlertRule:
    """Defines conditions for triggering alerts."""

    def __init__(
        self,
        rule_id: str,
        name: str,
        condition: Callable[[Dict[str, Any]], bool],
        severity: AlertSeverity,
        channels: List[AlertChannel],
        cooldown_minutes: int = 5,
        max_alerts_per_hour: int = 10,
    ):
        self.rule_id = rule_id
        self.name = name
        self.condition = condition
        self.severity = severity
        self.channels = channels
        self.cooldown_minutes = cooldown_minutes
        self.max_alerts_per_hour = max_alerts_per_hour
        self.last_triggered: Optional[datetime] = None
        self.alert_count_this_hour = 0
        self.hour_start = datetime.utcnow()

    def should_trigger(self, context: Dict[str, Any]) -> bool:
        """Check if rule should trigger based on condition and rate limits."""
        # Check condition
        if not self.condition(context):
            return False

        # Check cooldown
        if self.last_triggered:
            cooldown_elapsed = datetime.utcnow() - self.last_triggered
            if cooldown_elapsed < timedelta(minutes=self.cooldown_minutes):
                return False

        # Check hourly rate limit
        current_hour = datetime.utcnow()
        if (current_hour - self.hour_start) > timedelta(hours=1):
            self.alert_count_this_hour = 0
            self.hour_start = current_hour

        if self.alert_count_this_hour >= self.max_alerts_per_hour:
            return False

        return True

    def mark_triggered(self) -> None:
        """Mark rule as triggered."""
        self.last_triggered = datetime.utcnow()
        self.alert_count_this_hour += 1


class AlertDispatcher:
    """Dispatches alerts through configured channels."""

    def __init__(self):
        self.channel_handlers: Dict[AlertChannel, Callable[[Alert], None]] = {
            AlertChannel.LOG: self._handle_log_alert,
            AlertChannel.WEBHOOK: self._handle_webhook_alert,
            AlertChannel.EMAIL: self._handle_email_alert,
            AlertChannel.METRICS: self._handle_metrics_alert,
        }
        self.webhook_urls: List[str] = []
        self.email_recipients: List[str] = []

    def dispatch(self, alert: Alert) -> None:
        """Dispatch alert through configured channels."""
        for channel in alert.channels:
            handler = self.channel_handlers.get(channel)
            if handler:
                try:
                    handler(alert)
                except Exception as e:
                    logger.error(f"Failed to dispatch alert through {channel}: {e}")

    def _handle_log_alert(self, alert: Alert) -> None:
        """Handle alert through logging."""
        log_level = {
            AlertSeverity.INFO: logging.INFO,
            AlertSeverity.WARNING: logging.WARNING,
            AlertSeverity.CRITICAL: logging.ERROR,
            AlertSeverity.EMERGENCY: logging.CRITICAL,
        }.get(alert.severity, logging.ERROR)

        logger.log(
            log_level,
            f"[ALERT] {alert.title} - {alert.description}",
            extra={
                "alert_id": alert.alert_id,
                "agent_id": alert.agent_id,
                "error_type": alert.error_type,
                "metadata": alert.metadata,
            },
        )

    def _handle_webhook_alert(self, alert: Alert) -> None:
        """Handle alert through webhook."""
        # Webhook implementation would go here
        logger.info(f"Webhook alert dispatched: {alert.alert_id}")

    def _handle_email_alert(self, alert: Alert) -> None:
        """Handle alert through email."""
        # Email implementation would go here
        logger.info(f"Email alert dispatched: {alert.alert_id}")

    def _handle_metrics_alert(self, alert: Alert) -> None:
        """Handle alert through metrics system."""
        # Metrics implementation would go here
        logger.info(f"Metrics alert dispatched: {alert.alert_id}")


class AlertManager:
    """Manages alert rules and dispatching."""

    def __init__(self):
        self.dispatcher = AlertDispatcher()
        self.rules: Dict[str, AlertRule] = {}
        self.alerts: Dict[str, Alert] = {}
        self.alert_history: List[Alert] = []

        # Initialize default rules
        self._setup_default_rules()

    def _setup_default_rules(self) -> None:
        """Set up default alert rules."""
        # High error rate rule
        self.add_rule(
            AlertRule(
                rule_id="high_error_rate",
                name="High Error Rate",
                condition=lambda ctx: ctx.get("error_rate", 0) > 0.1,
                severity=AlertSeverity.WARNING,
                channels=[AlertChannel.LOG, AlertChannel.METRICS],
                cooldown_minutes=10,
            )
        )

        # Agent failure rule
        self.add_rule(
            AlertRule(
                rule_id="agent_failure",
                name="Agent Failure",
                condition=lambda ctx: ctx.get("agent_status") == "failed",
                severity=AlertSeverity.CRITICAL,
                channels=[AlertChannel.LOG, AlertChannel.WEBHOOK],
                cooldown_minutes=5,
            )
        )

        # Resource exhaustion rule
        self.add_rule(
            AlertRule(
                rule_id="resource_exhaustion",
                name="Resource Exhaustion",
                condition=lambda ctx: ctx.get("memory_usage", 0) > 0.9,
                severity=AlertSeverity.CRITICAL,
                channels=[AlertChannel.LOG, AlertChannel.EMAIL],
                cooldown_minutes=15,
            )
        )

    def add_rule(self, rule: AlertRule) -> None:
        """Add an alert rule."""
        self.rules[rule.rule_id] = rule

    def remove_rule(self, rule_id: str) -> None:
        """Remove an alert rule."""
        if rule_id in self.rules:
            del self.rules[rule_id]

    def check_rules(self, context: Dict[str, Any]) -> List[Alert]:
        """Check all rules against context and generate alerts."""
        alerts = []

        for rule in self.rules.values():
            if rule.should_trigger(context):
                alert = self._create_alert_from_rule(rule, context)
                alerts.append(alert)
                rule.mark_triggered()

                # Store and dispatch alert
                self.alerts[alert.alert_id] = alert
                self.alert_history.append(alert)
                self.dispatcher.dispatch(alert)

        return alerts

    def _create_alert_from_rule(self, rule: AlertRule, context: Dict[str, Any]) -> Alert:
        """Create alert from rule and context."""
        import uuid

        return Alert(
            alert_id=str(uuid.uuid4()),
            title=rule.name,
            description=f"Alert triggered by rule {rule.rule_id}",
            severity=rule.severity,
            agent_id=context.get("agent_id", "unknown"),
            error_type=rule.rule_id,
            metadata=context,
            channels=rule.channels,
        )

    def resolve_alert(self, alert_id: str, resolution_notes: str = "") -> bool:
        """Resolve an alert."""
        if alert_id in self.alerts:
            self.alerts[alert_id].resolve(resolution_notes)
            return True
        return False

    def get_active_alerts(self, agent_id: Optional[str] = None) -> List[Alert]:
        """Get active (unresolved) alerts."""
        alerts = [a for a in self.alerts.values() if not a.resolved]

        if agent_id:
            alerts = [a for a in alerts if a.agent_id == agent_id]

        return alerts

    def get_alert_statistics(self) -> Dict[str, Any]:
        """Get alert statistics."""
        total_alerts = len(self.alert_history)
        active_alerts = len([a for a in self.alerts.values() if not a.resolved])

        alerts_by_severity = {}
        for severity in AlertSeverity:
            count = len([a for a in self.alert_history if a.severity == severity])
            alerts_by_severity[severity.value] = count

        return {
            "total_alerts": total_alerts,
            "active_alerts": active_alerts,
            "alerts_by_severity": alerts_by_severity,
            "rules_count": len(self.rules),
        }


# Global alert manager instance
_alert_manager: Optional[AlertManager] = None


def get_alert_manager() -> AlertManager:
    """Get the global alert manager instance."""
    global _alert_manager
    if _alert_manager is None:
        _alert_manager = AlertManager()
    return _alert_manager
