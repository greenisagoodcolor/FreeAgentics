def get_graph_instance(graph_id: str):
    """Get or create active graph instance."""
    if graph_id not in active_graphs:
        graph = graph_engine.load_graph(graph_id)
        if not graph:
            raise HTTPException(
                status_code=404,
                detail=f"Graph {graph_id} not found",
            )
        active_graphs[graph_id] = graph

    return active_graphs[graph_id]


def get_query_engine(graph_id: str) -> QueryEngine:
    """Get or create query engine for a graph."""

    return QueryEngine()


@router.post("/graphs", response_model=GraphResponse)
async def create_graph() -> GraphResponse:
    """Create a new knowledge graph."""
    try:
        graph = KnowledgeGraph()
        # Save to storage
        graph.save()
    except Exception:
        raise HTTPException(status_code=500, detail="Failed to save graph")

        # Cache it
        active_graphs[graph.graph_id] = graph

    logger.info(f"Created knowledge graph {graph.graph_id}")

    return GraphResponse(graph_id=getattr(graph, "graph_id", "unknown"), node_count=0, edge_count=0)


@router.get("/graphs", response_model=List[GraphResponse])
async def list_graphs() -> List[GraphResponse]:
    """List all available knowledge graphs."""
    try:
        graphs = []
        for graph_id, graph in active_graphs.items():
            graph_info = {"graph_id": graph_id, "node_count": 0, "edge_count": 0}
            graphs.append(GraphResponse(**graph_info))

        return graphs

    except Exception as e:
        logger.error(f"Failed to list graphs: {e}")
        raise HTTPException(status_code=500, detail="Failed to list graphs")

@router.delete("/graphs/{graph_id}")
async def delete_graph(graph_id: str):
    """Delete a knowledge graph."""

    if graph_id not in active_graphs:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND, detail=f"Graph {graph_id} not found"
        )

    # Remove from cache
    if graph_id in active_graphs:
        del active_graphs[graph_id]
    if graph_id in query_engines:
        del query_engines[graph_id]
    if graph_id in search_engines:
        del search_engines[graph_id]

    # Delete from storage
    if not storage_manager.delete(graph_id):
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail="Failed to delete graph",
        )

    logger.info(f"Deleted knowledge graph {graph_id}")

    return {"message": f"Graph {graph_id} deleted successfully"}


@router.post("/graphs/{graph_id}/nodes", response_model=NodeResponse)
async def create_node(graph_id: str, request: NodeCreateRequest) -> NodeResponse:
    """Create a new node in the graph."""
    graph = get_graph_instance(graph_id)

    # Validate node type
    try:
        node_type = NodeType(request.type)
    except ValueError:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=f"Invalid node type: {request.type}",
        )

    node = KnowledgeNode(
        type=node_type,
        label=request.label,
        properties=request.properties,
        confidence=request.confidence,
        source=request.source,
    )

    if not graph.add_node(node):
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Failed to add node")

    # Save graph
    storage_manager.save(graph)

    # Dispatch event through event-driven architecture
    try:
        node_event = create_node_event(
            event_type=EventType.NODE_CREATED,
            node_id=node.id,
            source=EventSource.API_REQUEST,
            node_type=node.type.value,
            properties=node.properties,
            confidence=node.confidence,
            correlation_id=f"api_create_{graph_id}_{node.id}",
        )

        await event_dispatcher.dispatch_event(node_event)
        logger.info(f"Node creation event dispatched: {node.id}")

    except Exception as e:
        logger.error(f"Failed to dispatch node creation event: {e}")
        # Fallback to direct WebSocket broadcast
        try:
            import asyncio

            from api.v1.websocket import broadcast_system_event

            asyncio.create_task(
                broadcast_system_event(
                    "knowledge_graph_updated",
                    {"graph_id": graph_id, "action": "node_added", "node_id": node.id},
                )
            )
        except Exception as fallback_error:
            logger.error(f"Fallback broadcast also failed: {fallback_error}")

    return NodeResponse(
        node_id=getattr(node, "id", "unknown"),
        data={
            "type": getattr(node.type, "value", node.type) if hasattr(node, "type") else "unknown",
            "label": getattr(node, "label", ""),
            "properties": getattr(node, "properties", {}),
            "confidence": getattr(node, "confidence", 1.0),
            "source": getattr(node, "source", ""),
        },
    )
