"""
WebSocket authentication endpoints for token and permissions management.
"""

from typing import Any, Dict

from fastapi import APIRouter, Depends

from api.v1.auth import TokenData, get_current_user
from api.v1.security import get_user_permissions
from auth.security_implementation import rate_limit

router = APIRouter()


@router.get("/ws/auth/token")
@rate_limit(endpoint_type="auth")
async def get_websocket_token(
    current_user: TokenData = Depends(get_current_user),
) -> Dict[str, Any]:
    """Get WebSocket connection information for authenticated user."""
    return {
        "websocket_url": "/api/v1/ws/{client_id}",
        "connection_params": {
            "token": "YOUR_JWT_TOKEN",  # Client should use their actual token
            "client_id": "unique_client_identifier",
        },
        "user": {
            "user_id": current_user.user_id,
            "username": current_user.username,
            "role": current_user.role.value,
            "permissions": [p.value for p in current_user.permissions],
        },
        "instructions": [
            "1. Generate a unique client_id for your connection",
            "2. Connect to WebSocket with: ws://host/api/v1/ws/{client_id}?token={jwt_token}",
            "3. The token parameter must be included as a query parameter",
            "4. Connection will be closed with code 4001 if authentication fails",
            "5. Token expiration will close connection with code 4002",
        ],
        "close_codes": {
            "4001": "Authentication failed",
            "4002": "Token expired",
            "4003": "Authorization failed",
            "4500": "Internal error",
        },
    }


@router.get("/ws/auth/permissions")
@rate_limit(endpoint_type="auth")
async def get_websocket_permissions(
    current_user: TokenData = Depends(get_current_user),
) -> Dict[str, Any]:
    """Get WebSocket permissions for the current user."""
    permissions = get_user_permissions(current_user.role)
    return {
        "user": current_user.username,
        "role": current_user.role.value,
        "permissions": permissions,
        "websocket_capabilities": {
            "can_view_agents": "view_agents" in permissions,
            "can_control_agents": "modify_agent" in permissions,
            "can_create_agents": "create_agent" in permissions,
            "can_delete_agents": "delete_agent" in permissions,
            "can_view_metrics": "view_metrics" in permissions,
            "can_receive_system_events": "admin_system" in permissions,
        },
        "available_subscriptions": {
            "agent:created": "create_agent" in permissions,
            "agent:updated": "view_agents" in permissions,
            "agent:deleted": "delete_agent" in permissions,
            "agent:status": "view_agents" in permissions,
            "world:updated": "view_agents" in permissions,
            "metrics:updated": "view_metrics" in permissions,
            "system:alerts": "admin_system" in permissions,
        },
    }
