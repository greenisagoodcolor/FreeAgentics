"""Provides endpoints to validate and monitor security header configuration."""

import hashlib
import logging
from datetime import datetime
from typing import Any, Dict, List

from fastapi import APIRouter, Depends, Request, Response
from pydantic import BaseModel, Field

# Safe imports with fallbacks
try:
    from api.middleware.rate_limiting import rate_limit
except ImportError:

    def rate_limit(**kwargs):
        def decorator(func):
            return func

        return decorator


try:
    from auth.rbac import Permission, require_permission
    from auth.security_implementation import get_current_user
except ImportError:

    def get_current_user():
        return {"user_id": "test-user", "username": "test"}

    class Permission:
        ADMIN_SYSTEM = "admin_system"

    def require_permission(permission):
        def decorator(func):
            return func

        return decorator


logger = logging.getLogger(__name__)
router = APIRouter()


class SecurityHeadersResponse(BaseModel):
    """Response model for security headers validation."""

    csp_present: bool = Field(False, description="Content Security Policy present")
    hsts_present: bool = Field(False, description="HTTP Strict Transport Security present")
    x_content_type_present: bool = Field(False, description="X-Content-Type-Options present")
    x_frame_present: bool = Field(False, description="X-Frame-Options present")
    x_xss_present: bool = Field(False, description="X-XSS-Protection present")
    referrer_policy_present: bool = Field(False, description="Referrer-Policy present")
    permissions_policy_present: bool = Field(False, description="Permissions-Policy present")
    cors_headers_secure: bool = Field(False, description="CORS headers secure")
    cache_control_secure: bool = Field(False, description="Cache-Control secure")
    expect_ct_present: bool = Field(False, description="Expect-CT present")
    security_score: int = Field(0, description="Overall security score")
    recommendations: List[str] = Field(default_factory=list, description="Security recommendations")


class TLSConfigResponse(BaseModel):
    """Response model for TLS configuration."""

    tls_version_min: str = Field("1.2", description="Minimum TLS version")
    weak_ciphers_disabled: bool = Field(True, description="Weak ciphers disabled")
    forward_secrecy: bool = Field(True, description="Perfect forward secrecy enabled")
    hsts_enabled: bool = Field(True, description="HSTS enabled")
    certificate_transparency: bool = Field(True, description="Certificate transparency enabled")
    validation_passed: bool = Field(True, description="TLS validation passed")


@router.get("/headers/validate", response_model=SecurityHeadersResponse)
@rate_limit(endpoint_type="public", max_requests=60, window_minutes=1)
async def validate_security_headers(
    request: Request,
    response: Response,
) -> SecurityHeadersResponse:
    """Returns security headers with validation status and recommendations."""
    headers = {k.lower(): v for k, v in request.headers.items()}

    # Check security headers
    security_checks = {
        "csp_present": "content-security-policy" in headers,
        "hsts_present": "strict-transport-security" in headers,
        "x_content_type_present": "x-content-type-options" in headers
        and headers.get("x-content-type-options") == "nosniff",
        "x_frame_present": "x-frame-options" in headers
        and headers.get("x-frame-options") == "DENY",
        "x_xss_present": "x-xss-protection" in headers,
        "referrer_policy_present": "referrer-policy" in headers,
        "permissions_policy_present": "permissions-policy" in headers,
        "cors_headers_secure": all(
            h in headers for h in ["cross-origin-embedder-policy", "cross-origin-opener-policy"]
        ),
        "cache_control_secure": "cache-control" in headers
        and "no-store" in headers.get("cache-control", ""),
        "expect_ct_present": "expect-ct" in headers,
    }

    # Calculate security score
    security_score = sum(1 for check in security_checks.values() if check) * 10

    # Generate recommendations
    recommendations = []
    if not security_checks["csp_present"]:
        recommendations.append("Add Content-Security-Policy header")
    if not security_checks["hsts_present"]:
        recommendations.append("Add Strict-Transport-Security header")
    if not security_checks["x_content_type_present"]:
        recommendations.append("Add X-Content-Type-Options: nosniff header")

    return SecurityHeadersResponse(
        **security_checks, security_score=security_score, recommendations=recommendations
    )


@router.get("/headers/test-csp-nonce")
@rate_limit(endpoint_type="public", max_requests=60, window_minutes=1)
async def test_csp_nonce(
    request: Request,
    response: Response,
) -> Dict[str, Any]:
    """Demonstrates how to use CSP nonces for inline scripts."""
    import secrets

    nonce = secrets.token_urlsafe(16)

    # Set CSP header with nonce
    csp_header = f"default-src 'self'; script-src 'self' 'nonce-{nonce}'"
    response.headers["Content-Security-Policy"] = csp_header

    return {
        "nonce": nonce,
        "example_usage": f'<script nonce="{nonce}">console.log("Safe inline script");</script>',
        "csp_header": csp_header,
        "note": "Use this nonce value for any inline scripts in the response",
    }


@router.get("/tls/validate", response_model=TLSConfigResponse)
@rate_limit(endpoint_type="admin")
@require_permission(Permission.ADMIN_SYSTEM)
async def validate_tls_configuration(
    current_user: dict = Depends(get_current_user),
) -> TLSConfigResponse:
    """Validate TLS configuration. Requires ADMIN_SYSTEM permission."""
    # Mock TLS configuration for now
    tls_config = {
        "tls_version_min": "1.2",
        "weak_ciphers_disabled": True,
        "forward_secrecy": True,
        "hsts_enabled": True,
        "certificate_transparency": True,
    }

    validation_passed = all(
        [
            tls_config["tls_version_min"] >= "1.2",
            tls_config["weak_ciphers_disabled"],
            tls_config["forward_secrecy"],
        ]
    )

    return TLSConfigResponse(**tls_config, validation_passed=validation_passed)


@router.post("/headers/calculate-hash")
@rate_limit(endpoint_type="admin")
async def calculate_csp_hash(
    content: str,
    algorithm: str = "sha256",
) -> Dict[str, str]:
    """Calculate CSP hash for inline scripts."""
    if algorithm not in ["sha256", "sha384", "sha512"]:
        return {
            "error": "Invalid algorithm",
            "supported": ["sha256", "sha384", "sha512"],
        }

    try:
        # Calculate hash
        hash_func = getattr(hashlib, algorithm)
        hash_value = hash_func(content.encode("utf-8")).hexdigest()

        return {
            "algorithm": algorithm,
            "hash": hash_value,
            "csp_value": f"'{algorithm}-{hash_value}'",
            "usage": f"Add to script-src or style-src: '{algorithm}-{hash_value}'",
        }
    except Exception as e:
        return {"error": str(e)}


@router.get("/headers/report")
@rate_limit(endpoint_type="admin")
@require_permission(Permission.ADMIN_SYSTEM)
async def get_security_headers_report(
    request: Request,
    current_user: dict = Depends(get_current_user),
) -> Dict[str, Any]:
    """Get comprehensive security headers report. Requires ADMIN_SYSTEM permission."""
    headers = {k.lower(): v for k, v in request.headers.items()}

    csp = headers.get("content-security-policy", "")
    csp_analysis = {
        "uses_nonces": "'nonce-" in csp,
        "uses_hashes": "'sha" in csp,
        "allows_unsafe_inline": "'unsafe-inline'" in csp,
        "allows_unsafe_eval": "'unsafe-eval'" in csp,
        "upgrade_insecure_requests": "upgrade-insecure-requests" in csp,
        "default_src_self_only": "default-src 'self'" in csp,
    }

    # Calculate security score
    header_score = (
        sum(
            [
                bool(headers.get("content-security-policy")),
                bool(headers.get("strict-transport-security")),
                bool(headers.get("x-content-type-options")),
                bool(headers.get("x-frame-options")),
                bool(headers.get("referrer-policy")),
                bool(headers.get("permissions-policy")),
                not csp_analysis["allows_unsafe_inline"],
                not csp_analysis["allows_unsafe_eval"],
            ]
        )
        * 10
    )

    return {
        "timestamp": datetime.now().isoformat(),
        "security_score": header_score,
        "headers_configured": len(headers),
        "csp_analysis": csp_analysis,
        "recommendations": {
            "critical": [
                "Enable HSTS in production",
                "Remove unsafe-inline from CSP",
                "Enable Certificate Transparency",
            ],
            "important": [
                "Consider HSTS preloading",
                "Implement CSP reporting",
                "Use CSP nonces or hashes instead of unsafe-inline",
            ],
            "good_practices": [
                "Regularly review and update CSP policies",
                "Monitor CSP violations",
                "Test security headers with online tools",
            ],
        },
    }
