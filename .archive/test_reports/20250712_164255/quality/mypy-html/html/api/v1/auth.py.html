"""Authentication API endpoints."""

import logging
from typing import Any, Dict

from fastapi import APIRouter, Depends, HTTPException, Request, status
from pydantic import BaseModel

# Safe imports with fallbacks
try:
    from auth.security_implementation import auth_manager, get_current_user
except ImportError:

    def auth_manager():
        return None

    def get_current_user():
        return None


try:
    from auth.security_logging import SecurityEventSeverity, SecurityEventType, security_auditor
except ImportError:

    class SecurityEventType:
        USER_CREATED = "user_created"
        LOGOUT = "logout"

    class SecurityEventSeverity:
        INFO = "info"
        WARNING = "warning"

    class SecurityAuditor:
        def log_event(self, *args, **kwargs):
            pass

    security_auditor = SecurityAuditor()

try:
    from api.v1.models.user_models import TokenData, UserLogin, UserRegistration
except ImportError:

    class UserRegistration(BaseModel):
        username: str
        password: str
        email: str = ""
        role: str = "user"

    class UserLogin(BaseModel):
        username: str
        password: str

    class TokenData(BaseModel):
        user_id: str
        username: str
        role: str
        permissions: list = []
        exp: object = None
        jti: str = ""


try:
    from api.middleware.rate_limiting import rate_limit
except ImportError:

    def rate_limit(**kwargs):
        def decorator(func):
            return func

        return decorator


try:
    from auth.rbac import Permission
except ImportError:

    class Permission:
        CREATE_AGENT = "create_agent"
        DELETE_AGENT = "delete_agent"
        VIEW_METRICS = "view_metrics"
        ADMIN_SYSTEM = "admin_system"


logger = logging.getLogger(__name__)
router = APIRouter()


class TokenResponse(BaseModel):
    """Token response model."""

    access_token: str
    token_type: str = "bearer"
    user: Dict[str, Any]


def log_login_success(username: str, user_id: str, request: Request):
    """Log successful login."""
    logger.info(f"User {username} logged in successfully")


def log_login_failure(username: str, request: Request, reason: str):
    """Log failed login attempt."""
    logger.warning(f"Failed login attempt for {username}: {reason}")


@router.post("/register", response_model=TokenResponse)
@rate_limit(endpoint_type="auth", max_requests=3, window_minutes=10)  # Very strict for registration
async def register_user(user_data: UserRegistration, request: Request):
    """Register User function."""
    try:
        # Create user
        user = auth_manager().create_user(
            username=user_data.username,
            password=user_data.password,
            email=user_data.email,
            role=user_data.role,
        )

        # Create tokens
        access_token, _ = auth_manager().create_access_token(user)
        refresh_token, _ = auth_manager().create_refresh_token(user)

        # Log successful registration
        security_auditor.log_event(
            SecurityEventType.USER_CREATED,
            SecurityEventSeverity.INFO,
            f"New user registered: {user.username}",
            request=request,
            user_id=user.user_id,
            username=user.username,
            details={"role": user.role, "email": user.email},
        )

        return TokenResponse(
            access_token=access_token, refresh_token=refresh_token, user=user.dict()
        )

    except Exception as e:
        # Log failed registration attempt
        security_auditor.log_event(
            SecurityEventType.USER_CREATED,
            SecurityEventSeverity.WARNING,
            f"Failed user registration attempt for {user_data.username}: {str(e)}",
            request=request,
            details={"username": user_data.username, "error": str(e)},
        )
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail=str(e))


@router.post("/login", response_model=TokenResponse)
@rate_limit(endpoint_type="auth", max_requests=5, window_minutes=5)  # Strict rate limit for login
async def login_user(login_data: UserLogin, request: Request):
    user = auth_manager().authenticate_user(login_data.username, login_data.password)

    if not user:
        # Log failed login attempt
        log_login_failure(login_data.username, request, "Invalid credentials")
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid credentials")

    if not user.is_active:
        # Log failed login due to disabled account
        log_login_failure(login_data.username, request, "Account is disabled")
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Account is disabled")
    access_token, access_jti = auth_manager().create_access_token(user)
    refresh_token, refresh_jti = auth_manager().create_refresh_token(user)

    # Log successful login
    log_login_success(user.username, user.user_id, request)

    return TokenResponse(access_token=access_token, refresh_token=refresh_token, user=user.dict())


@router.get("/me", response_model=Dict[str, Any])
async def get_current_user_info(current_user: TokenData = Depends(get_current_user)):
    return {
        "user_id": current_user.user_id,
        "username": current_user.username,
        "role": current_user.role,
        "permissions": current_user.permissions,
        "exp": current_user.exp.isoformat(),
    }


@router.post("/logout")
async def logout_user(request: Request, current_user: TokenData = Depends(get_current_user)):
    # Use the new secure logout method
    success = await auth_manager().logout_user(current_user)

    # Log logout event
    security_auditor.log_event(
        SecurityEventType.LOGOUT,
        SecurityEventSeverity.INFO,
        f"User {current_user.username} logged out (tokens blacklisted: {success})",
        request=request,
        user_id=current_user.user_id,
        username=current_user.username,
        details={"secure_logout": success, "jti": current_user.jti},
    )

    return {
        "message": "Successfully logged out",
        "secure_logout": success,
        "note": "All tokens have been invalidated",
    }


@router.get("/permissions")
async def get_user_permissions(current_user: TokenData = Depends(get_current_user)):
    return {
        "permissions": current_user.permissions,
        "role": current_user.role,
        "can_create_agents": Permission.CREATE_AGENT in current_user.permissions,
        "can_delete_agents": Permission.DELETE_AGENT in current_user.permissions,
        "can_view_metrics": Permission.VIEW_METRICS in current_user.permissions,
        "can_admin_system": Permission.ADMIN_SYSTEM in current_user.permissions,
    }
