"""API endpoints for knowledge graph visualization.

This module provides REST API endpoints for serving graph visualization data,
including agent relationships, system topology, and knowledge structures.
"""

import logging
from datetime import datetime
from typing import Any, Dict, List, Optional

from fastapi import APIRouter, HTTPException, status
from pydantic import BaseModel, Field
from sqlalchemy import and_, select
from sqlalchemy.orm import joinedload

from database.conversation_models import Conversation
from database.knowledge_graph_models import (
    KnowledgeGraphConcept,
    KnowledgeGraphEntity,
    KnowledgeGraphRelationship,
)
from database.models import Agent
from database.session import get_db_context
from knowledge_graph.visualization_models import (
    GraphLayout,
    VisualizationConfig,
    VisualizationEdgeType,
    VisualizationNodeType,
    create_agent_relationship_graph,
    create_knowledge_structure_graph,
    create_system_topology_graph,
)

logger = logging.getLogger(__name__)
router = APIRouter()


class GraphVisualizationRequest(BaseModel):
    """Request model for graph visualization."""

    graph_type: str = Field(..., description="Type of graph to visualize")
    agent_ids: Optional[List[str]] = Field(None, description="Filter by agent IDs")
    coalition_ids: Optional[List[str]] = Field(None, description="Filter by coalition IDs")
    conversation_ids: Optional[List[str]] = Field(None, description="Filter by conversation IDs")
    time_range: Optional[Dict[str, str]] = Field(None, description="Time range filter")
    node_types: Optional[List[str]] = Field(None, description="Filter by node types")
    edge_types: Optional[List[str]] = Field(None, description="Filter by edge types")
    confidence_threshold: float = Field(
        0.0, ge=0.0, le=1.0, description="Minimum confidence threshold"
    )
    max_nodes: int = Field(1000, ge=1, le=10000, description="Maximum number of nodes")
    max_edges: int = Field(2000, ge=1, le=20000, description="Maximum number of edges")
    layout_algorithm: str = Field("force_directed", description="Layout algorithm to use")
    include_metadata: bool = Field(True, description="Include metadata in response")


class GraphVisualizationResponse(BaseModel):
    """Response model for graph visualization."""

    graph_id: str
    name: str
    description: str
    nodes: List[Dict[str, Any]]
    edges: List[Dict[str, Any]]
    layout: Dict[str, Any]
    config: Dict[str, Any]
    metadata: Dict[str, Any]
    created_at: str
    updated_at: str
    version: int


class AgentRelationshipRequest(BaseModel):
    """Request model for agent relationship visualization."""

    agent_ids: Optional[List[str]] = Field(None, description="Specific agent IDs to include")
    include_coalitions: bool = Field(True, description="Include coalition relationships")
    include_conversations: bool = Field(True, description="Include conversation relationships")
    include_belief_states: bool = Field(False, description="Include belief state information")
    relationship_types: Optional[List[str]] = Field(
        None, description="Filter by relationship types"
    )
    time_range: Optional[Dict[str, str]] = Field(None, description="Time range filter")


class SystemTopologyRequest(BaseModel):
    """Request model for system topology visualization."""

    component_types: Optional[List[str]] = Field(None, description="Filter by component types")
    include_performance_metrics: bool = Field(True, description="Include performance metrics")
    include_health_status: bool = Field(True, description="Include health status")
    connection_types: Optional[List[str]] = Field(None, description="Filter by connection types")


class KnowledgeStructureRequest(BaseModel):
    """Request model for knowledge structure visualization."""

    agent_ids: Optional[List[str]] = Field(None, description="Filter by agent IDs")
    knowledge_domains: Optional[List[str]] = Field(None, description="Filter by knowledge domains")
    entity_types: Optional[List[str]] = Field(None, description="Filter by entity types")
    confidence_threshold: float = Field(
        0.0, ge=0.0, le=1.0, description="Minimum confidence threshold"
    )
    include_concepts: bool = Field(True, description="Include concept nodes")
    include_relationships: bool = Field(True, description="Include relationship edges")
    time_range: Optional[Dict[str, str]] = Field(None, description="Time range filter")


@router.post("/visualizations/agent-relationships", response_model=GraphVisualizationResponse)
async def get_agent_relationships(request: AgentRelationshipRequest) -> GraphVisualizationResponse:
    """Get agent relationship visualization data."""
    try:
        with get_db_context() as session:
            agents = await _fetch_agents_for_visualization(session, request)
            agent_data = _convert_agents_to_visualization_format(agents, request)
            relationships = await _build_agent_relationships(session, agents, request)
            graph = _create_visualization_graph(agent_data, relationships, request)

            return _build_graph_response(graph, agent_data, relationships)

    except Exception as e:
        logger.error(f"Error generating agent relationships visualization: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Failed to generate agent relationships: {str(e)}",
        )


async def _fetch_agents_for_visualization(session, request: AgentRelationshipRequest):
    """Fetch agents from database based on request filters."""
    agent_query = select(Agent).options(joinedload(Agent.coalitions))

    if request.agent_ids:
        agent_query = agent_query.where(Agent.id.in_(request.agent_ids))

    # Apply time range filter
    if request.time_range:
        start_time = datetime.fromisoformat(request.time_range["start"])
        end_time = datetime.fromisoformat(request.time_range["end"])
        agent_query = agent_query.where(
            and_(Agent.created_at >= start_time, Agent.created_at <= end_time)
        )

    result = await session.execute(agent_query)
    return result.scalars().all()


def _convert_agents_to_visualization_format(agents, request: AgentRelationshipRequest):
    """Convert agent models to visualization format."""
    agent_data = []
    for agent in agents:
        agent_dict = {
            "id": str(agent.id),
            "name": agent.name,
            "status": agent.status.value,
            "beliefs": agent.beliefs or {},
            "properties": {
                "template": agent.template,
                "position": agent.position,
                "metrics": agent.metrics or {},
                "parameters": agent.parameters or {},
                "preferences": agent.preferences or {},
            },
            "created_at": agent.created_at.isoformat(),
            "updated_at": agent.updated_at.isoformat(),
        }

        if request.include_belief_states:
            agent_dict["belief_state"] = agent.beliefs or {}

        agent_data.append(agent_dict)

    return agent_data


async def _build_agent_relationships(session, agents, request: AgentRelationshipRequest):
    """Build relationships between agents."""
    relationships = []

    # Add coalition relationships
    if request.include_coalitions:
        relationships.extend(_build_coalition_relationships(agents))

    # Add conversation relationships
    if request.include_conversations:
        conv_relationships = await _build_conversation_relationships(session, agents, request)
        relationships.extend(conv_relationships)

    return relationships


def _build_coalition_relationships(agents):
    """Build coalition-based relationships."""
    relationships = []
    for agent in agents:
        for coalition in agent.coalitions:
            for other_agent in coalition.agents:
                if other_agent.id != agent.id:
                    relationships.append(
                        {
                            "source_agent_id": str(agent.id),
                            "target_agent_id": str(other_agent.id),
                            "type": "coalition_member",
                            "coalition_id": str(coalition.id),
                            "trust_level": 0.8,
                            "collaboration_strength": 0.7,
                            "properties": {
                                "coalition_name": coalition.name,
                                "coalition_status": coalition.status.value,
                            },
                        }
                    )
    return relationships


async def _build_conversation_relationships(session, agents, request: AgentRelationshipRequest):
    """Build conversation-based relationships."""
    conversation_query = select(Conversation).options(joinedload(Conversation.messages))
    if request.agent_ids:
        conversation_query = conversation_query.where(Conversation.agent_id.in_(request.agent_ids))

    conv_result = await session.execute(conversation_query)
    conversations = conv_result.scalars().all()

    # Track agent interactions through conversations
    interaction_counts = {}
    for conversation in conversations:
        agent_id = str(conversation.agent_id)
        if agent_id not in interaction_counts:
            interaction_counts[agent_id] = 0
        interaction_counts[agent_id] += len(conversation.messages)

    # Create conversation-based relationships
    relationships = []
    for agent_id, count in interaction_counts.items():
        if count > 0:
            relationships.append(
                {
                    "source_agent_id": agent_id,
                    "target_agent_id": agent_id,
                    "type": "conversation_activity",
                    "trust_level": min(1.0, count / 100.0),
                    "collaboration_strength": min(1.0, count / 50.0),
                    "properties": {
                        "message_count": count,
                        "activity_type": "conversation",
                    },
                }
            )

    return relationships


def _create_visualization_graph(agent_data, relationships, request: AgentRelationshipRequest):
    """Create visualization graph from agent data and relationships."""
    graph = create_agent_relationship_graph(agent_data, relationships)

    # Apply layout algorithm
    if request.time_range:
        graph.layout.algorithm = "hierarchical"
    else:
        graph.layout.algorithm = "force_directed"

    return graph


def _build_graph_response(graph, agent_data, relationships):
    """Build the graph response object."""
    return GraphVisualizationResponse(
        graph_id=graph.graph_id,
        name=graph.name,
        description=graph.description,
        nodes=[node.to_dict() for node in graph.nodes.values()],
        edges=[edge.to_dict() for edge in graph.edges.values()],
        layout=graph.layout.to_dict(),
        config=graph.config.to_dict(),
        metadata={
            "agent_count": len(agent_data),
            "relationship_count": len(relationships),
            "coalition_count": len(
                set(r.get("coalition_id") for r in relationships if r.get("coalition_id"))
            ),
            "generated_at": datetime.now().isoformat(),
        },
        created_at=graph.created_at.isoformat(),
        updated_at=graph.updated_at.isoformat(),
        version=graph.version,
    )

@router.post("/visualizations/knowledge-structure", response_model=GraphVisualizationResponse)
async def get_knowledge_structure(request: KnowledgeStructureRequest) -> GraphVisualizationResponse:
    """Get knowledge structure visualization data."""
    try:
        with get_db_context() as session:
            entities = await _fetch_knowledge_entities(session, request)
            knowledge_entities = _convert_entities_to_visualization_format(entities)

            relationships = await _fetch_knowledge_relationships(session, request)
            knowledge_relationships = _convert_relationships_to_visualization_format(
                relationships, entities
            )

            if request.include_concepts:
                concepts = await _fetch_knowledge_concepts(session, request)
                concept_entities = _convert_concepts_to_visualization_format(concepts)
                knowledge_entities.extend(concept_entities)

            graph = _create_knowledge_graph(knowledge_entities, knowledge_relationships, request)

            return _build_knowledge_graph_response(
                graph, knowledge_entities, knowledge_relationships, request
            )

    except Exception as e:
        logger.error(f"Error generating knowledge structure visualization: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Failed to generate knowledge structure: {str(e)}",
        )


async def _fetch_knowledge_entities(session, request: KnowledgeStructureRequest):
    """Fetch knowledge entities from database."""
    entity_query = select(KnowledgeGraphEntity)

    if request.agent_ids:
        entity_query = entity_query.where(KnowledgeGraphEntity.agent_id.in_(request.agent_ids))

    if request.entity_types:
        entity_query = entity_query.where(
            KnowledgeGraphEntity.entity_type.in_(request.entity_types)
        )

    if request.confidence_threshold > 0:
        entity_query = entity_query.where(
            KnowledgeGraphEntity.confidence >= request.confidence_threshold
        )

    # Apply time range filter
    if request.time_range:

        )

    return entity_result.scalars().all()

def _convert_relationships_to_visualization_format(relationships, entities):
    """Convert relationships to visualization format."""
    knowledge_relationships = []
    entity_ids = set(str(e.id) for e in entities)

    for relationship in relationships:
        # Only include relationships between entities we have
        if (
            str(relationship.source_entity_id) in entity_ids
            and str(relationship.target_entity_id) in entity_ids
        ):
            rel_dict = {
                "source_entity_id": str(relationship.source_entity_id),
                "target_entity_id": str(relationship.target_entity_id),
                "relationship_type": relationship.relationship_type,
                "confidence": relationship.confidence,
                "derivation_method": "inference",
                "evidence_strength": relationship.confidence,
                "properties": relationship.properties or {},
                "created_at": relationship.created_at.isoformat(),
                "updated_at": relationship.updated_at.isoformat(),
                "mention_count": relationship.mention_count,
            }
            knowledge_relationships.append(rel_dict)

    return knowledge_relationships

def _create_knowledge_graph(
    knowledge_entities, knowledge_relationships, request: KnowledgeStructureRequest
):
    """Create knowledge structure graph."""
    graph = create_knowledge_structure_graph(knowledge_entities, knowledge_relationships)

    # Configure for knowledge structure
    graph.layout.algorithm = "force_directed"
    graph.config.show_labels = True
    graph.config.show_tooltips = True
    graph.config.confidence_threshold = request.confidence_threshold

    return graph


def _build_knowledge_graph_response(
    graph, knowledge_entities, knowledge_relationships, request: KnowledgeStructureRequest
):
    """Build knowledge graph response."""
    return GraphVisualizationResponse(
        graph_id=graph.graph_id,
        name=graph.name,
        description=graph.description,
        nodes=[node.to_dict() for node in graph.nodes.values()],
        edges=[edge.to_dict() for edge in graph.edges.values()],
        layout=graph.layout.to_dict(),
        config=graph.config.to_dict(),
        metadata={
            "entity_count": len(knowledge_entities),
            "relationship_count": len(knowledge_relationships),
            "entity_types": list(set(e["entity_type"] for e in knowledge_entities)),
            "relationship_types": list(
                set(r["relationship_type"] for r in knowledge_relationships)
            ),
            "confidence_threshold": request.confidence_threshold,
            "generated_at": datetime.now().isoformat(),
        },
        created_at=graph.created_at.isoformat(),
        updated_at=graph.updated_at.isoformat(),
        version=graph.version,
    )
