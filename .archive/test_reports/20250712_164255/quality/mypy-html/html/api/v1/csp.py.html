"""
Content Security Policy (CSP) violation reporting endpoint.
"""

import json
import logging
from datetime import datetime
from typing import Any, Dict, List

from fastapi import APIRouter, Request, Response, status
from pydantic import BaseModel

from api.v1.headers import rate_limit
from auth.security_logging import SecurityEventSeverity, SecurityEventType, security_auditor

logger = logging.getLogger(__name__)
router = APIRouter()


class CSPReport(BaseModel):
    """
        CSP violation report model.

    blocked_uri: str = ""
    document_uri: str = ""
    violated_directive: str = ""
    effective_directive: str = ""
    original_policy: str = ""
    disposition: str = "enforce"
    referrer: str = ""
    status_code: int = 0
    script_sample: str = ""
    line_number: int = 0
    column_number: int = 0
    source_file: str = ""
    """


class CSPReportWrapper(BaseModel):
    """Wrapper for CSP violation reports."""

    csp_report: Dict[str, Any] = {}

    class Config:
        schema_extra = {
            "example": {
                "csp-report": {
                    "blocked-uri": "https://evil.com/script.js",
                    "document-uri": "https://example.com/page",
                    "violated-directive": "script-src",
                    "effective-directive": "script-src",
                    "original-policy": "default-src 'sel'; script-src 'self'",
                    "disposition": "enforce",
                    "referrer": "",
                    "status-code": 0,
                    "script-sample": "",
                }
            }
        }


# In-memory storage for recent violations (for monitoring)
recent_violations: List[Dict[str, Any]] = []
MAX_STORED_VIOLATIONS = 1000


@router.post("/csp-report", status_code=status.HTTP_204_NO_CONTENT)
@rate_limit(endpoint_type="public", max_requests=100, window_minutes=1)
async def report_csp_violation(request: Request):
    """
    Handle CSP violation reports.
    """
    try:
        # Get request body
        body = await request.body()

        # Parse JSON
        try:
            report_data = json.loads(body)
            violation = report_data.get("csp-report", report_data)

        except json.JSONDecodeError:
            logger.warning(f"Invalid CSP report format: {body[:200]}")
            return Response(status_code=status.HTTP_204_NO_CONTENT)

        # Extract key information
        blocked_uri = violation.get("blocked-uri", "unknown")
        document_uri = violation.get("document-uri", "unknown")
        violated_directive = violation.get("violated-directive", "unknown")
        effective_directive = violation.get("effective-directive", violated_directive)

        # Get client information
        client_ip = request.client.host if request.client else "unknown"
        user_agent = request.headers.get("User-Agent", "unknown")

        # Log the violation for security monitoring
        severity = SecurityEventSeverity.WARNING

        # Elevate severity for certain violations
        if "eval" in blocked_uri or "inline" in blocked_uri:
            severity = SecurityEventSeverity.HIGH
        elif any(domain in blocked_uri for domain in ["javascript:", "data:", "blob:"]):
            severity = SecurityEventSeverity.HIGH

        # Log to security audit
        await security_auditor.log_security_event(
            event_type=SecurityEventType.SUSPICIOUS_ACTIVITY,
            severity=severity,
            details={
                "violation_type": "csp_violation",
                "blocked_uri": blocked_uri,
                "document_uri": document_uri,
                "violated_directive": violated_directive,
                "effective_directive": effective_directive,
                "source_file": violation.get("source-file", ""),
                "line_number": violation.get("line-number", 0),
                "script_sample": violation.get("script-sample", "")[:100],  # Limit sample size
                "disposition": violation.get("disposition", "enforce"),
                "client_ip": client_ip,
                "user_agent": user_agent,
                "timestamp": datetime.utcnow().isoformat(),
            },
        )

        # Store recent violations for monitoring
        violation_summary = {
            "timestamp": datetime.utcnow().isoformat(),
            "blocked_uri": blocked_uri,
            "document_uri": document_uri,
            "directive": effective_directive,
            "client_ip": client_ip,
        }

        recent_violations.append(violation_summary)

        # Limit stored violations
        if len(recent_violations) > MAX_STORED_VIOLATIONS:
            recent_violations.pop(0)

        # Log for debugging
        logger.info(
            f"CSP violation: {effective_directive} blocked {blocked_uri} " f"on {document_uri}"
        )

        # Return 204 No Content as per CSP reporting spec
        return Response(status_code=status.HTTP_204_NO_CONTENT)

    except Exception as e:
        logger.error(f"Error processing CSP report: {e}")
        # Still return 204 to prevent browser retries
        return Response(status_code=status.HTTP_204_NO_CONTENT)


@router.get("/csp-violations")
@rate_limit(endpoint_type="admin")
async def get_recent_csp_violations(
    limit: int = 100,
) -> Dict[str, Any]:
    """
    Get recent CSP violations for monitoring.
    """
    # Prepare summary statistics
    directive_counts = {}
    blocked_uri_counts = {}

    # Process violations
    for violation in recent_violations[-limit:]:
        directive = violation.get("directive", "unknown")
        blocked_uri = violation.get("blocked_uri", "unknown")

        directive_counts[directive] = directive_counts.get(directive, 0) + 1

        # Normalize blocked URIs for counting
        if blocked_uri.startswith("data:"):
            blocked_uri = "data:"
        elif blocked_uri.startswith("blob:"):
            blocked_uri = "blob:"
        elif blocked_uri == "eval":
            blocked_uri = "eval"
        elif blocked_uri == "inline":
            blocked_uri = "inline"

        blocked_uri_counts[blocked_uri] = blocked_uri_counts.get(blocked_uri, 0) + 1

    # Get limited list of recent violations
    violations = recent_violations[-limit:] if recent_violations else []

    return {
        "total_violations": len(recent_violations),
        "recent_violations": violations,
        "statistics": {
            "by_directive": directive_counts,
            "by_blocked_uri": dict(
                sorted(blocked_uri_counts.items(), key=lambda x: x[1], reverse=True)[:10]  # Top 10
            ),
            "time_range": {
                "oldest": (recent_violations[0]["timestamp"] if recent_violations else None),
                "newest": (recent_violations[-1]["timestamp"] if recent_violations else None),
            },
        },
    }
