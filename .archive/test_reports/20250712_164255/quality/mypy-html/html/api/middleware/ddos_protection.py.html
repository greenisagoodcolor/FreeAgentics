"""DDoS Protection and Request Size Limiting Middleware."""

import logging
import time
from typing import Any, Callable, Dict, List

from fastapi import HTTPException, Request, Response, status

logger = logging.getLogger(__name__)


class DDoSProtectionMiddleware:
    """Comprehensive DDoS protection middleware with request size limits."""

    def __init__(
        self,
        app,
        max_request_size: int = 10485760,
        max_requests_per_ip: int = 60,
        blocked_user_agents=None,
    ):
        """Initialize DDoS protection middleware."""
        self.app = app
        self.max_request_size = max_request_size
        self.max_requests_per_ip = max_requests_per_ip
        self.request_count = {}
        self.attack_thresholds = {
            "rapid_requests": 100,  # More than 100 requests in 10 seconds
            "large_payload_attacks": 5,  # More than 5 large requests in 1 minute
            "path_traversal_attempts": 10,  # More than 10 path traversal attempts
        }

        # Block common bot/attack user agents
        self.blocked_user_agents = blocked_user_agents or [
            "nikto",
            "sqlmap",
            "nmap",
            "masscan",
            "python-requests/2.25.1",  # Common for automated attacks
            "curl/7.",  # Often used in automated attacks
            "wget/",
            "bot",
            "crawler",
            "spider",
        ]

        # In-memory tracking (in production, use Redis)
        self.request_counts: Dict[str, List[float]] = {}
        self.blocked_ips: Dict[str, float] = {}  # IP -> block_until_timestamp
        self.suspicious_activity: Dict[str, Dict[str, int]] = {}

    def _get_client_ip(self, request: Request) -> str:
        """Extract real client IP address."""
        forwarded_for = request.headers.get("X-Forwarded-For")

        if forwarded_for:
            return forwarded_for.split(",")[0].strip()

        real_ip = request.headers.get("X-Real-IP")
        if real_ip:
            return real_ip.strip()

        return request.client.host if request.client else "unknown"

    def _is_suspicious_user_agent(self, user_agent: str) -> bool:
        """Check if user agent is suspicious."""
        if not user_agent:
            return True

        user_agent_lower = user_agent.lower()
        return any(blocked in user_agent_lower for blocked in self.blocked_user_agents)

    def _detect_path_traversal(self, path: str) -> bool:
        """Detect path traversal attempts."""
        suspicious_patterns = [
            "../",
            "..\\",
            "/..",
            "\\...",
            "%2e%2e%2",
            "%2e%2e%5c",
            "%252e%252e%252",
            "etc/passwd",
            "windows/system32",
            "web.config",
        ]
        path_lower = path.lower()
        return any(pattern in path_lower for pattern in suspicious_patterns)

    def _is_ip_blocked(self, ip: str) -> bool:
        """Check if IP is currently blocked."""
        if ip in self.blocked_ips:
            if time.time() < self.blocked_ips[ip]:
                return True
            else:
                del self.blocked_ips[ip]
        return False

    def _block_ip(self, ip: str, duration: int) -> None:
        """Block IP for specified duration."""
        self.blocked_ips[ip] = time.time() + duration
        logger.warning(f"Blocked IP {ip} for {duration} seconds due to suspicious activity")

    def _track_request(self, ip: str) -> bool:
        """Track request and check for rate limiting."""
        now = time.time()
        minute_ago = now - 60
        ten_seconds_ago = now - 10

        if ip not in self.request_counts:
            self.request_counts[ip] = []

        # Clean old requests
        self.request_counts[ip] = [
            req_time for req_time in self.request_counts[ip] if req_time > minute_ago
        ]

        # Add current request
        self.request_counts[ip].append(now)

        # Check for rapid requests (100+ in 10 seconds)
        recent_requests = [
            req_time for req_time in self.request_counts[ip] if req_time > ten_seconds_ago
        ]

        if len(recent_requests) > self.attack_thresholds["rapid_requests"]:
            logger.warning(
                f"Rapid request pattern detected from IP {ip}: {len(recent_requests)} "
                "requests in 10s"
            )
            self._block_ip(ip, 600)  # Block for 10 minutes
            return False

        # Check total requests per minute
        if len(self.request_counts[ip]) > self.max_requests_per_ip:
            logger.warning(
                f"Rate limit exceeded for IP {ip}: {len(self.request_counts[ip])} "
                "requests/minute"
            )
            self._block_ip(ip, 300)  # Block for 5 minutes
            return False

        return True

    def _check_request_size(self, request: Request) -> bool:
        """Check if request size is within limits."""
        content_length = request.headers.get("content-length")

        if content_length:
            try:
                size = int(content_length)

                if size > self.max_request_size:
                    logger.warning(
                        f"Large request blocked: {size} bytes from "
                        f"{self._get_client_ip(request)}"
                    )
                    return False
            except ValueError:
                pass
        return True

    async def dispatch(self, request: Request, call_next: Callable) -> Response:
        """Process incoming request with DDoS protection checks."""
        start_time = time.time()
        client_ip = self._get_client_ip(request)

        # 1. Check if IP is blocked
        if self._is_ip_blocked(client_ip):
            logger.warning(f"Blocked IP {client_ip} attempted access")
            raise HTTPException(
                status_code=status.HTTP_429_TOO_MANY_REQUESTS,
                detail="IP temporarily blocked due to suspicious activity",
            )

        # 2. Check request size limits
        if not self._check_request_size(request):
            self._block_ip(client_ip, 60)  # Short block for oversized requests
            raise HTTPException(
                status_code=status.HTTP_413_REQUEST_ENTITY_TOO_LARGE,
                detail=(
                    f"Request too large. Maximum size: "
                    f"{self.max_request_size // (1024 * 1024)}MB"
                ),
            )

        # 3. Check user agent
        user_agent = request.headers.get("user-agent", "")

        if self._is_suspicious_user_agent(user_agent):
            logger.warning(f"Suspicious user agent blocked: {user_agent} from {client_ip}")
            self._block_ip(client_ip, 1800)  # Block for 30 minutes
            raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Access denied")

        # 4. Check for path traversal attempts
        if self._detect_path_traversal(str(request.url.path)):
            logger.warning(f"Path traversal attempt from {client_ip}: {request.url.path}")
            self._block_ip(client_ip, 3600)  # Block for 1 hour
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN, detail="Invalid request path"
            )

        # 5. Track request rate and detect rapid fire
        if not self._track_request(client_ip):
            raise HTTPException(
                status_code=status.HTTP_429_TOO_MANY_REQUESTS,
                detail="Too many requests. Please slow down.",
            )

        # Process the request
        try:
            response = await call_next(request)

            # Add security headers to response
            response.headers["X-Content-Type-Options"] = "nosniff"
            response.headers["X-Frame-Options"] = "DENY"
            response.headers["X-XSS-Protection"] = "1; mode=block"
            response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"

            # Add timing header for monitoring
            process_time = time.time() - start_time
            response.headers["X-Process-Time"] = str(process_time)

            return response

        except Exception as e:
            # Log potential attack patterns
            if "injection" in str(e).lower() or "sql" in str(e).lower():
                logger.error(f"Potential injection attempt from {client_ip}: {str(e)}")
                self._block_ip(client_ip, 7200)  # Block for 2 hours
            raise

    def get_stats(self) -> Dict[str, Any]:
        """Get DDoS protection statistics."""
        active_blocks = sum(
            1 for block_time in self.blocked_ips.values() if block_time > time.time()
        )

        return {
            "blocked_ips_count": active_blocks,
            "tracked_ips_count": len(self.request_counts),
            "max_request_size_mb": self.max_request_size // (1024 * 1024),
            "max_requests_per_minute": self.max_requests_per_ip,
            "blocked_user_agents": len(self.blocked_user_agents),
            "blocked_ips": (
                list(self.blocked_ips.keys())
                if len(self.blocked_ips) < 50
                else f"{len(self.blocked_ips)} IPs blocked"
            ),
        }

    def unblock_ip(self, ip: str) -> bool:
        """Manually unblock an IP address."""
        if ip in self.blocked_ips:
            del self.blocked_ips[ip]
            logger.info(f"Manually unblocked IP: {ip}")
            return True
        return False

    async def __call__(self, request: Request, call_next: Callable) -> Response:
        """Make middleware callable."""
        return await self.dispatch(request, call_next)
