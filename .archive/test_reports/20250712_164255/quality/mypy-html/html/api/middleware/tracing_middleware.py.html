class WebSocketTracingMiddleware:
    def __init__(self, app: ASGIApp):
        self.app = app
        self.tracer = get_tracer()

    async def __call__(self, scope, receive, send):
        if not self._should_trace(scope):
            await self.app(scope, receive, send)
            return

        # Extract trace context from headers
        trace_headers = self._extract_trace_headers(scope)
        if trace_headers:
            # self.tracer.extract_context(trace_headers)
            pass

        # Create span attributes
        span_attrs = self._create_span_attributes(scope)
        path = scope.get("path", "/")

        async with self.tracer.async_span(
            name=f"websocket {path}",
            kind="server",
            attributes=span_attrs,
        ) as span:
            await self._handle_traced_connection(scope, receive, send, span)

    def _should_trace(self, scope) -> bool:
        """Check if this request should be traced."""
        return (
            scope["type"] == "websocket"
            and self.tracer is not None
            and hasattr(self.tracer, "_initialized")
            and self.tracer._initialized
        )

    def _extract_trace_headers(self, scope) -> dict:
        """Extract trace headers from the request scope."""
        headers = dict(scope.get("headers", []))
        trace_headers = {}

        for name, value in headers.items():
            name = self._decode_header_value(name)
            value = self._decode_header_value(value)

            if name.lower() in ["traceparent", "tracestate", "baggage"]:
                trace_headers[name] = value

        return trace_headers

    def _decode_header_value(self, value):
        """Decode header value from bytes if necessary."""
        if isinstance(value, bytes):
            return value.decode("latin-1")
        return value

    def _create_span_attributes(self, scope) -> dict:
        """Create attributes for the tracing span."""
        path = scope.get("path", "/")
        return {
            "websocket.path": path,
            "websocket.query_string": scope.get("query_string", b"").decode("latin-1"),
            "websocket.client": self._get_client_info(scope),
        }

    def _get_client_info(self, scope) -> str:
        """Extract client information from scope."""
        client = scope.get("client")
        if client and isinstance(client, (list, tuple)) and len(client) > 0:
            return client[0]
        return "unknown"

    async def _handle_traced_connection(self, scope, receive, send, span):
        """Handle the traced WebSocket connection."""
        _ = time.time()
        message_count = 0

        # Create traced receive and send functions
        traced_receive = self._create_traced_receive(receive, span, message_count)
        traced_send = self._create_traced_send(send, span)

        try:
            await self.app(scope, traced_receive, traced_send)
            self._set_span_success(span)
        except Exception:
            self._set_span_error(span)
            raise
        finally:
            self._record_final_metrics(span, message_count)
