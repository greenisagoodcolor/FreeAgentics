"""GMN Validation Middleware."""

import logging
import time
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

from fastapi import Request
from fastapi.responses import JSONResponse
from starlette.middleware.base import BaseHTTPMiddleware

logger = logging.getLogger(__name__)


@dataclass
class GMNValidationRule:
    """Represents a GMN validation rule."""

    name: str
    description: str
    severity: str = "error"  # "error", "warning", "info"
    enabled: bool = True


@dataclass
class GMNValidationResult:
    """Result of GMN validation."""

    is_valid: bool = True
    errors: List[str] = field(default_factory=list)
    warnings: List[str] = field(default_factory=list)
    info: List[str] = field(default_factory=list)
    metadata: Dict[str, Any] = field(default_factory=dict)
    validation_time_ms: float = 0.0


@dataclass
class GMNValidationConfig:
    """Configuration for GMN validation middleware."""

    max_specification_length: int = 100000
    max_nodes: int = 1000
    max_edges: int = 5000
    max_node_name_length: int = 100
    require_belief_nodes: bool = True
    check_node_connectivity: bool = True
    enforce_acyclic_dependencies: bool = False
    validate_llm_integration: bool = True
    allowed_node_types: Optional[List[str]] = None
    allowed_edge_types: Optional[List[str]] = None
    custom_rules: List[GMNValidationRule] = field(default_factory=list)


class GMNValidationCache:
    """Simple cache for validation results."""

    def __init__(self, ttl_seconds: int = 300):
        """Initialize cache with TTL."""
        self.cache: Dict[str, Dict[str, Any]] = {}
        self.ttl_seconds = ttl_seconds

    def get(self, key: str) -> Optional[GMNValidationResult]:
        """Get cached result if still valid."""
        if key in self.cache:
            entry = self.cache[key]
            if time.time() - entry["timestamp"] < self.ttl_seconds:
                return entry["result"]
            else:
                del self.cache[key]
        return None

    def set(self, key: str, result: GMNValidationResult) -> None:
        """Cache a validation result."""
        self.cache[key] = {"result": result, "timestamp": time.time()}

    def clear(self) -> None:
        """Clear all cached entries."""
        self.cache.clear()


class GMNValidator:
    """Comprehensive GMN specification validator."""

    def __init__(self, config: GMNValidationConfig, cache: Optional[GMNValidationCache] = None):
        """Initialize validator with configuration."""
        self.config = config
        self.cache = cache
        self.parser = None  # Will be set if parser is available

    async def validate_gmn_specification(
        self, specification: str, context: Dict[str, Any] = None
    ) -> GMNValidationResult:
        """Validate a GMN specification."""
        start_time = time.time()
        context = context or {}

        # Check cache
        cached_result = await self._check_cache(specification)
        if cached_result:
            return cached_result

        # Perform validation
        result = await self._perform_validation(specification, context)

        # Finalize result
        await self._finalize_result(result, specification, start_time)

        return result

    async def _check_cache(self, specification: str) -> Optional[GMNValidationResult]:
        """Check cache for validation result."""
        if not self.cache:
            return None

        cache_key = f"gmn_spec_{hash(specification)}"
        return self.cache.get(cache_key)

    async def _perform_validation(
        self, specification: str, context: Dict[str, Any]
    ) -> GMNValidationResult:
        """Perform the actual validation."""
        result = GMNValidationResult(is_valid=True)

        try:
            # Basic format validation
            await self._validate_basic_format(specification, result)
            if not result.is_valid:
                return result

            # Parse and validate structure
            graph = await self._parse_specification(specification, result)
            if graph and result.is_valid:
                await self._validate_structure(graph, result, context)
                await self._validate_business_rules(graph, result, context)

        except Exception as e:
            result.errors.append(f"Validation error: {str(e)}")
            result.is_valid = False

        return result

    async def _parse_specification(
        self, specification: str, result: GMNValidationResult
    ) -> Optional[Any]:
        """Parse GMN specification if parser available."""
        if not self.parser:
            return None

        try:
            graph = self.parser.parse(specification)
            result.metadata["parsed_graph"] = {
                "num_nodes": len(graph.nodes),
                "num_edges": len(graph.edges),
            }
            return graph
        except Exception as e:
            result.errors.append(f"Failed to parse GMN specification: {str(e)}")
            result.is_valid = False
            return None

    async def _finalize_result(
        self, result: GMNValidationResult, specification: str, start_time: float
    ):
        """Finalize validation result and cache if enabled."""
        result.validation_time_ms = (time.time() - start_time) * 1000
        result.is_valid = len(result.errors) == 0

        # Cache result if enabled
        if self.cache:
            cache_key = f"gmn_spec_{hash(specification)}"
            self.cache.set(cache_key, result)

    async def _validate_basic_format(self, specification: str, result: GMNValidationResult) -> None:
        """Validate basic format requirements."""
        if len(specification) > self.config.max_specification_length:
            result.errors.append(
                f"Specification too long: {len(specification)} chars "
                f"(max: {self.config.max_specification_length})"
            )

        # Basic structure check
        if "[nodes]" not in specification:
            result.errors.append("Missing [nodes] section")

        if "[edges]" not in specification:
            result.errors.append("Missing [edges] section")

        # Check for dangerous content
        dangerous_patterns = ["import", "exec", "eval", "__", "subprocess", "os.system"]
        for pattern in dangerous_patterns:
            if pattern in specification.lower():
                result.errors.append(f"Potentially dangerous content detected: {pattern}")

    async def _validate_structure(
        self, graph: Any, result: GMNValidationResult, context: Dict[str, Any]
    ) -> None:
        """Validate graph structure."""
        if hasattr(graph, "nodes") and len(graph.nodes) > self.config.max_nodes:
            result.errors.append(
                f"Too many nodes: {len(graph.nodes)} (max: {self.config.max_nodes})"
            )

        if hasattr(graph, "edges") and len(graph.edges) > self.config.max_edges:
            result.errors.append(
                f"Too many edges: {len(graph.edges)} (max: {self.config.max_edges})"
            )

        # Check for empty graph
        if hasattr(graph, "nodes") and len(graph.nodes) == 0:
            result.errors.append("Graph has no nodes")

    async def _validate_business_rules(
        self, graph: Any, result: GMNValidationResult, context: Dict[str, Any]
    ) -> None:
        """Validate business rules."""
        # Check belief nodes requirement
        if self.config.require_belief_nodes and hasattr(graph, "nodes"):
            # This would need proper node type checking
            result.warnings.append("Belief node validation not implemented")


class GMNValidationMiddleware(BaseHTTPMiddleware):
    """FastAPI middleware for GMN validation."""

    def __init__(self, app, config: GMNValidationConfig):
        """Initialize middleware."""
        super().__init__(app)
        self.config = config
        self.cache = GMNValidationCache()
        self.validator = GMNValidator(config, self.cache)

    async def dispatch(self, request: Request, call_next):
        """Process request through validation middleware."""
        # Check if request needs validation
        if not self._should_validate_request(request):
            return await call_next(request)

        # Extract GMN specification
        try:
            gmn_spec = await self._extract_gmn_specification(request)
            if not gmn_spec:
                return await call_next(request)
        except Exception as e:
            logger.error(f"Failed to extract GMN specification: {e}")
            return JSONResponse(
                status_code=400, content={"detail": f"Invalid request format: {str(e)}"}
            )

        # Validate specification
        try:
            validation_result = await self.validator.validate_gmn_specification(gmn_spec)

            if not validation_result.is_valid:
                return JSONResponse(
                    status_code=422,
                    content={
                        "detail": "GMN specification validation failed",
                        "errors": validation_result.errors,
                        "warnings": validation_result.warnings,
                    },
                )

            # Add validation metadata to request for downstream use
            request.state.gmn_validation_result = validation_result

        except Exception as e:
            logger.error(f"GMN validation error: {e}")
            return JSONResponse(status_code=500, content={"detail": f"Validation error: {str(e)}"})

        return await call_next(request)

    def _should_validate_request(self, request: Request) -> bool:
        """Check if request should be validated."""
        if request.method not in ["POST", "PUT", "PATCH"]:
            return False

        path = str(request.url.path)
        return "/gmn" in path and not path.endswith("/validate")

    async def _extract_gmn_specification(self, request: Request) -> Optional[str]:
        """Extract GMN specification from request."""
        try:
            body = await request.body()
            if not body:
                return None

            # This would need proper JSON parsing
            return body.decode("utf-8")
        except Exception as e:
            logger.error(f"Failed to extract GMN specification: {e}")
            return None


# Factory function for creating middleware
def create_gmn_validation_middleware(config: Optional[GMNValidationConfig] = None):
    """Create GMN validation middleware with configuration."""
    if config is None:
        config = GMNValidationConfig()

    def middleware_factory(app):
        return GMNValidationMiddleware(app, config)

    return middleware_factory
