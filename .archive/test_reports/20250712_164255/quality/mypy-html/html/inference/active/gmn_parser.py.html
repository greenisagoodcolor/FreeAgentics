"""GMN Parser for Generative Model Network specifications."""

import logging
from enum import Enum
from typing import Any, Dict

logger = logging.getLogger(__name__)


class GMNNodeType(Enum):
    """GMN Node Type enumeration."""

    STATE = "state"
    OBSERVATION = "observation"
    ACTION = "action"
    BELIEF = "belief"
    PREFERENCE = "preference"
    TRANSITION = "transition"
    LIKELIHOOD = "likelihood"


class GMNParser:
    """Parser for GMN specifications."""

    def __init__(self):
        """Initialize the GMN parser."""
        self.parsed_specs = {}

    def parse(self, specification: Dict[str, Any]) -> Dict[str, Any]:
        """Parse a GMN specification."""

        if not specification:
            raise ValueError("Empty specification provided")

        # Basic validation
        if "nodes" not in specification:
            raise KeyError("Specification must contain 'nodes' key")

        return {
            "nodes": specification.get("nodes", []),
            "edges": specification.get("edges", []),
            "metadata": specification.get("metadata", {}),
        }

    def validate_specification(self, spec: Dict[str, Any]) -> bool:
        """Validate a GMN specification."""
        try:
            self.parse(spec)
            return True
        except (ValueError, KeyError):
            return False
