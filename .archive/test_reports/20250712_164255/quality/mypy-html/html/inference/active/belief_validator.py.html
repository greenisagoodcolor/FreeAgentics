"""Belief validation for active inference agents."""

import logging
from collections import Counter
from typing import TYPE_CHECKING, Any, Dict, List, Optional

import numpy as np

if TYPE_CHECKING:
    from inference.gnn.validator import GMNGraph, GMNNode
    from inference.gnn.parser import GMNNodeType

from .belief_types import (
    BeliefConstraints,
    BeliefUpdateOperation,
    ConditionalRule,
    ValidationResult,
)

logger = logging.getLogger(__name__)


def _get_gmn_types():
    """Import GMN types dynamically to avoid circular imports."""
    try:
        from inference.gnn.parser import GMNNodeType
        from inference.gnn.validator import GMNGraph, GMNNode

        return GMNGraph, GMNNode, GMNNodeType
    except ImportError:
        return None, None, None


class BeliefValidator:
    """Comprehensive belief validation for active inference."""

    def __init__(self, constraints: Optional[BeliefConstraints] = None):
        """Initialize belief validator with constraints."""
        self.constraints = constraints or BeliefConstraints()
        self.validation_history: List[ValidationResult] = []
        self.error_patterns = Counter()
        self.feature_stats: Dict[str, Any] = {}

    def validate_belief_vector(self, belief: np.ndarray) -> ValidationResult:
        """Validate a single belief vector."""
        errors = []
        warnings = []
        metadata = {}

        # Basic structural validation
        if not isinstance(belief, np.ndarray):
            errors.append("Belief must be a numpy array")
            return ValidationResult(is_valid=False, errors=errors)

        if belief.ndim != 1:
            errors.append(f"Belief must be 1-dimensional, got {belief.ndim}D")
            return ValidationResult(is_valid=False, errors=errors)

        if belief.size == 0:
            errors.append("Belief vector cannot be empty")
            return ValidationResult(is_valid=False, errors=errors)

        # Numerical validity
        if not np.isfinite(belief).all():
            errors.append("Belief vector contains non-finite values (NaN/Inf)")

        # Positivity constraint
        if self.constraints.enforce_positivity and (belief < 0).any():
            errors.append("Belief vector contains negative values")

        # Normalization check
        belief_sum = np.sum(belief)
        if self.constraints.require_normalization:
            if abs(belief_sum - 1.0) > self.constraints.normalization_tolerance:
                errors.append(f"Belief vector not normalized: sum = {belief_sum}")

        # Numerical stability
        if (belief < self.constraints.numerical_stability_threshold).any():
            warnings.append(
                "Belief vector contains very small values that may cause numerical instability"
            )

        # Precision bounds
        if belief.size > 0:
            max_precision = 1.0 / np.min(belief[belief > 0]) if np.any(belief > 0) else 0
            if max_precision > self.constraints.max_precision:
                warnings.append(
                    f"Belief precision ({max_precision:.2e}) exceeds maximum ({self.constraints.max_precision:.2e})"
                )

        # Entropy bounds
        entropy = self._calculate_entropy(belief)
        metadata["entropy"] = entropy

        if entropy < self.constraints.min_entropy:
            warnings.append(
                f"Belief entropy ({entropy:.6f}) below minimum ({self.constraints.min_entropy})"
            )
        elif entropy > self.constraints.max_entropy:
            warnings.append(
                f"Belief entropy ({entropy:.6f}) exceeds maximum ({self.constraints.max_entropy})"
            )

        # Store validation metadata
        metadata.update(
            {
                "belief_sum": belief_sum,
                "belief_size": belief.size,
                "min_value": np.min(belief),
                "max_value": np.max(belief),
                "mean_value": np.mean(belief),
                "std_value": np.std(belief),
            }
        )

        is_valid = len(errors) == 0
        return ValidationResult(
            is_valid=is_valid, errors=errors, warnings=warnings, metadata=metadata
        )

    def validate_belief_update(
        self,
        original_belief: np.ndarray,
        updated_belief: np.ndarray,
        operation: BeliefUpdateOperation,
    ) -> ValidationResult:
        """Validate a belief update operation."""
        errors = []
        warnings = []
        metadata = {}

        # Validate original and updated beliefs
        original_result = self.validate_belief_vector(original_belief)
        updated_result = self.validate_belief_vector(updated_belief)

        errors.extend([f"Original belief: {e}" for e in original_result.errors])
        errors.extend([f"Updated belief: {e}" for e in updated_result.errors])
        warnings.extend([f"Original belief: {w}" for w in original_result.warnings])
        warnings.extend([f"Updated belief: {w}" for w in updated_result.warnings])

        # Dimensional consistency:
        if original_belief.shape != updated_belief.shape:
            errors.append(
                f"Belief dimensions changed: {original_belief.shape} -> {updated_belief.shape}"
            )

        # KL divergence check
        if original_belief.shape == updated_belief.shape:
            kl_div = self._calculate_kl_divergence(original_belief, updated_belief)
            metadata["kl_divergence"] = kl_div

            if kl_div > self.constraints.max_kl_divergence:
                warnings.append(
                    f"Large KL divergence ({kl_div:.6f}) indicates significant belief change"
                )

        # Belief change rate validation
        if original_belief.shape == updated_belief.shape:
            change_rate = np.sum(np.abs(updated_belief - original_belief)) / 2.0
            metadata["change_rate"] = change_rate

            if change_rate > self.constraints.max_belief_change_rate:
                warnings.append(
                    f"Belief change rate ({change_rate:.6f}) exceeds maximum ({self.constraints.max_belief_change_rate})"
                )

        # Operation-specific validation
        operation_errors = self._validate_operation_consistency(
            original_belief, updated_belief, operation
        )
        errors.extend(operation_errors)

        # Active inference principles
        ai_errors = self._validate_active_inference_principles(
            original_belief, updated_belief, operation
        )
        errors.extend(ai_errors)

        is_valid = len(errors) == 0
        return ValidationResult(
            is_valid=is_valid, errors=errors, warnings=warnings, metadata=metadata
        )

    def validate_conditional_rule(
        self, rule: ConditionalRule, context: Dict[str, Any]
    ) -> ValidationResult:
        """Validate a conditional rule for belief updates."""
        errors = []
        warnings = []
        metadata = {}

        # Validate condition:
        if not rule.condition:
            errors.append("Conditional rule has empty condition")
        else:
            condition_valid = self._validate_condition_syntax(rule.condition, context)
            if not condition_valid:
                errors.append(f"Invalid condition syntax: {rule.condition}")

        # Validate action operation
        if not rule.action:
            errors.append("Conditional rule has no action")
        else:
            action_errors = self._validate_belief_operation(rule.action)
            errors.extend(action_errors)

        # Priority validation
        if rule.priority < 0:
            warnings.append(f"Negative priority ({rule.priority}) may cause unexpected behavior")

        # Temporal constraints
        if rule.action.temporal_constraints:
            temporal_errors = self._validate_temporal_constraints(rule.action.temporal_constraints)
            errors.extend(temporal_errors)

        metadata.update(
            {
                "condition": rule.condition,
                "priority": rule.priority,
                "operation_type": rule.action.operation_type,
            }
        )

        is_valid = len(errors) == 0
        return ValidationResult(
            is_valid=is_valid, errors=errors, warnings=warnings, metadata=metadata
        )

    def validate_gmn_graph_beliefs(self, graph: "GMNGraph") -> ValidationResult:
        """Validate belief-related nodes and edges in GMN graph."""
        errors = []
        warnings = []
        metadata = {}

        # Get belief-related nodes
        belief_nodes = [n for n in graph.nodes if n.type == "belief"]
        belief_update_nodes = [n for n in graph.nodes if n.type == "belief_update"]

        metadata["num_belief_nodes"] = len(belief_nodes)
        metadata["num_belief_update_nodes"] = len(belief_update_nodes)

        # Validate belief nodes
        for node in belief_nodes:
            node_errors = self._validate_belief_node(node, graph)
            errors.extend([f"Belief node {node.id}: {e}" for e in node_errors])

        # Validate belief update nodes:
        for node in belief_update_nodes:
            node_errors = self._validate_belief_update_node(node, graph)
            errors.extend([f"Belief update node {node.id}: {e}" for e in node_errors])

        # Validate belief flow consistency
        flow_errors = self._validate_belief_flow(graph)
        errors.extend(flow_errors)

        # Check for circular dependencies in belief updates
        circular_errors = self._check_belief_circular_dependencies(graph)
        errors.extend(circular_errors)

        is_valid = len(errors) == 0
        return ValidationResult(
            is_valid=is_valid, errors=errors, warnings=warnings, metadata=metadata
        )

    def validate_belief_sequence(self, belief_sequence: List[np.ndarray]) -> ValidationResult:
        """Validate a sequence of belief states for consistency."""
        errors = []
        warnings = []
        metadata = {}

        if not belief_sequence:
            errors.append("Empty belief sequence")
            return ValidationResult(is_valid=False, errors=errors)

        # Validate each belief in sequence
        for i, belief in enumerate(belief_sequence):
            result = self.validate_belief_vector(belief)
            if not result.is_valid:
                errors.extend([f"Belief {i}: {e}" for e in result.errors])
            warnings.extend([f"Belief {i}: {w}" for w in result.warnings])

        # Check sequence consistency
        if len(belief_sequence) > 1:
            # Dimensional consistency
            base_shape = belief_sequence[0].shape
            for i, belief in enumerate(belief_sequence[1:], 1):
                if belief.shape != base_shape:
                    errors.append(
                        f"Belief {i} shape {belief.shape} doesn't match base shape {base_shape}"
                    )

            # Temporal consistency
            total_kl_divergence = 0.0
            max_change_rate = 0.0

            for i in range(len(belief_sequence) - 1):
                curr_belief = belief_sequence[i]
                next_belief = belief_sequence[i + 1]

                if curr_belief.shape == next_belief.shape:
                    kl_div = self._calculate_kl_divergence(curr_belief, next_belief)
                    total_kl_divergence += kl_div

                    change_rate = np.sum(np.abs(next_belief - curr_belief)) / 2.0
                    max_change_rate = max(max_change_rate, change_rate)

            metadata.update(
                {
                    "sequence_length": len(belief_sequence),
                    "total_kl_divergence": total_kl_divergence,
                    "max_change_rate": max_change_rate,
                    "avg_kl_divergence": total_kl_divergence / max(1, len(belief_sequence) - 1),
                }
            )

            if total_kl_divergence > self.constraints.max_kl_divergence * len(belief_sequence):
                warnings.append(
                    f"High cumulative KL divergence ({total_kl_divergence:.6f}) in sequence"
                )

        is_valid = len(errors) == 0
        return ValidationResult(
            is_valid=is_valid, errors=errors, warnings=warnings, metadata=metadata
        )

    def _calculate_entropy(self, belief: np.ndarray) -> float:
        """Calculate entropy of belief distribution."""
        belief_normalized = belief / np.sum(belief)
        belief_clipped = np.clip(belief_normalized, 1e-15, 1.0)
        return -np.sum(belief_clipped * np.log(belief_clipped))

    def _calculate_kl_divergence(self, p: np.ndarray, q: np.ndarray) -> float:
        """Calculate KL divergence between two distributions."""
        p_normalized = p / np.sum(p)
        q_normalized = q / np.sum(q)
        p_clipped = np.clip(p_normalized, 1e-15, 1.0)
        q_clipped = np.clip(q_normalized, 1e-15, 1.0)
        return np.sum(p_clipped * np.log(p_clipped / q_clipped))

    def _validate_operation_consistency(
        self,
        original: np.ndarray,
        updated: np.ndarray,
        operation: BeliefUpdateOperation,
    ) -> List[str]:
        """Validate operation consistency."""
        errors = []

        if operation.operation_type == "set":
            # Check if values match parameters
            expected_values = operation.parameters.get("values", [])
            if expected_values and len(expected_values) == updated.size:
                expected_normalized = np.array(expected_values) / np.sum(expected_values)
                if not np.allclose(updated, expected_normalized, rtol=1e-6):
                    errors.append("Set operation result doesn't match expected values")

        elif operation.operation_type == "normalize":
            # Check if normalization was applied correctly
            if original.size == updated.size:
                expected_normalized = original / np.sum(original)
                if not np.allclose(updated, expected_normalized, rtol=1e-6):
                    errors.append("Normalize operation result incorrect")

        elif operation.operation_type == "add":
            # Check additive consistency
            add_values = operation.parameters.get("values", [])
            if add_values and len(add_values) == original.size:
                expected_result = original + np.array(add_values)
                expected_result = np.maximum(expected_result, 0)
                expected_normalized = expected_result / np.sum(expected_result)
                if not np.allclose(updated, expected_normalized, rtol=1e-6):
                    errors.append("Add operation result doesn't match expected calculation")

        return errors

    def _validate_active_inference_principles(
        self,
        original: np.ndarray,
        updated: np.ndarray,
        operation: BeliefUpdateOperation,
    ) -> List[str]:
        """Validate against active inference principles."""
        errors = []

        if operation.operation_type == "revise":
            # Check that revision moves toward evidence
            evidence = operation.parameters.get("evidence", [])
            if evidence and len(evidence) == original.size:
                evidence_array = np.array(evidence)
                # Posterior should be closer to evidence than prior
                prior_evidence_kl = self._calculate_kl_divergence(original, evidence_array)
                posterior_evidence_kl = self._calculate_kl_divergence(updated, evidence_array)

                if posterior_evidence_kl > prior_evidence_kl:
                    errors.append("Belief revision increases divergence from evidence")

        # Precision-weighted update principle
        if "precision" in operation.parameters:
            precision = operation.parameters["precision"]
            if not isinstance(precision, (int, float)) or precision <= 0:
                errors.append("Precision must be a positive number")

        # Temporal consistency for active inference:
        if operation.temporal_constraints:
            time_step = operation.temporal_constraints.get("time_step", 1)
            if time_step <= 0:
                errors.append("Time step must be positive")

        return errors

    def _validate_condition_syntax(self, condition: str, context: Dict[str, Any]) -> bool:
        """Validate condition syntax."""
        try:
            # Check for dangerous patterns
            dangerous_patterns = ["import", "exec", "eval", "__", "open", "file"]
            for pattern in dangerous_patterns:
                if pattern in condition.lower():
                    return False

            # Try to parse as expression
            import ast

            parsed = ast.parse(condition, mode="eval")

            # Check that only safe operations are used
            for node in ast.walk(parsed):
                if isinstance(node, (ast.Call, ast.Import, ast.ImportFrom)):
                    return False

            return True
        except (SyntaxError, ValueError):
            return False

    def _validate_belief_operation(self, operation: BeliefUpdateOperation) -> List[str]:
        """Validate belief update operation."""
        errors = []

        if not operation.target_belief:
            errors.append("Belief operation must have target belief")

        valid_operations = ["set", "add", "multiply", "normalize", "revise"]
        if operation.operation_type not in valid_operations:
            errors.append(f"Invalid operation type: {operation.operation_type}")

        # Validate parameters based on operation type
        if operation.operation_type in ["set", "add"]:
            values = operation.parameters.get("values", [])
            if not values:
                errors.append(f"Operation {operation.operation_type} requires 'values' parameter")
            elif not isinstance(values, list):
                errors.append("Values parameter must be a list")

        elif operation.operation_type == "multiply":
            factors = operation.parameters.get("factors", [])
            if not factors:
                errors.append("Multiply operation requires 'factors' parameter")
            elif not isinstance(factors, list):
                errors.append("Factors parameter must be a list")

        elif operation.operation_type == "revise":
            evidence = operation.parameters.get("evidence", [])
            if not evidence:
                errors.append("Revise operation requires 'evidence' parameter")
            elif not isinstance(evidence, list):
                errors.append("Evidence parameter must be a list")

        return errors

    def _validate_temporal_constraints(self, constraints: Dict[str, Any]) -> List[str]:
        """Validate temporal constraints."""
        errors = []

        if "time_step" in constraints:
            time_step = constraints["time_step"]
            if not isinstance(time_step, (int, float)) or time_step <= 0:
                errors.append("Time step must be a positive number")

        if "duration" in constraints:
            duration = constraints["duration"]
            if not isinstance(duration, (int, float)) or duration <= 0:
                errors.append("Duration must be a positive number")

        if "max_updates" in constraints:
            max_updates = constraints["max_updates"]
            if not isinstance(max_updates, int) or max_updates <= 0:
                errors.append("Max updates must be a positive integer")

        return errors

    def _validate_belief_node(self, node: "GMNNode", graph: "GMNGraph") -> List[str]:
        """Validate a belief node in the graph."""
        errors = []
        GMNGraph, GMNNode, GMNNodeType = _get_gmn_types()

        # Check belief vector property
        if "vector" in node.properties:
            vector = node.properties["vector"]
            if not isinstance(vector, list):
                errors.append("Belief vector must be a list")
            elif not vector:
                errors.append("Belief vector cannot be empty")
            else:
                # Validate the vector
                belief_array = np.array(vector)
                result = self.validate_belief_vector(belief_array)
                errors.extend(result.errors)

        # Check connections to state nodes
        connected_states = []
        for edge in graph.edges:
            if edge.source == node.id and edge.type.value == "depends_on":
                target_node = graph.nodes.get(edge.target)
                if target_node and target_node.type == GMNNodeType.STATE:
                    connected_states.append(target_node.id)

        if not connected_states:
            errors.append("Belief node must be connected to at least one state node")

        return errors

    def _validate_belief_update_node(self, node: "GMNNode", graph: "GMNGraph") -> List[str]:
        """Validate a belief update node in the graph."""
        errors = []

        # Check required properties
        if "target_belief" not in node.properties:
            errors.append("Belief update node must specify target_belief")

        if "operation_type" not in node.properties:
            errors.append("Belief update node must specify operation_type")
        else:
            operation_type = node.properties["operation_type"]
            valid_operations = ["set", "add", "multiply", "normalize", "revise"]
            if operation_type not in valid_operations:
                errors.append(f"Invalid operation type: {operation_type}")

        # Check for proper parameters
        operation = BeliefUpdateOperation(
            target_belief=node.properties.get("target_belief", ""),
            operation_type=node.properties.get("operation_type", "set"),
            parameters=node.properties.get("parameters", {}),
            conditions=node.properties.get("conditions"),
            temporal_constraints=node.properties.get("temporal_constraints"),
        )

        operation_errors = self._validate_belief_operation(operation)
        errors.extend(operation_errors)

        return errors

    def _validate_belief_flow(self, graph: "GMNGraph") -> List[str]:
        """Validate belief flow consistency in the graph."""
        errors = []

        # Collect all belief nodes
        belief_nodes = {node.id for node in graph.nodes.values() if node.type == GMNNodeType.BELIEF}

        # Check belief updates target existing beliefs
        for update_node in graph.nodes.values():
            if update_node.type == GMNNodeType.BELIEF_UPDATE:
                target_belief = update_node.properties.get("target_belief", "")
                if target_belief and target_belief not in belief_nodes:
                    errors.append(
                        f"Belief update {update_node.id} targets non-existent belief {target_belief}"
                    )

        # Check belief update chains
        for node in graph.nodes.values():
            if node.type == GMNNodeType.BELIEF_UPDATE:
                # Check if this update is triggered by appropriate conditions
                trigger_count = 0
                for edge in graph.edges:
                    if edge.target == node.id and edge.type.value == "triggers":
                        trigger_count += 1

                if trigger_count == 0:
                    errors.append(f"Belief update {node.id} has no triggers")

        return errors

    def _check_belief_circular_dependencies(self, graph: GMNGraph) -> List[str]:
        """Check for circular dependencies in belief updates."""
        errors = []
        dependencies = {}

        # Build dependency graph:
        for node in graph.nodes.values():
            if node.type == GMNNodeType.BELIEF_UPDATE:
                target = node.properties.get("target_belief", "")
                if target:
                    dependencies[node.id] = [target]

        # Check for cycles:
        def has_cycle(node_id: str, visited: set, rec_stack: set) -> bool:
            visited.add(node_id)
            rec_stack.add(node_id)

            for neighbor in dependencies.get(node_id, []):
                if neighbor not in visited:
                    if has_cycle(neighbor, visited, rec_stack):
                        return True
                elif neighbor in rec_stack:
                    return True

            rec_stack.remove(node_id)
            return False

        visited = set()
        rec_stack = set()

        for node_id in dependencies:
            if node_id not in visited:
                if has_cycle(node_id, visited, rec_stack):
                    errors.append(
                        f"Circular dependency detected in belief updates involving {node_id}"
                    )

        return errors

    def get_validation_statistics(self) -> Dict[str, Any]:
        """Get validation statistics."""
        return {
            "total_validations": len(self.validation_history),

            "constraints": {
                "min_precision": self.constraints.min_precision,
                "max_precision": self.constraints.max_precision,

                "max_entropy": self.constraints.max_entropy,
                "normalization_tolerance": self.constraints.normalization_tolerance,

        }
