"""Belief types and data structures for active inference."""

from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional


@dataclass
class ValidationResult:
    """Result of belief validation."""

    is_valid: bool
    errors: List[str] = field(default_factory=list)
    warnings: List[str] = field(default_factory=list)
    metadata: Dict[str, Any] = field(default_factory=dict)


@dataclass
class BeliefConstraints:
    """Constraints for belief validation."""

    min_precision: float = 1e-10
    max_precision: float = 1e10
    min_entropy: float = 0.0
    max_entropy: float = 10.0
    normalization_tolerance: float = 1e-6
    numerical_stability_threshold: float = 1e-15
    max_kl_divergence: float = 10.0
    max_belief_change_rate: float = 1.0
    enforce_positivity: bool = True
    require_normalization: bool = True


@dataclass
class BeliefUpdateOperation:
    """Represents a dynamic belief update operation."""

    target_belief: str
    operation_type: str  # "set", "add", "multiply", "normalize", "revise"
    parameters: Dict[str, Any] = field(default_factory=dict)
    conditions: Optional[List[str]] = None
    temporal_constraints: Optional[Dict[str, Any]] = None


@dataclass
class ConditionalRule:
    """Represents a conditional rule for belief updates."""

    condition: str
    action: BeliefUpdateOperation
    priority: int = 0
    enabled: bool = True
