#!/usr/bin/env python3
"""H3 spatial integration for GNN models.

This module provides H3-based hierarchical spatial analysis and spatially-aware Active Inference.
"""

import logging
from dataclasses import dataclass
from typing import Any, Dict, List, Optional, Tuple

import numpy as np

try:
    import h3

    H3_AVAILABLE = True
except ImportError:
    logger = logging.getLogger(__name__)
    logger.warning("H3 library not available - spatial features will use fallback")
    H3_AVAILABLE = False

try:
    import torch

    TORCH_AVAILABLE = True
except ImportError:
    logger = logging.getLogger(__name__)
    logger.warning("PyTorch not available - using numpy fallbacks for tensors")
    TORCH_AVAILABLE = False


@dataclass
class SpatialConfig:
    """Configuration for spatial processing."""

    default_resolution: int = 7
    max_cache_size: int = 10000
    enable_caching: bool = True


class H3SpatialProcessor:
    """H3 spatial processing for hierarchical spatial indexing."""

    def __init__(self, config: Optional[SpatialConfig] = None):
        """Initialize H3 spatial processor."""
        self.config = config or SpatialConfig()
        self.h3_cache: Dict[str, str] = {}
        self.logger = logging.getLogger(__name__)

    def lat_lng_to_h3(
        self, lat: float, lon: float, resolution: Optional[int] = None
    ) -> Optional[str]:
        """Convert lat/lng coordinates to H3 index."""
        if not H3_AVAILABLE:
            return None

        resolution = resolution or self.config.default_resolution
        cache_key = f"{lat:.6f}_{lon:.6f}_{resolution}"

        if self.config.enable_caching and cache_key in self.h3_cache:
            return self.h3_cache[cache_key]

        try:
            h3_index = h3.latlng_to_cell(lat, lon, resolution)
            if self.config.enable_caching:
                self.h3_cache[cache_key] = h3_index
            return h3_index
        except Exception as e:
            self.logger.warning(f"H3 conversion failed for {lat}, {lon}: {e}")
            return None

    def h3_to_latlng(self, h3_index: str) -> Optional[Tuple[float, float]]:
        """Convert H3 index to lat/lng coordinates."""
        if not H3_AVAILABLE:
            return None

        try:
            return h3.cell_to_latlng(h3_index)
        except Exception as e:
            self.logger.warning(f"H3 to lat/lng conversion failed for {h3_index}: {e}")
            return None

    def get_h3_neighbors(self, h3_index: str, k: int = 1) -> List[str]:
        """Get H3 neighbors within k distance."""
        if not H3_AVAILABLE:
            return []

        try:
            return h3.grid_ring(h3_index, k)
        except Exception as e:
            self.logger.warning(f"H3 neighbors failed for {h3_index}: {e}")
            return []

    def get_h3_distance(self, h3_index1: str, h3_index2: str) -> Optional[int]:
        """Get distance between two H3 indices."""
        if not H3_AVAILABLE:
            return None

        try:
            return h3.grid_distance(h3_index1, h3_index2)
        except Exception as e:
            self.logger.warning(f"H3 distance failed for {h3_index1}, {h3_index2}: {e}")
            return None

    def adaptive_resolution(self, agent_density: float, observation_scale: float) -> int:
        """Compute adaptive H3 resolution based on agent density and observation scale."""
        # Base resolution 7, adjust based on density and scale
        base_resolution = 7

        # Higher density requires higher resolution
        density_adjustment = int(np.log10(max(agent_density, 1.0)))

        # Smaller observation scale requires higher resolution
        scale_adjustment = -int(np.log10(max(observation_scale, 0.1)))

        resolution = max(0, min(15, base_resolution + density_adjustment + scale_adjustment))
        return resolution

    def create_spatial_graph(self, h3_indices: List[str]) -> Dict[str, List[str]]:
        """Create spatial graph from H3 indices."""
        if not H3_AVAILABLE:
            return {}

        graph = {}
        for h3_index in h3_indices:
            if h3_index:
                neighbors = self.get_h3_neighbors(h3_index, k=1)
                graph[h3_index] = [n for n in neighbors if n in h3_indices]

        return graph


class H3MultiResolutionAnalyzer:
    """H3 multi-resolution spatial analysis."""

    def __init__(self, resolutions: Optional[List[int]] = None):
        """Initialize multi-resolution analyzer."""
        self.resolutions = resolutions or [5, 7, 9]
        self.processor = H3SpatialProcessor()

    def extract_multi_resolution_features(
        self, positions: List[Tuple[float, float]]
    ) -> Dict[str, Any]:
        """Extract features at multiple H3 resolutions."""
        try:
            import torch
        except ImportError:
            return {"fallback_features": [[lat, lon] for lat, lon in positions]}

        features = {}

        for resolution in self.resolutions:
            res_features = []
            h3_indices = []

            for lat, lon in positions:
                h3_index = self.processor.latlng_to_h3(lat, lon, resolution)
                h3_indices.append(h3_index)

                if h3_index:
                    # H3 index as numerical feature (hash to fixed-size vector)
                    h3_hash = hash(h3_index) % 1000000  # Normalize to reasonable range
                    center_lat, center_lon = self.processor.h3_to_latlng(h3_index)
                    res_features.append([center_lat, center_lon, h3_hash / 1000000.0])
                else:
                    res_features.append([lat, lon, 0.0])

            features[f"resolution_{resolution}"] = torch.tensor(res_features, dtype=torch.float32)
            features[f"h3_indices_{resolution}"] = h3_indices

        return features

    def compute_spatial_relationships(self, h3_indices: List[str]) -> Dict[str, Any]:
        """Compute spatial relationships between H3 indices."""
        relationships = {
            "adjacency_matrix": [],
            "distance_matrix": [],
            "neighbor_counts": [],
            "cluster_info": {},
        }

        valid_indices = [idx for idx in h3_indices if idx is not None]
        n = len(valid_indices)

        if n == 0:
            return relationships

        # Compute distance matrix
        distance_matrix = np.zeros((n, n))
        adjacency_matrix = np.zeros((n, n))

        for i, h3_i in enumerate(valid_indices):
            neighbors = self.processor.get_h3_neighbors(h3_i, k=1)
            neighbor_count = len(neighbors)
            relationships["neighbor_counts"].append(neighbor_count)

            for j, h3_j in enumerate(valid_indices):
                if i != j:
                    distance = self.processor.get_h3_distance(h3_i, h3_j)
                    if distance is not None:
                        distance_matrix[i, j] = distance
                        if distance == 1:  # Adjacent cells:
                            adjacency_matrix[i, j] = 1

        relationships["adjacency_matrix"] = adjacency_matrix
        relationships["distance_matrix"] = distance_matrix

        # Identify spatial clusters
        relationships["cluster_info"] = self._identify_h3_clusters(valid_indices)

        return relationships

    def _identify_h3_clusters(self, h3_indices: List[str]) -> Dict[str, Any]:
        """Identify spatial clusters in H3 indices."""
        clusters = []  # Simple clustering placeholder
        return {
            "num_clusters": len(clusters),
            "clusters": clusters,
            "largest_cluster_size": max(len(c) for c in clusters) if clusters else 0,
            "average_cluster_size": (
                sum(len(c) for c in clusters) / len(clusters) if clusters else 0
            ),
        }


class GNNSpatialIntegration:
    """GNN integration with H3 spatial features."""

    def __init__(self):
        """Initialize GNN spatial integration."""
        self.h3_processor = H3SpatialProcessor()
        self.multi_res_analyzer = H3MultiResolutionAnalyzer()

    def extract_spatial_features(self, nodes: List[Dict[str, Any]]) -> Dict[str, Any]:
        """Extract spatial features from nodes."""
        try:
            import torch
        except ImportError:
            return {"error": "PyTorch not available"}

        positions = []
        for node in nodes:
            if "position" in node:
                pos = node["position"]
                if isinstance(pos, dict) and "lat" in pos and "lon" in pos:
                    positions.append((pos["lat"], pos["lon"]))
                elif isinstance(pos, (list, tuple)) and len(pos) >= 2:
                    # Assume list coordinates are lat/lon
                    positions.append((pos[0], pos[1]))
                else:
                    positions.append((0.0, 0.0))
            else:
                positions.append((0.0, 0.0))

        if not positions:
            return {"empty_features": torch.zeros(0, 2)}

        # Multi-resolution H3 features
        multi_res_features = self.multi_res_analyzer.extract_multi_resolution_features(positions)

        # Spatial graph construction
        if H3_AVAILABLE and "h3_indices_7" in multi_res_features:
            h3_indices = multi_res_features["h3_indices_7"]
            edge_index, edge_weights = self.h3_processor.create_h3_spatial_graph(h3_indices)
            multi_res_features["spatial_edge_index"] = edge_index
            multi_res_features["spatial_edge_weights"] = edge_weights

        # Spatial relationships
        if H3_AVAILABLE and "h3_indices_7" in multi_res_features:
            relationships = self.multi_res_analyzer.compute_spatial_relationships(
                multi_res_features["h3_indices_7"]
            )
            multi_res_features.update(relationships)

        return multi_res_features

    def adaptive_spatial_resolution(self, agents: List[Any], observation_scale: float) -> int:
        """Compute adaptive spatial resolution."""
        # Simple adaptive resolution based on agent count
        agent_count = len(agents)
        if agent_count > 100:
            return 9  # High resolution for dense areas
        elif agent_count > 20:
            return 7  # Medium resolution
        else:
            return 5  # Low resolution for sparse areas

    def integrate_with_pymdp(self, agent: Any, spatial_features: Dict[str, Any]) -> Dict[str, Any]:
        """Integrate spatial features with PyMDP active inference."""
        if not hasattr(agent, "pymdp_agent") or agent.pymdp_agent is None:
            return {}

        integration_results = {}

        # Use H3 spatial structure to inform PyMDP state space
        if (
            "spatial_edge_index" in spatial_features
            and spatial_features["spatial_edge_index"].numel() > 0
        ):
            edge_index = spatial_features["spatial_edge_index"]
            if TORCH_AVAILABLE:
                edge_weights = spatial_features.get(
                    "spatial_edge_weights", torch.ones(edge_index.size(1))
                )

                # Create spatial adjacency matrix for PyMDP
                num_nodes = max(edge_index.max().item() + 1, 1)
                spatial_adjacency = torch.zeros(num_nodes, num_nodes)

                for i, (src, dst) in enumerate(edge_index.t()):
                    spatial_adjacency[src, dst] = edge_weights[i]

                integration_results["spatial_adjacency"] = spatial_adjacency
                integration_results["spatial_connectivity"] = torch.sum(
                    spatial_adjacency > 0
                ).item()
            else:
                # Fallback to numpy if torch not available
                edge_weights = spatial_features.get(
                    "spatial_edge_weights", np.ones(edge_index.shape[1])
                )
                num_nodes = max(edge_index.max() + 1, 1)
                spatial_adjacency = np.zeros((num_nodes, num_nodes))

                for i, (src, dst) in enumerate(edge_index.T):
                    spatial_adjacency[src, dst] = edge_weights[i]

                integration_results["spatial_adjacency"] = spatial_adjacency
                integration_results["spatial_connectivity"] = np.sum(spatial_adjacency > 0).item()

        # Multi-resolution spatial context for belief priors
        if "resolution_7" in spatial_features:
            spatial_context = spatial_features["resolution_7"]
            if spatial_context.numel() > 0:
                # Compute spatial variance as uncertainty measure
                spatial_variance = torch.var(spatial_context, dim=0)
                integration_results["spatial_uncertainty"] = spatial_variance.mean().item()

        return integration_results
