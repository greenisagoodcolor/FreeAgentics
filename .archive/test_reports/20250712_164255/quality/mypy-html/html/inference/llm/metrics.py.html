"""Metrics collection for LLM providers with Prometheus integration."""

import logging
import time
from dataclasses import dataclass
from enum import Enum
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)

try:
    PROMETHEUS_AVAILABLE = True
except ImportError:
    PROMETHEUS_AVAILABLE = False
    logger.warning("Prometheus client not available - metrics collection will be limited")


class MetricType(Enum):
    """Types of metrics for LLM providers."""

    COUNTER = "counter"
    HISTOGRAM = "histogram"
    GAUGE = "gauge"
    INFO = "info"


@dataclass
class LLMMetricDefinition:
    """Definition of an LLM metric."""

    name: str
    metric_type: MetricType
    description: str
    labels: Optional[List[str]] = None


class LLMMetricsCollector:
    """Collector for LLM provider metrics with Prometheus integration."""

    def __init__(self) -> None:
        """Initialize metrics collector."""
        self.metrics: Dict[str, Dict[str, Any]] = {}
        self._init_metrics()

    def _init_metrics(self) -> None:
        """Initialize all LLM-related metrics."""
        self.metric_definitions = [
            LLMMetricDefinition(
                name="llm_requests_total",
                description="Total number of LLM requests by provider and status",
                metric_type=MetricType.COUNTER,
                labels=["provider", "status", "model"],
            ),
            LLMMetricDefinition(
                name="llm_request_duration_seconds",
                description="Time taken for LLM requests",
                metric_type=MetricType.HISTOGRAM,
                labels=["provider", "model"],
            ),
            LLMMetricDefinition(
                name="llm_tokens_total",
                description="Total tokens processed by LLM providers",
                metric_type=MetricType.COUNTER,
                labels=["provider", "model", "type"],  # type can be: input/output
            ),
            LLMMetricDefinition(
                name="llm_cost_total",
                description="Total cost of LLM requests",
                metric_type=MetricType.COUNTER,
                labels=["provider", "model"],
            ),
            # Error metrics
            LLMMetricDefinition(
                name="llm_errors_total",
                description="Total number of LLM errors by provider and type",
                metric_type=MetricType.COUNTER,
                labels=["provider", "error_type"],
            ),
            LLMMetricDefinition(
                name="llm_circuit_breaker_state",
                description="Current state of circuit breaker for LLM providers",
                metric_type=MetricType.GAUGE,
                labels=["provider"],
            ),
            LLMMetricDefinition(
                name="llm_circuit_breaker_failures",
                description="Number of failures recorded by circuit breaker",
                metric_type=MetricType.GAUGE,
                labels=["provider"],
            ),
            # Health metrics
            LLMMetricDefinition(
                name="llm_provider_health_status",
                description="Health status of LLM providers",
                metric_type=MetricType.GAUGE,
                labels=["provider"],
            ),
            LLMMetricDefinition(
                name="llm_provider_last_success_timestamp",
                description="Timestamp of last successful request",
                metric_type=MetricType.GAUGE,
                labels=["provider"],
            ),
            # Retry metrics
            LLMMetricDefinition(
                name="llm_retries_total",
                description="Total number of retry attempts",
                metric_type=MetricType.COUNTER,
                labels=["provider", "operation"],
            ),
            LLMMetricDefinition(
                name="llm_fallback_total",
                description="Total number of fallback attempts",
                metric_type=MetricType.COUNTER,
                labels=["from_provider", "to_provider"],
            ),
        ]

        # Create Prometheus metrics if available
        if PROMETHEUS_AVAILABLE:
            self.prometheus_metrics: Dict[str, Any] = {}
            for metric_def in self.metric_definitions:
                self._create_prometheus_metric(metric_def)

        # Initialize counters for basic metrics
        for metric_def in self.metric_definitions:
            self.metrics[metric_def.name] = {}

    def _create_prometheus_metric(self, metric_def: LLMMetricDefinition) -> None:
        """Create a Prometheus metric based on definition."""
        try:
            # Implementation would create actual Prometheus metrics here
            pass
        except Exception as e:
            logger.error(f"Failed to create Prometheus metric {metric_def.name}: {e}")

    def record_request(
        self,
        provider: str,
        model: str,
        status: str,
        duration_seconds: float,
        input_tokens: int = 0,
        output_tokens: int = 0,
        cost: float = 0.0,
    ) -> None:
        """Record a request metric."""
        try:
            if PROMETHEUS_AVAILABLE:
                self.prometheus_metrics["llm_requests_total"].labels(
                    provider=provider, status=status, model=model
                ).inc()

                # Request duration
                self.prometheus_metrics["llm_request_duration_seconds"].labels(
                    provider=provider, model=model
                ).observe(duration_seconds)

                # Tokens
                if input_tokens > 0:
                    self.prometheus_metrics["llm_tokens_total"].labels(
                        provider=provider, model=model, type="input"
                    ).inc(input_tokens)

                if output_tokens > 0:
                    self.prometheus_metrics["llm_tokens_total"].labels(
                        provider=provider, model=model, type="output"
                    ).inc(output_tokens)

                # Cost
                if cost > 0:
                    self.prometheus_metrics["llm_cost_total"].labels(
                        provider=provider, model=model
                    ).inc(cost)

            # Basic metrics
            metric_key = f"{provider}_{model}_{status}"
            self.metrics["llm_requests_total"][metric_key] = (
                self.metrics["llm_requests_total"].get(metric_key, 0) + 1
            )

        except Exception as e:
            logger.error(f"Failed to record request metric: {e}")

    def record_error(self, provider: str, error_type: str) -> None:
        """Record an error metric."""
        try:
            if PROMETHEUS_AVAILABLE:
                self.prometheus_metrics["llm_errors_total"].labels(
                    provider=provider, error_type=error_type
                ).inc()

            # Basic metrics
            metric_key = f"{provider}_{error_type}"
            self.metrics["llm_errors_total"][metric_key] = (
                self.metrics["llm_errors_total"].get(metric_key, 0) + 1
            )

        except Exception as e:
            logger.error(f"Failed to record error metric: {e}")

    def record_circuit_breaker_state(self, provider: str, state: str, failure_count: int) -> None:
        """Record circuit breaker state."""
        try:
            if PROMETHEUS_AVAILABLE:
                state_values = {"closed": 0, "open": 1, "half_open": 2}
                state_value = state_values.get(state, -1)

                self.prometheus_metrics["llm_circuit_breaker_state"].labels(provider=provider).set(
                    state_value
                )

                self.prometheus_metrics["llm_circuit_breaker_failures"].labels(
                    provider=provider
                ).set(failure_count)

            # Basic metrics
            self.metrics["llm_circuit_breaker_state"][provider] = state
            self.metrics["llm_circuit_breaker_failures"][provider] = failure_count

        except Exception as e:
            logger.error(f"Failed to record circuit breaker metric: {e}")

    def record_health_status(self, provider: str, is_healthy: bool) -> None:
        """Record provider health status."""
        try:
            if PROMETHEUS_AVAILABLE:
                self.prometheus_metrics["llm_provider_health_status"].labels(provider=provider).set(
                    1 if is_healthy else 0
                )

                if is_healthy:
                    self.prometheus_metrics["llm_provider_last_success_timestamp"].labels(
                        provider=provider
                    ).set(time.time())

            # Basic metrics
            self.metrics["llm_provider_health_status"][provider] = is_healthy

        except Exception as e:
            logger.error(f"Failed to record health metric: {e}")

    def record_retry(self, provider: str, operation: str) -> None:
        """Record a retry attempt."""
        try:
            if PROMETHEUS_AVAILABLE:
                self.prometheus_metrics["llm_retries_total"].labels(
                    provider=provider, operation=operation
                ).inc()

            # Basic metrics
            metric_key = f"{provider}_{operation}"
            self.metrics["llm_retries_total"][metric_key] = (
                self.metrics["llm_retries_total"].get(metric_key, 0) + 1
            )

        except Exception as e:
            logger.error(f"Failed to record retry metric: {e}")

    def record_fallback(self, from_provider: str, to_provider: str) -> None:
        """Record a fallback attempt."""
        try:
            if PROMETHEUS_AVAILABLE:
                self.prometheus_metrics["llm_fallback_total"].labels(
                    from_provider=from_provider, to_provider=to_provider
                ).inc()

            # Basic metrics
            metric_key = f"{from_provider}_{to_provider}"
            self.metrics["llm_fallback_total"][metric_key] = (
                self.metrics["llm_fallback_total"].get(metric_key, 0) + 1
            )

        except Exception as e:
            logger.error(f"Failed to record fallback metric: {e}")

    def get_metrics_summary(self) -> Dict[str, Any]:
        """Get a summary of all metrics."""
        return {
            "basic_metrics": self.metrics,
            "prometheus_available": PROMETHEUS_AVAILABLE,
            "prometheus_metrics": (
                list(self.prometheus_metrics.keys()) if PROMETHEUS_AVAILABLE else []
            ),
        }


# Global metrics collector instance
llm_metrics_collector = LLMMetricsCollector()
