"""Coalition formation strategies for multi-agent systems."""

import logging
import time
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime
from typing import Any, Dict, List, Optional, Set, Tuple

from coalitions.coalition import Coalition, CoalitionObjective, CoalitionRole

logger = logging.getLogger(__name__)


@dataclass
class AgentProfile:
    """Profile of an agent for coalition formation."""

    agent_id: str
    capabilities: List[str]
    capacity: float = 1.0
    reputation: float = 1.0
    preferences: Dict[str, float] = field(default_factory=dict)
    current_coalitions: List[str] = field(default_factory=list)
    max_coalitions: int = 3
    last_activity: datetime = field(default_factory=datetime.now)


@dataclass
class FormationResult:
    """Result of a coalition formation process."""

    coalitions: List[Coalition]
    unassigned_objectives: List[CoalitionObjective]
    formation_quality: float
    objective_coverage: float
    agent_utilization: float


class FormationStrategy(ABC):
    """Abstract base class for coalition formation strategies."""

    def __init__(self, name: str):
        """Initialize the strategy with a name."""
        self.name = name

    @abstractmethod
    def form_coalitions(
        self,
        agents: List[AgentProfile],
        objectives: List[CoalitionObjective],
        constraints: Optional[Dict[str, Any]] = None,
    ) -> FormationResult:
        """Form coalitions from available agents and objectives."""

    def _calculate_capability_match(
        self, agent_capabilities: Set[str], required_capabilities: Set[str]
    ) -> float:
        """Calculate how well agent capabilities match requirements."""
        if not required_capabilities:
            return 1.0

        matched = agent_capabilities.intersection(required_capabilities)
        return len(matched) / len(required_capabilities)

    def _calculate_formation_quality(
        self, coalitions: List[Coalition], objectives: List[CoalitionObjective]
    ) -> float:
        """Calculate overall quality of the formation."""
        if not coalitions:
            return 0.0

        total_score = 0.0
        for coalition in coalitions:
            # Score based on member count and objective count
            member_score = min(len(coalition.members) / 5.0, 1.0)  # Optimal size around 5
            objective_score = len(coalition.objectives) / max(len(objectives), 1)
            coalition_score = (member_score + objective_score) / 2
            total_score += coalition_score

        return total_score / len(coalitions)


class GreedyFormation(FormationStrategy):
    """Greedy coalition formation strategy."""

    def __init__(self):
        """Initialize the instance."""
        super().__init__("Greedy Formation")

    def form_coalitions(
        self,
        agents: List[AgentProfile],
        objectives: List[CoalitionObjective],
        constraints: Optional[Dict[str, Any]] = None,
    ) -> FormationResult:
        """Form coalitions using greedy algorithm."""
        constraints = constraints or {}
        max_coalition_size = constraints.get("max_coalition_size", 10)

        coalitions = []
        available_agents = agents.copy()
        assigned_agents = set()

        # Process objectives by priority
        sorted_objectives = sorted(objectives, key=lambda x: x.priority, reverse=True)

        for obj_idx, objective in enumerate(sorted_objectives):
            coalition = self._create_coalition_for_objective(
                obj_idx, objective, max_coalition_size, available_agents, assigned_agents
            )

            if coalition and coalition.members:
                coalition.activate()
                coalitions.append(coalition)
                logger.info(
                    f"Formed coalition {coalition.coalition_id} with {len(coalition.members)} members"
                )

        # Calculate metrics and return result
        return self._create_greedy_formation_result(coalitions, objectives, agents, assigned_agents)

    def _create_coalition_for_objective(
        self,
        obj_idx: int,
        objective: CoalitionObjective,
        max_coalition_size: int,
        available_agents: List[AgentProfile],
        assigned_agents: Set[str],
    ) -> Optional[Coalition]:
        """Create a coalition for a single objective."""
        coalition_id = f"coalition_{obj_idx + 1}"
        coalition = Coalition(
            coalition_id=coalition_id,
            name=f"Coalition for {objective.description}",
            objective=objective.description,
            max_capacity=max_coalition_size,
        )

        coalition.add_objective(objective)

        # Get and score candidate agents
        candidate_agents = [a for a in available_agents if a.agent_id not in assigned_agents]
        agent_scores = self._score_agents_for_objective(candidate_agents, objective)

        # Add agents to coalition
        self._add_agents_to_coalition(
            coalition, agent_scores, objective, max_coalition_size, assigned_agents
        )

        return coalition

    def _score_agents_for_objective(
        self, candidate_agents: List[AgentProfile], objective: CoalitionObjective
    ) -> List[Tuple[AgentProfile, float]]:
        """Score agents based on their suitability for an objective."""
        agent_scores = []
        required_caps = set(objective.required_capabilities)

        for agent in candidate_agents:
            agent_caps = set(agent.capabilities)
            capability_match = self._calculate_capability_match(agent_caps, required_caps)
            score = capability_match * agent.reputation * agent.capacity
            agent_scores.append((agent, score))

        # Sort by score (highest first)
        agent_scores.sort(key=lambda x: x[1], reverse=True)
        return agent_scores

    def _add_agents_to_coalition(
        self,
        coalition: Coalition,
        agent_scores: List[Tuple[AgentProfile, float]],
        objective: CoalitionObjective,
        max_coalition_size: int,
        assigned_agents: Set[str],
    ) -> None:
        """Add suitable agents to the coalition."""
        required_capabilities = set(objective.required_capabilities)
        covered_capabilities: Set[str] = set()

        for agent, score in agent_scores:
            if self._should_stop_adding_agents(
                coalition, max_coalition_size, covered_capabilities, required_capabilities
            ):
                break

            if self._can_agent_join(agent, covered_capabilities, required_capabilities, coalition):
                self._add_agent_to_coalition(
                    agent, coalition, assigned_agents, covered_capabilities
                )

    def _should_stop_adding_agents(
        self,
        coalition: Coalition,
        max_coalition_size: int,
        covered_capabilities: Set[str],
        required_capabilities: Set[str],
    ) -> bool:
        """Check if we should stop adding agents to the coalition."""
        if len(coalition.members) >= max_coalition_size:
            return True
        if required_capabilities.issubset(covered_capabilities):
            return True
        return False

    def _can_agent_join(
        self,
        agent: AgentProfile,
        covered_capabilities: Set[str],
        required_capabilities: Set[str],
        coalition: Coalition,
    ) -> bool:
        """Check if an agent can join the coalition."""
        if len(agent.current_coalitions) >= agent.max_coalitions:
            return False

        agent_caps = set(agent.capabilities)
        new_capabilities = agent_caps - covered_capabilities

        # Add agent if they bring new capabilities or if coalition is empty
        return (
            new_capabilities and not required_capabilities.issubset(covered_capabilities)
        ) or len(coalition.members) == 0

    def _add_agent_to_coalition(
        self,
        agent: AgentProfile,
        coalition: Coalition,
        assigned_agents: Set[str],
        covered_capabilities: Set[str],
    ) -> None:
        """Add an agent to the coalition."""
        role = CoalitionRole.LEADER if len(coalition.members) == 0 else CoalitionRole.MEMBER

        if coalition.add_member(agent.agent_id, role, agent.capabilities):
            assigned_agents.add(agent.agent_id)
            agent.current_coalitions.append(coalition.coalition_id)
            covered_capabilities.update(agent.capabilities)

    def _create_greedy_formation_result(
        self,
        coalitions: List[Coalition],
        objectives: List[CoalitionObjective],
        agents: List[AgentProfile],
        assigned_agents: Set[str],
    ) -> FormationResult:
        """Create the formation result with calculated metrics."""
        objective_coverage = self._calculate_greedy_objective_coverage(coalitions, objectives)
        formation_score = objective_coverage * 0.8  # Simplified score based on coverage
        unassigned_objectives = self._get_unassigned_objectives(coalitions, objectives)

        return FormationResult(
            coalitions=coalitions,
            unassigned_objectives=unassigned_objectives,
            formation_quality=formation_score,
            objective_coverage=objective_coverage,
            agent_utilization=len(assigned_agents) / len(agents) if agents else 0.0,
        )

    def _calculate_greedy_objective_coverage(
        self, coalitions: List[Coalition], objectives: List[CoalitionObjective]
    ) -> float:
        """Calculate objective coverage for greedy formation."""
        if not objectives:
            return 1.0

        achievable_objectives = 0
        for coalition in coalitions:
            for objective in coalition.objectives:
                coalition_caps = coalition.get_capabilities()
                if objective.required_capabilities.issubset(coalition_caps):
                    achievable_objectives += 1

        return achievable_objectives / len(objectives)

    def _get_unassigned_objectives(
        self, coalitions: List[Coalition], objectives: List[CoalitionObjective]
    ) -> List[CoalitionObjective]:
        """Get objectives not assigned to any coalition."""
        assigned_objective_ids = set()
        for coalition in coalitions:
            for obj in coalition.objectives:
                assigned_objective_ids.add(obj.objective_id)

        return [obj for obj in objectives if obj.objective_id not in assigned_objective_ids]


class OptimalFormation(FormationStrategy):
    """Optimal coalition formation using exhaustive search (for small problems)."""

    def __init__(self, max_search_size: int = 10):
        super().__init__("Optimal Formation")
        self.max_search_size = max_search_size

    def form_coalitions(
        self,
        agents: List[AgentProfile],
        objectives: List[CoalitionObjective],
        constraints: Optional[Dict[str, Any]] = None,
    ) -> FormationResult:
        """
        Form coalitions using exhaustive search for optimal solution.
        """
        time.time()

        # Check if problem is too large
        if self._should_fallback_to_greedy(agents):
            return self._fallback_to_greedy(agents, objectives, constraints)

        constraints = constraints or {}
        max_coalition_size = constraints.get("max_coalition_size", len(agents))

        # Find best formation
        best_formation = self._find_best_formation(agents, objectives, max_coalition_size)

        # Calculate result metrics
        return self._create_formation_result(best_formation, objectives, agents)

    def _should_fallback_to_greedy(self, agents: List[AgentProfile]) -> bool:
        """Check if problem is too large for exhaustive search."""
        if len(agents) > self.max_search_size:
            logger.warning("Problem too large for optimal search, falling back to greedy")
            return True
        return False

    def _fallback_to_greedy(
        self,
        agents: List[AgentProfile],
        objectives: List[CoalitionObjective],
        constraints: Optional[Dict[str, Any]],
    ) -> FormationResult:
        """Fallback to greedy strategy for large problems."""
        greedy = GreedyFormation()
        return greedy.form_coalitions(agents, objectives, constraints)

    def _find_best_formation(
        self,
        agents: List[AgentProfile],
        objectives: List[CoalitionObjective],
        max_coalition_size: int,
    ) -> List[Coalition]:
        """Find the best coalition formation through exhaustive search."""
        best_formation = None
        best_score = -1.0

        for assignment in self._generate_assignments(agents, objectives, max_coalition_size):
            coalitions = self._create_coalitions_from_assignment(assignment, objectives)
            score = self._evaluate_formation(coalitions, objectives)

            if score > best_score:
                best_score = score
                best_formation = coalitions

        return best_formation if best_formation is not None else []

    def _create_formation_result(
        self,
        coalitions: List[Coalition],
        objectives: List[CoalitionObjective],
        agents: List[AgentProfile],
    ) -> FormationResult:
        """Create formation result with calculated metrics."""
        assigned_agents = self._get_assigned_agents(coalitions)
        objective_coverage = self._calculate_objective_coverage(coalitions, objectives)
        unassigned_objectives = self._get_unassigned_objectives(coalitions, objectives)
        formation_quality = objective_coverage * 0.9  # Quality based on coverage

        return FormationResult(
            coalitions=coalitions,
            unassigned_objectives=unassigned_objectives,
            formation_quality=formation_quality,
            objective_coverage=objective_coverage,
            agent_utilization=len(assigned_agents) / len(agents) if agents else 0.0,
        )

    def _get_assigned_agents(self, coalitions: List[Coalition]) -> Set[str]:
        """Get set of all assigned agents."""
        assigned_agents = set()
        for coalition in coalitions:
            assigned_agents.update(coalition.members.keys())
        return assigned_agents

    def _calculate_objective_coverage(
        self, coalitions: List[Coalition], objectives: List[CoalitionObjective]
    ) -> float:
        """Calculate how many objectives can be achieved."""
        if not objectives:
            return 1.0

        achievable_objectives = 0
        for coalition in coalitions:
            for objective in coalition.objectives:
                if coalition.can_achieve_objective(objective):
                    achievable_objectives += 1

        return achievable_objectives / len(objectives)

    def _get_unassigned_objectives(
        self, coalitions: List[Coalition], objectives: List[CoalitionObjective]
    ) -> List[CoalitionObjective]:
        """Get objectives not assigned to any coalition."""
        assigned_objective_ids = set()
        for coalition in coalitions:
            for obj in coalition.objectives:
                assigned_objective_ids.add(obj.objective_id)

        return [obj for obj in objectives if obj.objective_id not in assigned_objective_ids]

class HierarchicalFormation(FormationStrategy):
    """Hierarchical coalition formation strategy."""

    def __init__(self):
        super().__init__("Hierarchical Formation")

    def form_coalitions(
        self,
        agents: List[AgentProfile],
        objectives: List[CoalitionObjective],
        constraints: Optional[Dict[str, Any]] = None,
    ) -> FormationResult:
        """Form coalitions using hierarchical strategy."""
        time.time()
        constraints = constraints or {}
        max_coalition_size = constraints.get("max_coalition_size", 15)
        min_subgroup_size = constraints.get("min_subgroup_size", 3)

        coalitions = []
        assigned_agents = set()

        # Group objectives by similarity/dependencies
        objective_groups = self._group_objectives(objectives)

        for group_idx, obj_group in enumerate(objective_groups):
            coalition = self._create_coalition_for_group(group_idx, obj_group, max_coalition_size)
            # Add objectives to coalition
            for obj in obj_group:
                coalition.add_objective(obj)

            # Find and assign coordinator
            coordinator = self._find_coordinator(agents, assigned_agents)
            if not coordinator:
                continue
            self._assign_coordinator(coalition, coordinator, assigned_agents, group_idx)

            # Add specialists for each objective
            self._add_specialists_to_coalition(
                coalition,
                obj_group,
                agents,
                assigned_agents,
                max_coalition_size,
                min_subgroup_size,
                coordinator,
            )

            # Promote coordinators for large coalitions
            self._promote_coordinators_if_needed(coalition)

            if coalition.members:
                coalition.activate()
                coalitions.append(coalition)
                logger.info(
                    f"Formed hierarchical coalition {coalition.coalition_id} with {len(coalition.members)} members"
                )

        return self._calculate_formation_result(coalitions, objectives, agents, assigned_agents)

    def _create_coalition_for_group(
        self, group_idx: int, obj_group: List[CoalitionObjective], max_size: int
    ) -> Coalition:
        """Create a coalition for an objective group."""
        coalition_id = f"hierarchical_coalition_{group_idx + 1}"
        return Coalition(
            coalition_id=coalition_id,
            name=f"Hierarchical Coalition {group_idx + 1}",
            objective=obj_group[0].description if obj_group else "Multi-objective",
            max_capacity=max_size,
        )

    def _find_coordinator(
        self, agents: List[AgentProfile], assigned_agents: set
    ) -> Optional[AgentProfile]:
        """Find the best coordinator from available agents."""
        available_agents = [a for a in agents if a.agent_id not in assigned_agents]
        if not available_agents:
            return None
        return max(available_agents, key=lambda a: a.reputation * len(a.capabilities))

    def _assign_coordinator(
        self, coalition: Coalition, coordinator: AgentProfile, assigned_agents: set, group_idx: int
    ):
        """Assign coordinator to coalition."""
        coalition.add_member(coordinator.agent_id, CoalitionRole.LEADER, coordinator.capabilities)
        assigned_agents.add(coordinator.agent_id)
        coordinator.current_coalitions.append(coalition.coalition_id)

    def _add_specialists_to_coalition(
        self,
        coalition: Coalition,
        obj_group: List[CoalitionObjective],
        agents: List[AgentProfile],
        assigned_agents: set,
        max_coalition_size: int,
        min_subgroup_size: int,
        coordinator: AgentProfile,
    ):
        """Add specialists to coalition for each objective."""
        for objective in obj_group:
            required_caps = set(objective.required_capabilities)
            covered_caps = set(coordinator.capabilities)
            specialist_candidates = self._find_specialist_candidates(
                agents, assigned_agents, required_caps
            )
            self._add_selected_specialists(
                coalition,
                specialist_candidates,
                assigned_agents,
                max_coalition_size,
                min_subgroup_size,
                covered_caps,
                required_caps,
            )

    def _find_specialist_candidates(
        self, agents: List[AgentProfile], assigned_agents: set, required_caps: set
    ) -> List[Tuple[AgentProfile, int]]:
        """Find candidates who can contribute to the objective."""
        specialist_candidates = []
        for agent in agents:
            if agent.agent_id in assigned_agents:
                continue

            agent_caps = set(agent.capabilities)
            contribution = len(agent_caps.intersection(required_caps))

            if contribution > 0:
                specialist_candidates.append((agent, contribution))

        # Sort by contribution
        specialist_candidates.sort(key=lambda x: x[1], reverse=True)
        return specialist_candidates

    def _add_selected_specialists(
        self,
        coalition: Coalition,
        specialist_candidates: List[Tuple[AgentProfile, int]],
        assigned_agents: set,
        max_coalition_size: int,
        min_subgroup_size: int,
        covered_caps: set,
        required_caps: set,
    ):
        """Add selected specialists to the coalition."""
        added_count = 0
        for agent, contribution in specialist_candidates:
            if self._should_stop_adding_specialists(
                coalition, added_count, max_coalition_size, min_subgroup_size, agent
            ):
                break

            coalition.add_member(agent.agent_id, CoalitionRole.MEMBER, agent.capabilities)
            assigned_agents.add(agent.agent_id)
            agent.current_coalitions.append(coalition.coalition_id)
            added_count += 1

            # Update covered capabilities
            covered_caps.update(agent.capabilities)

            # Stop if we have all required capabilities:
            if required_caps.issubset(covered_caps):
                break

    def _should_stop_adding_specialists(
        self,
        coalition: Coalition,
        added_count: int,
        max_coalition_size: int,
        min_subgroup_size: int,
        agent: AgentProfile,
    ) -> bool:
        """Check if we should stop adding specialists."""
        if len(coalition.members) >= max_coalition_size or added_count >= min_subgroup_size:
            return True
        if len(agent.current_coalitions) >= agent.max_coalitions:
            return True
        return False

    def _promote_coordinators_if_needed(self, coalition: Coalition):
        """Promote some members to coordinators for large coalitions."""
        if len(coalition.members) > 6:
            # Promote some members to coordinators
            members = list(coalition.members.values())
            members.sort(key=lambda m: m.contribution_score, reverse=True)

            for i in range(min(2, len(members) // 3)):
                if members[i].role == CoalitionRole.MEMBER:
                    members[i].role = CoalitionRole.COORDINATOR

    def _calculate_formation_result(
        self,
        coalitions: List[Coalition],
        objectives: List[CoalitionObjective],
        agents: List[AgentProfile],
        assigned_agents: set,
    ) -> FormationResult:
        """Calculate the final formation result."""
        # Calculate metrics
        achievable_objectives = self._count_achievable_objectives(coalitions)
        objective_coverage = achievable_objectives / len(objectives) if objectives else 1.0
        formation_score = objective_coverage * 0.9  # Hierarchical bonus
        unassigned_objectives = self._get_unassigned_objectives(coalitions, objectives)

        return FormationResult(
            coalitions=coalitions,
            unassigned_objectives=unassigned_objectives,
            formation_quality=formation_score,
            objective_coverage=objective_coverage,
            agent_utilization=len(assigned_agents) / len(agents) if agents else 0.0,
        )

    def _count_achievable_objectives(self, coalitions: List[Coalition]) -> int:
        """Count objectives that can be achieved by coalitions."""
        achievable_objectives = 0
        for coalition in coalitions:
            for objective in coalition.objectives:
                # Assume objective is achievable if coalition has required capabilities
                coalition_caps = coalition.get_capabilities()
                if objective.required_capabilities.issubset(coalition_caps):
                    achievable_objectives += 1
        return achievable_objectives

    def _get_unassigned_objectives(
        self, coalitions: List[Coalition], objectives: List[CoalitionObjective]
    ) -> List[CoalitionObjective]:
        """Get objectives that weren't assigned to any coalition."""
        assigned_objective_ids = set()
        for coalition in coalitions:
            for obj in coalition.objectives:
                assigned_objective_ids.add(obj.objective_id)

        return [obj for obj in objectives if obj.objective_id not in assigned_objective_ids]

    def _group_objectives(
        self, objectives: List[CoalitionObjective]
    ) -> List[List[CoalitionObjective]]:
        """Group objectives by similarity or dependencies."""
        # Simple grouping - each objective in its own group for now
        return [[obj] for obj in objectives]
