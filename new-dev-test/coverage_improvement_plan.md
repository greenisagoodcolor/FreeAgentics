        "password": "invalid"
    })

    assert "detail" in response.json()

    """Test token validation endpoint"""

        "username": "testuser",
        "password": "testpass"
    })
    token = login_response.json()["access_token"]

    # Then validate it
    response = client.get("/v1/auth/validate", headers={
        "Authorization": f"Bearer {token}"
